25 /**
26  * A reader that produces an instance of `T` from a subtype of [[BSONValue]].
27  */
28 trait BSONReader[B <: BSONValue, T] { self =>
29 
30   /**
31    * Reads a BSON value and produce an instance of `T`.
32    *
33    * This method may throw exceptions at runtime.
34    * If used outside a reader, one should consider `readTry(bson: B): Try[T]` or `readOpt(bson: B): Option[T]`.
35    */
36   def read(bson: B): T
37 
38   /** Tries to produce an instance of `T` from the `bson` value, returns `None` if an error occurred. */
39   def readOpt(bson: B): Option[T] = readTry(bson).toOption
40 
41   /** Tries to produce an instance of `T` from the `bson` value. */
42   def readTry(bson: B): Try[T] = Try(read(bson))
43 
44   /**
45    * Returns a BSON reader that returns the result of applying `f`
46    * on the result of this reader.
47    *
48    * @param f the function to apply
49    */
50   final def afterRead[U](f: T => U): BSONReader[B, U] =
51     BSONReader[B, U]((read _) andThen f)
52 
53   private[reactivemongo] def widenReader[U >: T]: UnsafeBSONReader[U] =
54     new UnsafeBSONReader[U] {
55       def readTry(value: BSONValue): Try[U] =
56         Try(value.asInstanceOf[B]) match {
57           case Failure(_) => Failure(exceptions.TypeDoesNotMatch(
58             s"Cannot convert $value: ${value.getClass} with ${self.getClass}"))
59 
60           case Success(bson) => self.readTry(bson)
61         }
62     }
63 }

77  */
78 trait BSONWriter[T, B <: BSONValue] {
79   /**
80    * Writes an instance of `T` as a BSON value.
81    *
82    * This method may throw exceptions at runtime.
83    * If used outside a reader, one should consider `writeTry(bson: B): Try[T]` or `writeOpt(bson: B): Option[T]`.
84    */
85   def write(t: T): B
86 
87   /** Tries to produce a BSON value from an instance of `T`, returns `None` if an error occurred. */
88   def writeOpt(t: T): Option[B] = writeTry(t).toOption
89 
90   /** Tries to produce a BSON value from an instance of `T`. */
91   def writeTry(t: T): Try[B] = Try(write(t))
92 
93   /**
94    * Returns a BSON writer that returns the result of applying `f`
95    * on the BSON value from this writer.
96    *
97    * @param f the function to apply
98    */
99   final def afterWrite[U <: BSONValue](f: B => U): BSONWriter[T, U] =
100     BSONWriter[T, U]((write _) andThen f)
101 
102   final def beforeWrite[U](f: U => T): BSONWriter[U, B] =
103     BSONWriter[U, B](f andThen (write _))
104 }

196 object BSONHandler {
197   private[bson] class MappedHandler[B <: BSONValue, T, U](
198       parent: BSONHandler[B, T],
199       to: T => U,
200       from: U => T) extends BSONHandler[B, U] {
201     def write(u: U) = parent.write(from(u))
202     def read(b: B) = to(parent.read(b))
203   }
204 
205   private[bson] class DefaultHandler[B <: BSONValue, T](r: B => T, w: T => B)
206       extends BSONHandler[B, T] {
207     def read(x: B): T = r(x)
208     def write(x: T): B = w(x)
209   }
210 
211   def apply[B <: BSONValue, T](read: B => T, write: T => B): BSONHandler[B, T] =
212     new DefaultHandler(read, write)
213 }

311   abstract class IdentityBSONConverter[T <: BSONValue](implicit m: Manifest[T]) extends BSONReader[T, T] with BSONWriter[T, T] {
312     override def write(t: T): T = m.runtimeClass.cast(t).asInstanceOf[T]
313     override def writeOpt(t: T): Option[T] = if (m.runtimeClass.isInstance(t)) Some(t.asInstanceOf[T]) else None
314     override def read(bson: T): T = m.runtimeClass.cast(bson).asInstanceOf[T]
315     override def readOpt(bson: T): Option[T] = if (m.runtimeClass.isInstance(bson)) Some(bson.asInstanceOf[T]) else None
316   }
