248   /** Concats two array - fast way */
249   private def concat[T](a1: Array[T], a2: Array[T])(implicit m: Manifest[T]): Array[T] = {
250     var i, j = 0
251     val result = new Array[T](a1.length + a2.length)
252     while (i < a1.length) {
253       result(i) = a1(i)
254       i = i + 1
255     }
256     while (j < a2.length) {
257       result(i + j) = a2(j)
258       j = j + 1
259     }
260     result
261   }

263   /**
264    * Gets an `Iteratee` that will consume data to put into a GridFS store.
265    * @param file Metadata of the file to store.
266    * @param chunkSize Size of the chunks. Defaults to 256kB.
267    *
268    * @tparam Id Type of the id of this file (generally `BSONObjectID` or `BSONValue`).
269    *
270    * @return An `Iteratee` that will consume data to put into a GridFS store.
271    */
272   @deprecated("Will be moved to the separate iteratee module", "0.12.0")
273   def iteratee[Id <: pack.Value](file: FileToSave[pack.type, Id], chunkSize: Int = 262144)(implicit readFileReader: pack.Reader[ReadFile[Id]], ctx: ExecutionContext, idProducer: IdProducer[Id], docWriter: BSONDocumentWriter[file.pack.Document]): Iteratee[Array[Byte], Future[ReadFile[Id]]] = {
274     implicit def ec = db.connection.actorSystem
275     import java.security.MessageDigest
276 
277     case class Chunk(
278         previous: Array[Byte] = new Array(0),
279         n: Int = 0,
280         md: MessageDigest = MessageDigest.getInstance("MD5"),
281         length: Int = 0) {
282 
283       def feed(chunk: Array[Byte]): Future[Chunk] = {
284         val wholeChunk = self.concat(previous, chunk)
285 
286         val normalizedChunkNumber = wholeChunk.length / chunkSize
287 
288         logger.debug(s"wholeChunk size is ${wholeChunk.length} => ${normalizedChunkNumber}")
289 
290         val zipped =
291           for (i <- 0 until normalizedChunkNumber)
292             yield Arrays.copyOfRange(
293             wholeChunk, i * chunkSize, (i + 1) * chunkSize) -> i
294 
295         val left = Arrays.copyOfRange(
296           wholeChunk, normalizedChunkNumber * chunkSize, wholeChunk.length)
297 
298         Future.traverse(zipped) { ci =>
299           writeChunk(n + ci._2, ci._1)
300         }.map { _ =>
301           logger.debug("all futures for the last given chunk are redeemed.")
302           Chunk(
303             if (left.isEmpty) Array.empty else left,
304             n + normalizedChunkNumber,
305             md, //{ md.update(chunk) ; md },
306             length + chunk.length)
307         }
308       }
309 
310       import reactivemongo.api.collections.bson.{
311         BSONCollection,
312         BSONCollectionProducer
313       }
314 
315       def finish(): Future[ReadFile[Id]] = {
316         logger.debug(s"writing last chunk (n=$n)!")
317 
318         val uploadDate = file.uploadDate.getOrElse(System.currentTimeMillis)
319 
320         writeChunk(n, previous).flatMap { f =>
321           val bson = BSONDocument(idProducer("_id" -> file.id)) ++ (
322             "filename" -> file.filename.map(BSONString(_)),
323             "chunkSize" -> BSONInteger(chunkSize),
324             "length" -> BSONLong(length),
325             "uploadDate" -> BSONDateTime(uploadDate),
326             "contentType" -> file.contentType.map(BSONString(_)),
327             "metadata" -> option(!pack.isEmpty(file.metadata), file.metadata))
328 
329           files.as[BSONCollection]().insert(bson).map { _ =>
330             val buf = ChannelBufferWritableBuffer()
331             BSONSerializationPack.writeToBuffer(buf, bson)
332             pack.readAndDeserialize(buf.toReadableBuffer, readFileReader)
333           }
334         }
335       }
336 
337       def writeChunk(n: Int, array: Array[Byte]) = {
338         logger.debug(s"writing chunk $n")
339 
340         val bson = BSONDocument(
341           "files_id" -> file.id,
342           "n" -> BSONInteger(n),
343           "data" -> BSONBinary(array, Subtype.GenericBinarySubtype))
344 
345         chunks.as[BSONCollection]().insert(bson)
346       }
347     }
348 
349     Iteratee.foldM(Chunk()) { (previous, chunk: Array[Byte]) =>
350       logger.debug(s"processing new enumerated chunk from n=${previous.n}...\n")
351       previous.feed(chunk)
352     }.map(_.finish)
353   }

355   /**
356    * Produces an enumerator of chunks of bytes from the `chunks` collection
357    * matching the given file metadata.
358    *
359    * @param file the file to be read
360    */
361   def enumerate[Id <: pack.Value](file: ReadFile[Id])(implicit ctx: ExecutionContext, idProducer: IdProducer[Id]): Enumerator[Array[Byte]] = {
362     import reactivemongo.api.collections.bson.{
363       BSONCollection,
364       BSONCollectionProducer
365     }
366 
367     def selector = BSONDocument(idProducer("files_id" -> file.id)) ++ (
368       "n" -> BSONDocument(
369         "$gte" -> 0,
370         "$lte" -> BSONLong(file.length / file.chunkSize + (
371           if (file.length % file.chunkSize > 0) 1 else 0))))
372 
373     @inline def cursor = chunks.as[BSONCollection]().find(selector).
374       sort(BSONDocument("n" -> 1)).cursor[BSONDocument](defaultReadPreference)
375 
376     @inline def pushChunk(chan: Concurrent.Channel[Array[Byte]], doc: BSONDocument): Cursor.State[Unit] = doc.get("data") match {
377       case Some(BSONBinary(data, _)) => {
378         val array = new Array[Byte](data.readable)
379         data.slice(data.readable).readBytes(array)
380         Cursor.Cont(chan push array)
381       }
382 
383       case _ => {
384         val errmsg = "not a chunk! failed assertion: data field is missing: ${BSONDocument pretty doc}"
385 
386         logger.error(errmsg)
387         Cursor.Fail(ReactiveMongoException(errmsg))
388       }
389     }
390 
391     Concurrent.unicast[Array[Byte]] { chan =>
392       cursor.foldWhile({})(
393         (_, doc) => pushChunk(chan, doc),
394         Cursor.FailOnError()).onComplete { case _ => chan.eofAndEnd() }
395     }
396   }
