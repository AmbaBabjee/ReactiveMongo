323   /**
324    * Inserts a document into the collection and wait for the [[reactivemongo.api.commands.WriteResult]].
325    *
326    * Please read the documentation about [[reactivemongo.core.commands.GetLastError]] to know how to use it properly.
327    *
328    * @tparam T the type of the document to insert. An implicit `Writer[T]` typeclass for handling it has to be in the scope.
329    *
330    * @param document the document to insert.
331    * @param writeConcern the [[reactivemongo.core.commands.GetLastError]] command message to send in order to control how the document is inserted. Defaults to GetLastError().
332    *
333    * @return a future [[reactivemongo.api.commands.WriteResult]] that can be used to check whether the insertion was successful.
334    */
335   def insert[T](document: T, writeConcern: WriteConcern = defaultWriteConcern)(implicit writer: pack.Writer[T], ec: ExecutionContext): Future[WriteResult] =
336     db.connection.metadata match {
337       case Some(metadata) if metadata.maxWireVersion >= MongoWireVersion.V26 =>
338         Failover2(db.connection, failoverStrategy) { () =>
339           runCommand(BatchCommands.InsertCommand.Insert(
340             writeConcern = writeConcern)(document)).flatMap { wr =>
341             val flattened = wr.flatten
342             if (!flattened.ok) {
343               // was ordered, with one doc => fail if has an error
344               Future.failed(WriteResult.lastError(flattened).
345                 getOrElse[Exception](GenericDriverException(
346                   s"fails to insert: $document")))
347 
348             } else Future.successful(wr)
349           }
350         }.future
351 
352       case Some(_) => { // Mongo < 2.6 // TODO: Deprecates/remove
353         val op = Insert(0, fullCollectionName)
354         val bson = writeDoc(document, writer)
355         val checkedWriteRequest = CheckedWriteRequest(op, BufferSequence(bson), writeConcern)
356 
357         Failover2(db.connection, failoverStrategy) { () =>
358           db.connection.sendExpectingResponse(checkedWriteRequest).
359             map(pack.readAndDeserialize(_, LastErrorReader))
360         }.future
361       }
362 
363       case None =>
364         Future.failed(ConnectionNotInitialized.MissingMetadata)
365     }

367   /**
368    * Updates one or more documents matching the given selector with the given modifier or update object.
369    *
370    * @tparam S the type of the selector object. An implicit `Writer[S]` typeclass for handling it has to be in the scope.
371    * @tparam U the type of the modifier or update object. An implicit `Writer[U]` typeclass for handling it has to be in the scope.
372    *
373    * @param selector the selector object, for finding the documents to update.
374    * @param update the modifier object (with special keys like \$set) or replacement object.
375    * @param writeConcern the [[reactivemongo.core.commands.GetLastError]] command message to send in order to control how the documents are updated. Defaults to GetLastError().
376    * @param upsert states whether the update objet should be inserted if no match found. Defaults to false.
377    * @param multi states whether the update may be done on all the matching documents.
378    *
379    * @return a future [[reactivemongo.api.commands.WriteResult]] that can be used to check whether the update was successful.
380    */
381   def update[S, U](selector: S, update: U, writeConcern: WriteConcern = defaultWriteConcern, upsert: Boolean = false, multi: Boolean = false)(implicit selectorWriter: pack.Writer[S], updateWriter: pack.Writer[U], ec: ExecutionContext): Future[UpdateWriteResult] = db.connection.metadata match {
382     case Some(metadata) if (
383       metadata.maxWireVersion >= MongoWireVersion.V26) => {
384       import BatchCommands.UpdateCommand.{ Update, UpdateElement }
385 
386       Failover2(db.connection, failoverStrategy) { () =>
387         runCommand(Update(writeConcern = writeConcern)(
388           UpdateElement(selector, update, upsert, multi))).flatMap { wr =>
389           val flattened = wr.flatten
390           if (!flattened.ok) {
391             // was ordered, with one doc => fail if has an error
392             Future.failed(WriteResult.lastError(flattened).
393               getOrElse[Exception](GenericDriverException(
394                 s"fails to update: $update")))
395 
396           } else Future.successful(wr)
397         }
398       }.future
399     }
400 
401     case Some(_) => { // Mongo < 2.6 // TODO: Deprecate/remove
402       val flags = 0 | (if (upsert) UpdateFlags.Upsert else 0) | (if (multi) UpdateFlags.MultiUpdate else 0)
403       val op = Update(fullCollectionName, flags)
404       val bson = writeDoc(selector, selectorWriter)
405       bson.writeBytes(writeDoc(update, updateWriter))
406       val checkedWriteRequest =
407         CheckedWriteRequest(op, BufferSequence(bson), writeConcern)
408 
409       Failover2(db.connection, failoverStrategy) { () =>
410         db.connection.sendExpectingResponse(checkedWriteRequest).map { r =>
411           val res = pack.readAndDeserialize(r, LastErrorReader)
412           UpdateWriteResult(res.ok, res.n, res.n,
413             res.upserted.map(Upserted(-1, _)).toSeq,
414             Nil, None, res.code, res.errmsg)
415 
416         }
417       }.future
418     }
419 
420     case None => Future.failed(ConnectionNotInitialized.MissingMetadata)
421   }

504   /**
505    * [[http://docs.mongodb.org/manual/reference/command/aggregate/ Aggregates]] the matching documents.
506    *
507    * {{{
508    * import scala.concurrent.Future
509    * import scala.concurrent.ExecutionContext.Implicits.global
510    *
511    * import reactivemongo.bson._
512    * import reactivemongo.api.collections.bson.BSONCollection
513    *
514    * def populatedStates(cities: BSONCollection): Future[List[BSONDocument]] = {
515    *   import cities.BatchCommands.AggregationFramework
516    *   import AggregationFramework.{ Group, Match, SumField }
517    *
518    *   cities.aggregate(Group(BSONString("\$state"))(
519    *     "totalPop" -> SumField("population")), List(
520    *       Match(document("totalPop" ->
521    *         document("\$gte" -> 10000000L))))).map(_.documents)
522    * }
523    * }}}
