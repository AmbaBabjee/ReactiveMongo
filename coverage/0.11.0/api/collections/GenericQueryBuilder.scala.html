16 package reactivemongo.api.collections

18 import scala.concurrent.{ ExecutionContext, Future }
19 import reactivemongo.api._
20 import reactivemongo.bson.buffer.ReadableBuffer
21 import reactivemongo.core.protocol.{ Query, QueryFlags }
22 import reactivemongo.core.netty.{ BufferSequence, ChannelBufferWritableBuffer }

24 /**
25  * A builder that helps to make a fine-tuned query to MongoDB.
26  *
27  * When the query is ready, you can call `cursor` to get a [[Cursor]], or `one` if you want to retrieve just one document.
28  *
29  */
30 trait GenericQueryBuilder[P <: SerializationPack] {
31   val pack: P
32   type Self <: GenericQueryBuilder[pack.type]
33 
34   def queryOption: Option[pack.Document]
35   def sortOption: Option[pack.Document]
36   def projectionOption: Option[pack.Document]
37   def hintOption: Option[pack.Document]
38   def explainFlag: Boolean
39   def snapshotFlag: Boolean
40   def commentString: Option[String]
41   def options: QueryOpts
42   def failover: FailoverStrategy
43   def collection: Collection
44   def maxTimeMsOption: Option[Long]
45 
46   def merge(readPreference: ReadPreference): pack.Document
47 
48   def copy(
49     queryOption: Option[pack.Document] = queryOption,
50     sortOption: Option[pack.Document] = sortOption,
51     projectionOption: Option[pack.Document] = projectionOption,
52     hintOption: Option[pack.Document] = hintOption,
53     explainFlag: Boolean = explainFlag,
54     snapshotFlag: Boolean = snapshotFlag,
55     commentString: Option[String] = commentString,
56     options: QueryOpts = options,
57     failover: FailoverStrategy = failover,
58     maxTimeMsOption: Option[Long] = maxTimeMsOption): Self
59 
60   private def write(document: pack.Document, buffer: ChannelBufferWritableBuffer = ChannelBufferWritableBuffer()): ChannelBufferWritableBuffer = {
61     pack.writeToBuffer(buffer, document)
62     buffer
63   }
64 
65   /**
66    * Sends this query and gets a [[Cursor]] of instances of `T`.
67    *
68    * An implicit `Reader[T]` must be present in the scope.
69    */
70   @deprecated(message = "Use `cursor` function with read preference.",
71     since = "0.11.0")
72   def cursor[T](implicit reader: pack.Reader[T], ec: ExecutionContext, cp: CursorProducer[T]): cp.ProducedCursor = cursor(ReadPreference.primary)
73 
74   /**
75    * Makes a [[Cursor]] of this query, which can be enumerated.
76    *
77    * An implicit `Reader[T]` must be present in the scope.
78    *
79    * @param readPreference The ReadPreference for this request. If the ReadPreference implies that this request might be run on a Secondary, the slaveOk flag will be set.
80    */
81   def cursor[T](readPreference: ReadPreference = ReadPreference.primary, isMongo26WriteOp: Boolean = false)(implicit reader: pack.Reader[T], ec: ExecutionContext, cp: CursorProducer[T]): cp.ProducedCursor = cp.produce(defaultCursor[T](readPreference, isMongo26WriteOp))
82 
83   private def defaultCursor[T](readPreference: ReadPreference, isMongo26WriteOp: Boolean = false)(implicit reader: pack.Reader[T], ec: ExecutionContext): Cursor[T] = {
84     val documents = BufferSequence {
85       val buffer = write(merge(readPreference), ChannelBufferWritableBuffer())
86       projectionOption.map { projection =>
87         write(projection, buffer)
88       }.getOrElse(buffer).buffer
89     }
90 
91     val flags = if (readPreference.slaveOk) options.flagsN | QueryFlags.SlaveOk else options.flagsN
92 
93     val op = Query(flags, collection.fullCollectionName, options.skipN, options.batchSizeN)
94 
95     DefaultCursor.query(pack, op, documents, readPreference, collection.db.connection, failover, isMongo26WriteOp)(reader)
96   }
97 
98   /**
99    * Sends this query and gets a future `Option[T]`.
100    * Alias for `.cursor[T]().headOption`.
101    *
102    * An implicit `Reader[T]` must be present in the scope.
103    */
104   def one[T](implicit reader: pack.Reader[T], ec: ExecutionContext): Future[Option[T]] = one(ReadPreference.primary)
105 
106   /**
107    * Sends this query and gets a future `Option[T]`.
108    * Alias for `.cursor[T](readPreference).headOption`.
109    *
110    * An implicit `Reader[T]` must be present in the scope.
111    *
112    * @param readPreference The ReadPreference for this request. If the ReadPreference implies that this request might be run on a Secondary, the slaveOk flag will be set.
113    */
114   def one[T](readPreference: ReadPreference)(implicit reader: pack.Reader[T], ec: ExecutionContext): Future[Option[T]] = copy(options = options.batchSize(1)).defaultCursor(readPreference)(reader, ec).headOption
115 
116   /**
117    * Sets the query (the selector document).
118    *
119    * @tparam Qry The type of the query. An implicit `Writer[Qry]` typeclass for handling it has to be in the scope.
120    */
121   def query[Qry](selector: Qry)(implicit writer: pack.Writer[Qry]): Self =
122     copy(queryOption = Some(pack.serialize(selector, writer)))
123 
124   /** Sets the query (the selector document). */
125   def query(selector: pack.Document): Self = copy(queryOption = Some(selector))
126 
127   /** Sets the sorting document. */
128   def sort(document: pack.Document): Self = copy(sortOption = Some(document))
129 
130   def options(options: QueryOpts): Self = copy(options = options)
131 
132   /**
133    * Sets the projection document (for [[http://docs.mongodb.org/manual/core/read-operations-introduction/ retrieving only a subset of fields]]).
134    *
135    * @tparam Pjn The type of the projection. An implicit `Writer[Pjn]` typeclass for handling it has to be in the scope.
136    */
137   def projection[Pjn](p: Pjn)(implicit writer: pack.Writer[Pjn]): Self =
138     copy(projectionOption = Some(pack.serialize(p, writer)))
139 
140   def projection(p: pack.Document): Self = copy(projectionOption = Some(p))
141 
142   /** Sets the hint document (a document that declares the index MongoDB should use for this query). */
143   def hint(document: pack.Document): Self = copy(hintOption = Some(document))
144 
145   /** Sets the hint document (a document that declares the index MongoDB should use for this query). */
146   // TODO def hint(indexName: String): Self = copy(hintOption = Some(BSONDocument(indexName -> BSONInteger(1))))
147 
148   /** Toggles [[https://docs.mongodb.org/manual/reference/method/cursor.explain/#cursor.explain explain mode]]. */
149   def explain(flag: Boolean = true): Self = copy(explainFlag = flag)
150 
151   /** Toggles [[https://docs.mongodb.org/manual/faq/developers/#faq-developers-isolate-cursors snapshot mode]]. */
152   def snapshot(flag: Boolean = true): Self = copy(snapshotFlag = flag)
153 
154   /** Adds a comment to this query, that may appear in the MongoDB logs. */
155   def comment(message: String): Self = copy(commentString = Some(message))
156 
157   /** Adds maxTimeMs to query https://docs.mongodb.org/v3.0/reference/operator/meta/maxTimeMS/ */
158   def maxTimeMs(p: Long): Self = copy(maxTimeMsOption = Some(p))
159 
160 }
