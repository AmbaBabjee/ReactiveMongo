37 case class BSONQueryBuilder(
38   collection: Collection,
39   failover: FailoverStrategy,
40   queryOption: Option[BSONDocument] = None,
41   sortOption: Option[BSONDocument] = None,
42   projectionOption: Option[BSONDocument] = None,
43   hintOption: Option[BSONDocument] = None,
44   explainFlag: Boolean = false,
45   snapshotFlag: Boolean = false,
46   commentString: Option[String] = None,
47   options: QueryOpts = QueryOpts(),
48   maxTimeMsOption: Option[Long] = None)
49     extends GenericQueryBuilder[BSONSerializationPack.type] {
50   import reactivemongo.util.option
51 
52   type Self = BSONQueryBuilder
53   val pack = BSONSerializationPack
54 
55   def copy(
56     queryOption: Option[BSONDocument] = queryOption,
57     sortOption: Option[BSONDocument] = sortOption,
58     projectionOption: Option[BSONDocument] = projectionOption,
59     hintOption: Option[BSONDocument] = hintOption,
60     explainFlag: Boolean = explainFlag,
61     snapshotFlag: Boolean = snapshotFlag,
62     commentString: Option[String] = commentString,
63     options: QueryOpts = options,
64     failover: FailoverStrategy = failover,
65     maxTimeMsOption: Option[Long] = maxTimeMsOption): BSONQueryBuilder =
66     BSONQueryBuilder(collection, failover, queryOption, sortOption, projectionOption, hintOption, explainFlag, snapshotFlag, commentString, options, maxTimeMsOption)
67 
68   def merge(readPreference: ReadPreference): BSONDocument = {
69     // Primary and SecondaryPreferred are encoded as the slaveOk flag;
70     // the others are encoded as $readPreference field.
71 
72     def pref = {
73       val mode = readPreference match {
74         case ReadPreference.Primary                    => "primary"
75         case ReadPreference.PrimaryPreferred(filter)   => "primaryPreferred"
76         case ReadPreference.Secondary(filter)          => "secondary"
77         case ReadPreference.SecondaryPreferred(filter) => "secondaryPreferred"
78         case ReadPreference.Nearest(filter)            => "nearest"
79       }
80       val base = Seq[BSONElement]("mode" -> BSONString(mode))
81 
82       BSONDocument(readPreference match {
83         case ReadPreference.Taggable(tagSet) => base :+ ("tags" -> BSONArray(
84           tagSet.map(tags => BSONDocument(tags.toList.map {
85             case (k, v) => k -> BSONString(v)
86           }))))
87 
88         case _ => base
89       })
90     }
91 
92     val optional = List(
93       queryOption.map { "$query" -> _ },
94       sortOption.map { "$orderby" -> _ },
95       hintOption.map { "$hint" -> _ },
96       maxTimeMsOption.map { "$maxTimeMS" -> BSONLong(_) },
97       commentString.map { "$comment" -> BSONString(_) },
98       option(explainFlag, "$explain" -> BSONBoolean(true)),
99       option(snapshotFlag, "$snapshot" -> BSONBoolean(true))).flatten
100 
101     BSONDocument(optional :+ ("$readPreference" -> pref))
102   }
103 }
