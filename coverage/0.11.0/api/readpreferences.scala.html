5 /**
6  * MongoDB Read Preferences enable to read from primary or secondaries
7  * with a predefined strategy.
8  */
9 sealed trait ReadPreference {
10   def slaveOk: Boolean = true
11   def filterTag: Option[BSONDocument => Boolean]
12 }
13 
14 object ReadPreference {
15   /** Read only from the primary. This is the default choice. */
16   object Primary extends ReadPreference {
17     override def slaveOk = false
18     override def filterTag = None
19   }
20   /** Read from the primary if it is available, or secondaries if it is not. */
21   case class PrimaryPreferred(filterTag: Option[BSONDocument => Boolean]) extends ReadPreference
22   /** Read only from any secondary. */
23   case class Secondary(filterTag: Option[BSONDocument => Boolean]) extends ReadPreference
24   /** Read from any secondary, or from the primary if they are not available. */
25   case class SecondaryPreferred(filterTag: Option[BSONDocument => Boolean]) extends ReadPreference
26   /**
27    * Read from the faster node (ie the node which replies faster than all others), regardless its status
28    * (primary or secondary).
29    */
30   case class Nearest(filterTag: Option[BSONDocument => Boolean]) extends ReadPreference

32   private implicit class BSONDocumentWrapper(val underlying: BSONDocument) extends AnyVal {
33     def contains(doc: BSONDocument): Boolean = {
34 
35       val els = underlying.elements
36       doc.elements.forall { element =>
37         els.find {
38           case (name, value) => element._1 == name && ((element._2, value) match {
39             case (d1: BSONDocument, d2: BSONDocument) => d1.elements == d2.elements
40             case (a1: BSONArray, a2: BSONArray)       => a1.values == a2.values
41             case (v1, v2)                             => v1 == v2
42           })
43         }.isDefined
44       }
45     }
46   }
