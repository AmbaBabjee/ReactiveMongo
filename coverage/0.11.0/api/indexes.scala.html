24 /** Type of Index */
25 sealed trait IndexType {
26   /** Value of the index (`{fieldName: value}`). */
27   def value: BSONValue
28   private[indexes] def valueStr: String
29 }
30 
31 object IndexType {
32   object Ascending extends IndexType {
33     def value = BSONInteger(1)
34     def valueStr = "1"
35   }
36 
37   object Descending extends IndexType {
38     def value = BSONInteger(-1)
39     def valueStr = "-1"
40   }
41 
42   object Geo2D extends IndexType {
43     def value = BSONString("2d")
44     def valueStr = "2d"
45   }
46 
47   object Geo2DSpherical extends IndexType {
48     def value = BSONString("2dsphere")
49     def valueStr = "2dsphere"
50   }
51 
52   object GeoHaystack extends IndexType {
53     def value = BSONString("geoHaystack")
54     def valueStr = "geoHaystack"
55   }
56 
57   object Hashed extends IndexType {
58     def value = BSONString("hashed")
59     def valueStr = "hashed"
60   }
61 
62   object Text extends IndexType {
63     def value = BSONString("text")
64     def valueStr = "text"
65   }
66 
67   def apply(value: BSONValue) = value match {
68     case BSONInteger(i) if i > 0             => Ascending
69     case BSONInteger(i) if i < 0             => Descending
70     case BSONDouble(i) if i > 0              => Ascending
71     case BSONDouble(i) if i < 0              => Descending
72     case BSONLong(i) if i > 0                => Ascending
73     case BSONLong(i) if i < 0                => Descending
74     case BSONString(s) if s == "2d"          => Geo2D
75     case BSONString(s) if s == "2dsphere"    => Geo2DSpherical
76     case BSONString(s) if s == "geoHaystack" => GeoHaystack
77     case BSONString(s) if s == "hashed"      => Hashed
78     case BSONString(s) if s == "text"        => Text
79     case _                                   => throw new IllegalArgumentException("unsupported index type")
80   }
81 }

83 /**
84  * A MongoDB index (excluding the namespace).
85  *
86  * Consider reading [[http://www.mongodb.org/display/DOCS/Indexes the documentation about indexes in MongoDB]].
87  *
88  * @param key The index key (it can be composed of multiple fields). This list should not be empty!
89  * @param name The name of this index. If you provide none, a name will be computed for you.
90  * @param unique Enforces uniqueness.
91  * @param background States if this index should be built in background. You should read [[http://www.mongodb.org/display/DOCS/Indexes#Indexes-background%3Atrue the documentation about background indexing]] before using it.
92  * @param dropDups States if duplicates should be discarded (if unique = true). Warning: you should read [[http://www.mongodb.org/display/DOCS/Indexes#Indexes-dropDups%3Atrue the documentation]].
93  * @param sparse States if the index to build should only consider the documents that have the indexed fields. See [[http://www.mongodb.org/display/DOCS/Indexes#Indexes-sparse%3Atrue the documentation]] on the consequences of such an index.
94  * @param version Indicates the [[http://www.mongodb.org/display/DOCS/Index+Versions version]] of the index (1 for >= 2.0, else 0). You should let MongoDB decide.
95  * @param options Optional parameters for this index (typically specific to an IndexType like Geo2D).
96  * @param partialFilter Optional [[https://docs.mongodb.com/manual/core/index-partial/#partial-index-with-unique-constraints partial filter]] (since MongoDB 3.2)
97  */
98 case class Index(
99     key: Seq[(String, IndexType)],
100     name: Option[String] = None,
101     unique: Boolean = false,
102     background: Boolean = false,
103     dropDups: Boolean = false, // Deprecated since 2.6, TODO: Remove
104     sparse: Boolean = false,
105     version: Option[Int] = None, // let MongoDB decide
106     // TODO: storageEngine (new for Mongo3)
107     partialFilter: Option[BSONDocument] = None,
108     options: BSONDocument = BSONDocument()) {
109 
110   /** The name of the index (a default one is computed if none). */
111   lazy val eventualName: String = name.getOrElse(key.foldLeft("") { (name, kv) =>
112     name + (if (name.length > 0) "_" else "") + kv._1 + "_" + kv._2.valueStr
113   })
114 }

116 /**
117  * A MongoDB namespaced index.
118  * A MongoDB index is composed with the namespace (the fully qualified collection name) and the other fields of [[reactivemongo.api.indexes.Index]].
119  *
120  * Consider reading [[http://www.mongodb.org/display/DOCS/Indexes the documentation about indexes in MongoDB]].
121  *
122  * @param namespace The fully qualified name of the indexed collection.
123  * @param index The other fields of the index.
124  */
125 case class NSIndex(namespace: String, index: Index) {
126   val (dbName: String, collectionName: String) = {
127     val spanned = namespace.span(_ != '.')
128     spanned._1 -> spanned._2.drop(1)
129   }
130 }

132 /** Indexes manager at database level. */
133 sealed trait IndexesManager {
134 
135   /** Gets a future list of all the index on this database. */
136   def list(): Future[List[NSIndex]]
137 
138   /**
139    * Creates the given index only if it does not exist on this database.
140    *
141    * Warning: given the options you choose, and the data to index, it can be a long and blocking operation on the database.
142    * You should really consider reading [[http://www.mongodb.org/display/DOCS/Indexes]] before doing this, especially in production.
143    *
144    * @param nsIndex The index to create.
145    *
146    * @return a future containing true if the index was created, false if it already exists.
147    */
148   def ensure(nsIndex: NSIndex): Future[Boolean]
149 
150   /**
151    * Creates the given index.
152    *
153    * Warning: given the options you choose, and the data to index, it can be a long and blocking operation on the database.
154    * You should really consider reading [[http://www.mongodb.org/display/DOCS/Indexes]] before doing this, especially in production.
155    *
156    * @param nsIndex The index to create.
157    */
158   def create(nsIndex: NSIndex): Future[WriteResult]
159 
160   /**
161    * Deletes the given index on the given collection.
162    *
163    * @return The deleted index number.
164    */
165   @deprecated("Use drop instead", "0.11.0")
166   def delete(nsIndex: NSIndex): Future[Int] =
167     delete(nsIndex.collectionName, nsIndex.index.eventualName)
168 
169   /**
170    * Deletes the given index on the given collection.
171    *
172    * @return The number of indexes that were dropped.
173    */
174   @deprecated("Use drop instead", "0.11.0")
175   def delete(collectionName: String, indexName: String): Future[Int] =
176     drop(collectionName, indexName)
177 
178   /**
179    * Drops the given index on the given collection.
180    *
181    * @return The number of indexes that were dropped.
182    */
183   def drop(nsIndex: NSIndex): Future[Int] =
184     drop(nsIndex.collectionName, nsIndex.index.eventualName)
185 
186   /**
187    * Drops the given index on the given collection.
188    *
189    * @return The number of indexes that were dropped.
190    */
191   def drop(collectionName: String, indexName: String): Future[Int]
192 
193   /**
194    * Drops all the indexes on the given collection.
195    */
196   def dropAll(collectionName: String): Future[Int]
197 
198   /** Gets a manager for the given collection. */
199   def onCollection(name: String): CollectionIndexesManager
200 }

202 /**
203  * A helper class to manage the indexes on a Mongo 2.x database.
204  *
205  * @param db The subject database.
206  */
207 final class LegacyIndexesManager(db: DB)(
208     implicit context: ExecutionContext) extends IndexesManager {
209 
210   val collection = db("system.indexes")
211 
212   def list(): Future[List[NSIndex]] = collection.find(BSONDocument()).cursor(db.connection.options.readPreference)(IndexesManager.NSIndexReader, context, CursorProducer.defaultCursorProducer).collect[List]()
213 
214   def ensure(nsIndex: NSIndex): Future[Boolean] = {
215     val query = BSONDocument(
216       "ns" -> BSONString(nsIndex.namespace),
217       "name" -> BSONString(nsIndex.index.eventualName))
218 
219     collection.find(query).one.flatMap { idx =>
220       if (!idx.isDefined)
221         create(nsIndex).map(_ => true)
222       // there is a match, returning a future ok. TODO
223       else Future.successful(false)
224     }
225   }
226 
227   def create(nsIndex: NSIndex): Future[WriteResult] = {
228     implicit val writer = IndexesManager.NSIndexWriter
229     collection.insert(nsIndex)
230   }
231 
232   def drop(collectionName: String, indexName: String): Future[Int] = {
233     import reactivemongo.api.commands.bson.BSONDropIndexesImplicits._
234     db.collection(collectionName).runValueCommand(DropIndexes(indexName))
235   }
236 
237   def dropAll(collectionName: String): Future[Int] = drop(collectionName, "*")
238 
239   def onCollection(name: String): CollectionIndexesManager =
240     new LegacyCollectionIndexesManager(db.name, name, this)
241 }

243 /**
244  * A helper class to manage the indexes on a Mongo 3.x database.
245  *
246  * @param db The subject database.
247  */
248 final class DefaultIndexesManager(db: DB with DBMetaCommands)(
249     implicit context: ExecutionContext) extends IndexesManager {
250 
251   import reactivemongo.api.commands.ListIndexes
252 
253   private def listIndexes(collections: List[String], indexes: List[NSIndex]): Future[List[NSIndex]] = collections match {
254     case c :: cs => onCollection(c).list().flatMap(ix =>
255       listIndexes(cs, indexes ++ ix.map(NSIndex(s"${db.name}.$c", _))))
256     case _ => Future.successful(indexes)
257   }
258 
259   def list(): Future[List[NSIndex]] =
260     db.collectionNames.flatMap(listIndexes(_, Nil))
261 
262   def ensure(nsIndex: NSIndex): Future[Boolean] =
263     onCollection(nsIndex.collectionName).ensure(nsIndex.index)
264 
265   def create(nsIndex: NSIndex): Future[WriteResult] =
266     onCollection(nsIndex.collectionName).create(nsIndex.index)
267 
268   def drop(collectionName: String, indexName: String): Future[Int] =
269     onCollection(collectionName).drop(indexName)
270 
271   def dropAll(collectionName: String): Future[Int] =
272     onCollection(collectionName).dropAll()
273 
274   def onCollection(name: String): CollectionIndexesManager =
275     new DefaultCollectionIndexesManager(db, name)
276 }

370 private class DefaultCollectionIndexesManager(db: DB, collectionName: String)(
371     implicit context: ExecutionContext) extends CollectionIndexesManager {
372 
373   import reactivemongo.api.commands.{
374     CreateIndexes,
375     Command,
376     ListIndexes
377   }
378   import reactivemongo.api.commands.bson.BSONListIndexesImplicits._
379   import reactivemongo.api.commands.bson.BSONCreateIndexesImplicits._
380 
381   private lazy val collection = db(collectionName)
382   private lazy val listCommand = ListIndexes(db.name)
383 
384   def list(): Future[List[Index]] =
385     Command.run(BSONSerializationPack)(collection, listCommand) recoverWith {
386       case err: WriteResult if err.code.exists(_ == 26 /* no database */ ) =>
387         Future.successful(List.empty[Index])
388 
389       case err => Future.failed(err)
390     }
391 
392   def ensure(index: Index): Future[Boolean] = for {
393     is <- list().map(_.dropWhile(_.key != index.key).headOption.isDefined)
394     cr <- if (!is) create(index).map(_ => true) else Future.successful(false)
395   } yield cr
396 
397   def create(index: Index): Future[WriteResult] =
398     Command.run(BSONSerializationPack)(collection,
399       CreateIndexes(db.name, List(index)))
400 
401   @deprecated("Use drop instead", "0.11.0")
402   def delete(index: Index) = drop(index.eventualName)
403 
404   @deprecated("Use drop instead", "0.11.0")
405   def delete(name: String) = drop(name)
406 
407   @deprecated("Use [[IndexesManager.drop]]", "0.11.0")
408   def drop(nsIndex: NSIndex): Future[Int] = {
409     import reactivemongo.api.commands.bson.BSONDropIndexesImplicits._
410     Command.run(BSONSerializationPack)(
411       db(nsIndex.collectionName), DropIndexes(nsIndex.index.eventualName)).
412       map(_.value)
413   }
414 
415   def drop(indexName: String): Future[Int] = {
416     import reactivemongo.api.commands.bson.BSONDropIndexesImplicits._
417     Command.run(BSONSerializationPack)(collection, DropIndexes(indexName)).
418       map(_.value)
419   }
420 
421   @inline def dropAll(): Future[Int] = drop("*")
422 }

424 /** Factory for indexes manager scoped with a specified collection. */
425 object CollectionIndexesManager {
426   /**
427    * Returns an indexes manager for specified collection.
428    *
429    * @param db the database
430    * @param collectionName the collection name
431    */
432   def apply(db: DB, collectionName: String)(implicit context: ExecutionContext): CollectionIndexesManager = {
433     val wireVer = db.connection.metadata.map(_.maxWireVersion)
434 
435     if (wireVer.exists(_ >= MongoWireVersion.V30)) {
436       new DefaultCollectionIndexesManager(db, collectionName)
437     } else new LegacyCollectionIndexesManager(db.name, collectionName,
438       new LegacyIndexesManager(db))
439   }
440 }

442 object IndexesManager {
443   import reactivemongo.util.option
444 
445   /**
446    * Returns an indexes manager for specified database.
447    *
448    * @param db the database
449    */
450   def apply(db: DB with DBMetaCommands)(implicit context: ExecutionContext): IndexesManager = {
451     val wireVer = db.connection.metadata.map(_.maxWireVersion)
452 
453     if (wireVer.exists(_ >= MongoWireVersion.V30)) new DefaultIndexesManager(db)
454     else new LegacyIndexesManager(db)
455   }
456 
457   protected def toBSONDocument(nsIndex: NSIndex) = {
458     BSONDocument(
459       "ns" -> BSONString(nsIndex.namespace),
460       "name" -> BSONString(nsIndex.index.eventualName),
461       "key" -> BSONDocument(
462         (for (kv <- nsIndex.index.key)
463           yield kv._1 -> kv._2.value).toStream),
464       "background" -> option(nsIndex.index.background, BSONBoolean(true)),
465       "dropDups" -> option(nsIndex.index.dropDups, BSONBoolean(true)),
466       "sparse" -> option(nsIndex.index.sparse, BSONBoolean(true)),
467       "unique" -> option(nsIndex.index.unique, BSONBoolean(true)),
468       "partialFilterExpression" -> nsIndex.index.partialFilter) ++ nsIndex.index.options
469   }
470 
471   implicit object NSIndexWriter extends BSONDocumentWriter[NSIndex] {
472     def write(nsIndex: NSIndex): BSONDocument = {
473       if (nsIndex.index.key.isEmpty)
474         throw new RuntimeException("the key should not be empty!")
475       toBSONDocument(nsIndex)
476     }
477   }
478 
479   implicit object IndexReader extends BSONDocumentReader[Index] {
480     def read(doc: BSONDocument): Index = doc.getAs[BSONDocument]("key").map(
481       _.elements.map { elem => elem._1 -> IndexType(elem._2) }.toList).
482       fold[Index](throw new Exception("the key must be defined")) { key =>
483         val options = doc.elements.filterNot { element =>
484           element._1 == "ns" || element._1 == "key" || element._1 == "name" ||
485             element._1 == "unique" || element._1 == "background" ||
486             element._1 == "dropDups" || element._1 == "sparse" ||
487             element._1 == "v" || element._1 == "partialFilterExpression"
488         }.toSeq
489 
490         (for {
491           name <- doc.getAsUnflattenedTry[String]("name")
492           unique <- doc.getAsUnflattenedTry[BSONBooleanLike]("unique").
493             map(_.fold(false)(_.toBoolean))
494 
495           background <- doc.getAsUnflattenedTry[BSONBooleanLike]("background").
496             map(_.fold(false)(_.toBoolean))
497 
498           dropDups <- doc.getAsUnflattenedTry[BSONBooleanLike]("dropDups").
499             map(_.fold(false)(_.toBoolean))
500 
501           sparse <- doc.getAsUnflattenedTry[BSONBooleanLike]("sparse").
502             map(_.fold(false)(_.toBoolean))
503 
504           version <- doc.getAsUnflattenedTry[BSONNumberLike]("v").
505             map(_.map(_.toInt))
506 
507           partialFilter <- doc.getAsUnflattenedTry[BSONDocument](
508             "partialFilterExpression")
509         } yield Index(key, name, unique, background, dropDups,
510           sparse, version, partialFilter, BSONDocument(options))).get
511       }
512   }
513 
514   implicit object NSIndexReader extends BSONDocumentReader[NSIndex] {
515     def read(doc: BSONDocument): NSIndex =
516       doc.getAs[BSONString]("ns").map(_.value).fold[NSIndex](
517         throw new Exception("the namespace ns must be defined"))(
518           NSIndex(_, doc.as[Index]))
519   }
