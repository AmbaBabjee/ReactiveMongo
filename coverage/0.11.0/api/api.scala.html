70 /**
71  * A helper that sends the given message to the given actor,
72  * following a failover strategy.
73  * This helper holds a future reference that is completed with a response,
74  * after 1 or more attempts (specified in the given strategy).
75  * If the all the tryouts configured by the given strategy were unsuccessful,
76  * the future reference is completed with a Throwable.
77  *
78  * Should not be used directly for most use cases.
79  *
80  * @tparam T Type of the message to send.
81  * @param message The message to send to the given actor. This message will be wrapped into an ExpectingResponse message by the `expectingResponseMaker` function.
82  * @param connection The reference to the MongoConnection the given message will be sent to.
83  * @param strategy The Failover strategy.
84  * @param expectingResponseMaker A function that takes a message of type `T` and wraps it into an ExpectingResponse message.
85  */
86 class Failover[T](message: T, connection: MongoConnection, strategy: FailoverStrategy)(expectingResponseMaker: T => ExpectingResponse)(implicit ec: ExecutionContext) {
87   import Failover2.logger
88   import reactivemongo.core.errors._
89   import reactivemongo.core.actors.Exceptions._
90 
91   private val promise = Promise[Response]()
92 
93   /**
94    * A future that is completed with a response,
95    * after 1 or more attempts (specified in the given strategy).
96    */
97   val future: Future[Response] = promise.future
98 
99   private def send(n: Int) {
100     val expectingResponse = expectingResponseMaker(message)
101     connection.mongosystem ! expectingResponse
102     expectingResponse.future.onComplete {
103       case Failure(e) if isRetryable(e) =>
104         if (n < strategy.retries) {
105           val `try` = n + 1
106           val delayFactor = strategy.delayFactor(`try`)
107           val delay = Duration.unapply(strategy.initialDelay * delayFactor).map(t => FiniteDuration(t._1, t._2)).getOrElse(strategy.initialDelay)
108 
109           logger.debug(s"Got an error, retrying... (try #${`try`} is scheduled in ${delay.toMillis} ms)", e)
110 
111           connection.actorSystem.scheduler.scheduleOnce(delay)(send(`try`))
112         } else {
113           // generally that means that the primary is not available or the nodeset is unreachable
114           logger.error("Got an error, no more attempts to do. Completing with a failure...", e)
115           promise.failure(e)
116         }
117 
118       case Failure(e) => {
119         logger.trace(
120           "Got an non retryable error, completing with a failure...", e)
121         promise.failure(e)
122       }
123 
124       case Success(response) => {
125         logger.trace("Got a successful result, completing...")
126         promise.success(response)
127       }
128     }
129   }
130 
131   private def isRetryable(throwable: Throwable) = throwable match {
132     case PrimaryUnavailableException | NodeSetNotReachable => true
133     case e: DatabaseException if e.isNotAPrimaryError || e.isUnauthorized => true
134     case _: ConnectionException => true
135     case _: ConnectionNotInitialized => true
136     case _ => false
137   }
138 
139   send(0)
140 }

165     next().map[Try[A]](Success(_)).recover[Try[A]] {
166       case err => Failure(err)
167     }.flatMap {
168       case Failure(e) if isRetryable(e) => {
169         if (n < strategy.retries) {
170           val `try` = n + 1
171           val delayFactor = strategy.delayFactor(`try`)
172           val delay = Duration.unapply(strategy.initialDelay * delayFactor).
173             fold(strategy.initialDelay)(t => FiniteDuration(t._1, t._2))

175           logger.debug(s"Got an error, retrying... (try #${`try`} is scheduled in ${delay.toMillis} ms)", e)

177           after(delay, connection.actorSystem.scheduler)(send(`try`))
178         } else {
179           // generally that means that the primary is not available
180           // or the nodeset is unreachable

186

190       }
191 
192       case Success(response) => {
193         trace("Got a successful result, completing...")
194         Future.successful(response)
195       }
196     }

331   /** Returns a future that will be successful when node set is available. */
332   private[api] def waitIsAvailable(failoverStrategy: FailoverStrategy)(implicit ec: ExecutionContext): Future[Unit] = {
333     logger.debug(s"[$lnm] Waiting is available...")
334 
335     @inline def nextTimeout(i: Int): FiniteDuration = {
336       val delayFactor: Double = failoverStrategy.delayFactor(i)
337 
338       Duration.unapply(failoverStrategy.initialDelay * delayFactor).
339         fold(failoverStrategy.initialDelay)(t => FiniteDuration(t._1, t._2))
340     }
341 
342     type Retry = (FiniteDuration, Throwable)
343 
344     def wait(iteration: Int, attempt: Int, timeout: FiniteDuration): Future[Unit] = {
345       logger.trace(
346         s"[$lnm] Wait is available: $attempt @ ${System.currentTimeMillis}")
347 
348       if (attempt == 0) {
349         Future.failed(new Exceptions.NodeSetNotReachable(this))
350       } else {
351         @inline def res: Either[Retry, Unit] = try {
352           val before = System.currentTimeMillis
353           val result = Await.result(isAvailable, timeout)
354           val duration = System.currentTimeMillis - before
355 
356           if (result) Right({})
357           else Left(FiniteDuration(
358             timeout.toMillis - duration, MILLISECONDS) -> null)
359 
360         } catch {
361           case e: Throwable => Left(timeout -> e)
362         }
363 
364         @inline def doRetry(delay: FiniteDuration): Future[Unit] = after(delay, actorSystem.scheduler) {
365           val nextIt = iteration + 1
366           wait(nextIt, attempt - 1, nextTimeout(nextIt))
367         }
368 
369         res match {
370           case Right(_)            => Future.successful({})
371           case Left((delay, null)) => doRetry(delay)
372           case Left((delay, error)) =>
373             logger.warn(s"[$lnm] Got an error, retrying", error)
374             doRetry(delay)
375         }
376       }
377     }
378 
379     wait(0, 1 + failoverStrategy.retries, failoverStrategy.initialDelay).
380       flatMap { _ =>
381         metadata match {
382           case Some(ProtocolMetadata(
383             _, MongoWireVersion.V24AndBefore, _, _, _)) =>
384             Future.failed[Unit](ConnectionException(
385               s"unsupported MongoDB version < 2.6 ($lnm)"))
386 
387           case Some(_) => Future successful {}
388           case _ => Future.failed[Unit](ConnectionException(
389             s"protocol metadata not available ($lnm)"))
390         }
391       }
392   }

563   /**

565    * @param options the connection options

567    * @param db the name of the database
568    * @param authenticate the authenticate information (see [[MongoConnectionOptions.authMode]])

570   final case class ParsedURI(
571     hosts: List[(String, Int)],
572     options: MongoConnectionOptions,
573     ignoredOptions: List[String],
574     db: Option[String],

577   /**
578    * Parses a MongoURI.
579    *
580    * @param uri the connection URI (see [[http://docs.mongodb.org/manual/reference/connection-string/ the MongoDB URI documentation]] for more information)
581    */
582   def parseURI(uri: String): Try[ParsedURI] = {
583     val prefix = "mongodb://"
584 
585     Try {
586       val useful = uri.replace(prefix, "")
587       def opts = makeOptions(parseOptions(useful))
588 
589       if (useful.indexOf("@") == -1) {
590         val (db, hosts) = parseHostsAndDbName(useful)
591         val (unsupportedKeys, options) = opts
592         ParsedURI(hosts, options, unsupportedKeys, db, None)
593       } else {
594         val WithAuth = """([^:]+):([^@]*)@(.+)""".r
595 
596         useful match {
597           case WithAuth(user, pass, hostsPortsAndDbName) => {
598             val (db, hosts) = parseHostsAndDbName(hostsPortsAndDbName)
599 
600             db.fold[ParsedURI](throw new URIParsingException(s"Could not parse URI '$uri': authentication information found but no database name in URI")) { database =>
601               val (unsupportedKeys, options) = opts
602 
603               ParsedURI(hosts, options, unsupportedKeys, Some(database), Some(Authenticate.apply(options.authSource.getOrElse(database), user, pass)))
604             }
605           }
606 
607           case _ => throw new URIParsingException(s"Could not parse URI '$uri'")
608         }
609       }
610     }
611   }
612 
613   private def parseHosts(hosts: String) = hosts.split(",").toList.map { host =>
614     host.split(':').toList match {
615       case host :: port :: Nil => host -> {
616         try {
617           val p = port.toInt
618           if (p > 0 && p < 65536) p
619           else throw new URIParsingException(s"Could not parse hosts '$hosts' from URI: invalid port '$port'")
620         } catch {
621           case _: NumberFormatException => throw new URIParsingException(s"Could not parse hosts '$hosts' from URI: invalid port '$port'")
622           case NonFatal(e)              => throw e
623         }
624       }
625       case host :: Nil => host -> DefaultPort
626       case _           => throw new URIParsingException(s"Could not parse hosts from URI: invalid definition '$hosts'")
627     }
628   }
629 
630   private def parseHostsAndDbName(hostsPortAndDbName: String): (Option[String], List[(String, Int)]) = hostsPortAndDbName.split("/").toList match {
631     case hosts :: Nil           => None -> parseHosts(hosts.takeWhile(_ != '?'))
632     case hosts :: dbName :: Nil => Some(dbName.takeWhile(_ != '?')) -> parseHosts(hosts)
633     case _ =>
634       throw new URIParsingException(s"Could not parse hosts and database from URI: '$hostsPortAndDbName'")
635   }

637   private def parseOptions(uriAndOptions: String): Map[String, String] =
638     uriAndOptions.split('?').toList match {
639       case uri :: options :: Nil => options.split("&").map { option =>
640         option.split("=").toList match {
641           case key :: value :: Nil => (key -> value)
642           case _                   => throw new URIParsingException(s"Could not parse URI '$uri': invalid options '$options'")
643         }
644       }.toMap
645       case _ => Map.empty
646     }

651   private def makeOptions(opts: Map[String, String]): (List[String], MongoConnectionOptions) = {
652     val (remOpts, step1) = opts.iterator.foldLeft(
653       Map.empty[String, String] -> MongoConnectionOptions()) {
654         case ((unsupported, result), kv) => kv match {
