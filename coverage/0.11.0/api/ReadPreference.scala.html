25   private def TagFilter(tagSet: Seq[Map[String, String]]): Option[BSONDocument => Boolean] = if (tagSet.isEmpty) None else Some { doc: BSONDocument =>
26     val matching = tagSet.find(_.foldLeft(Map.empty[String, String]) {
27       case (ms, (k, v)) =>
28         doc.getAs[String](k).filter(_ == v) match {
29           case None => ms + (k -> v)
30           case _    => ms
31         }
32     }.isEmpty)

203   private implicit class BSONDocumentWrapper(val underlying: BSONDocument) extends AnyVal {
204     def contains(doc: BSONDocument): Boolean = {
205 
206       val els = underlying.elements
207       doc.elements.forall { element =>
208         els.find {
209           case (name, value) => element._1 == name && ((element._2, value) match {
210             case (d1: BSONDocument, d2: BSONDocument) => d1.elements == d2.elements
211             case (a1: BSONArray, a2: BSONArray)       => a1.values == a2.values
212             case (v1, v2)                             => v1 == v2
213           })
214         }.isDefined
215       }
216     }
217   }
