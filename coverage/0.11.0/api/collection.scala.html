20 /**
21  * A MongoDB Collection.
22  * You should consider the generic API
23  * ([[reactivemongo.api.collections.GenericCollection]])
24  * and the default [[reactivemongo.api.collections.bson.BSONCollection]].
25  *
26  * {{{
27  * import reactivemongo.bson._
28  * import reactivemongo.api.collections.bson.BSONCollection
29  *
30  * object Samples {
31  *
32  *   val connection = MongoConnection(List("localhost"))
33  *
34  *   // Gets a reference to the database "plugin"
35  *   val db = connection("plugin")
36  *
37  *   // Gets a reference to the collection "acoll"
38  *   // By default, you get a BSONCollection.
39  *   val collection = db[BSONCollection]("acoll")
40  *
41  *   def listDocs() = {
42  *     // Select only the documents which field 'firstName' equals 'Jack'
43  *     val query = BSONDocument("firstName" -> "Jack")
44  *     // select only the field 'lastName'
45  *     val filter = BSONDocument(
46  *       "lastName" -> 1,
47  *       "_id" -> 0)
48  *
49  *     // Get a cursor of BSONDocuments
50  *     val cursor = collection.find(query, filter).cursor[BSONDocument]
51  *     // Let's enumerate this cursor and print a readable representation of each document in the response
52  *     cursor.enumerate().apply(Iteratee.foreach { doc =>
53  *       println("found document: " + BSONDocument.pretty(doc))
54  *     })
55  *
56  *     // Or, the same with getting a list
57  *     val cursor2 = collection.find(query, filter).cursor[BSONDocument]
58  *     val futureList = cursor.collect[List]()
59  *     futureList.map { list =>
60  *       list.foreach { doc =>
61  *         println("found document: " + BSONDocument.pretty(doc))
62  *       }
63  *     }
64  *   }
65  * }
66  * }}}
67  */
68 trait Collection {
69   /** The database which this collection belong to. */
70   def db: DB
71 
72   /** The name of the collection. */
73   def name: String
74 
75   /** The default failover strategy for the methods of this collection. */
76   def failoverStrategy: FailoverStrategy
77 
78   /** Gets the full qualified name of this collection. */
79   def fullCollectionName = db.name + "." + name
80 
81   /**
82    * Gets another implementation of this collection.
83    * An implicit CollectionProducer[C] must be present in the scope, or it will be the default implementation ([[reactivemongo.api.collections.bson.BSONCollection]]).
84    *
85    * @param failoverStrategy Overrides the default strategy.
86    */
87   def as[C <: Collection](failoverStrategy: FailoverStrategy = failoverStrategy)(implicit producer: CollectionProducer[C] = collections.bson.BSONCollectionProducer): C = {
88     producer.apply(db, name, failoverStrategy)
89   }
90 
91   /**
92    * Gets another collection in the current database.
93    * An implicit CollectionProducer[C] must be present in the scope, or it will be the default implementation ([[reactivemongo.api.collections.bson.BSONCollection]]).
94    *
95    * @param name The other collection name.
96    * @param failoverStrategy Overrides the default strategy.
97    */
98   @deprecated("Consider using `sibling` instead", "0.10")
99   def sister[C <: Collection](name: String, failoverStrategy: FailoverStrategy = failoverStrategy)(implicit producer: CollectionProducer[C] = collections.bson.BSONCollectionProducer): C =
100     sibling(name, failoverStrategy)
101 
102   /**
103    * Gets another collection in the current database.
104    * An implicit CollectionProducer[C] must be present in the scope, or it will be the default implementation ([[reactivemongo.api.collections.bson.BSONCollection]]).
105    *
106    * @param name The other collection name.
107    * @param failoverStrategy Overrides the default strategy.
108    */
109   def sibling[C <: Collection](name: String, failoverStrategy: FailoverStrategy = failoverStrategy)(implicit producer: CollectionProducer[C] = collections.bson.BSONCollectionProducer): C =
110     producer.apply(db, name, failoverStrategy)
111 }
112 
113 /**
114  * A Producer of [[Collection]] implementation.
115  *
116  * This is used to get an implementation implicitly when getting a reference of a [[Collection]].
117  */
118 trait CollectionProducer[+C <: Collection] {
119   def apply(db: DB, name: String, failoverStrategy: FailoverStrategy = FailoverStrategy()): C
120 }

122 /** A mixin that provides commands about this Collection itself. */
123 trait CollectionMetaCommands {
124   self: Collection =>
125 
126   import concurrent.{ ExecutionContext, Future }
127   import reactivemongo.api.commands._
128   import reactivemongo.api.commands.bson._
129   import CommonImplicits._
130   import BSONCreateImplicits._
131   import BSONEmptyCappedImplicits._
132   import BSONCollStatsImplicits._
133   import BSONRenameCollectionImplicits._
134   import BSONConvertToCappedImplicits._
135   import reactivemongo.api.indexes.CollectionIndexesManager
136 
137   /**
138    * Creates this collection.
139    *
140    * The returned future will be completed with an error if this collection already exists.
141    *
142    * @param autoIndexId States if should automatically add an index on the _id field. By default, regular collections will have an indexed _id field, in contrast to capped collections.
143    */
144   def create(autoIndexId: Boolean = true)(implicit ec: ExecutionContext): Future[Unit] = Command.run(BSONSerializationPack).unboxed(self, Create(None, autoIndexId))
145 
146   /**
147    * Creates this collection as a capped one.
148    *
149    * The returned future will be completed with an error if this collection already exists.
150    *
151    * @param size the byte size of the collections.
152    * @param maxDocuments the maximum number of documents.
153    * @param autoIndexId States if should automatically add an index on the _id field. By default, capped collections will NOT have an indexed _id field, in contrast to regular collections.
154    */
155   def createCapped(size: Long, maxDocuments: Option[Int], autoIndexId: Boolean = false)(implicit ec: ExecutionContext): Future[Unit] =
156     Command.run(BSONSerializationPack).
157       unboxed(self, Create(Some(Capped(size, maxDocuments)), autoIndexId))
158 
159   /**
160    * Drops this collection.
161    *
162    * The returned future will be completed with an error
163    * if this collection does not exist.
164    */
165   @deprecated("Use `drop(Boolean)`", "0.12.0")
166   def drop()(implicit ec: ExecutionContext): Future[Unit] =
167     drop(true).map(_ => {})
168 
169   /**
170    * Drops this collection.
171    *
172    * If the collection existed and is successfully dropped,
173    * the returned future will be completed with true.
174    *
175    * If `failIfNotFound` is false and the collection doesn't exist,
176    * the returned future will be completed with false.
177    *
178    * Otherwise in case, the future will be completed with the encountered error.
179    */
180   def drop(failIfNotFound: Boolean)(implicit ec: ExecutionContext): Future[Boolean] = {
181     import BSONDropCollectionImplicits._
182 
183     Command.run(BSONSerializationPack)(self, DropCollection).flatMap {
184       case DropCollectionResult(false) if failIfNotFound =>
185         Future.failed[Boolean](GenericDatabaseException(
186           s"fails to drop collection: $name", Some(26)))
187 
188       case DropCollectionResult(dropped) => Future.successful(dropped)
189     }
190   }
191 
192   /**
193    * If this collection is capped, removes all the documents it contains.
194    *
195    * Deprecated because it became an internal command, unavailable by default.
196    */
197   @deprecated("Deprecated because emptyCapped became an internal command, unavailable by default.", "0.9")
198   def emptyCapped()(implicit ec: ExecutionContext): Future[Unit] =
199     Command.run(BSONSerializationPack).unboxed(self, EmptyCapped)
200 
201   /**
202    * Converts this collection to a capped one.
203    *
204    * @param size The size of this capped collection, in bytes.
205    * @param maxDocuments The maximum number of documents this capped collection can contain.
206    */
207   def convertToCapped(size: Long, maxDocuments: Option[Int])(implicit ec: ExecutionContext): Future[Unit] =
208     Command.run(BSONSerializationPack).unboxed(self, ConvertToCapped(Capped(size, maxDocuments)))
209 
210   /**
211    * Renames this collection.
212    *
213    * @param to The new name of this collection.
214    * @param dropExisting If a collection of name `to` already exists, then drops that collection before renaming this one.
215    */
216   def rename(to: String, dropExisting: Boolean = false)(implicit ec: ExecutionContext): Future[Unit] =
217     Command.run(BSONSerializationPack).unboxed(self.db, RenameCollection(db.name + "." + name, db.name + "." + to, dropExisting))
218 
219   /**
220    * Returns various information about this collection.
221    */
222   def stats()(implicit ec: ExecutionContext): Future[CollStatsResult] =
223     Command.run(BSONSerializationPack)(self, CollStats(None))
224 
225   /**
226    * Returns various information about this collection.
227    *
228    * @param scale A scale factor (for example, to get all the sizes in kilobytes).
229    */
230   def stats(scale: Int)(implicit ec: ExecutionContext): Future[CollStatsResult] =
231     Command.run(BSONSerializationPack)(self, CollStats(Some(scale)))
232 
233   /** Returns an index manager for this collection. */
234   def indexesManager(implicit ec: ExecutionContext): CollectionIndexesManager =
235     CollectionIndexesManager(self.db, name)
236 }
