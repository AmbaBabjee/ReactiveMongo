27 /**
28  * A Mongo Database.
29  *
30  * You should consider the provided [[reactivemongo.api.DefaultDB]] implementation.
31  *
32  * Example:
33  * {{{
34  * import reactivemongo.api._
35  *
36  * val connection = MongoConnection( List( "localhost:27016" ) )
37  * val db = connection.database("plugin")
38  * val collection = db("acoll")
39  * }}}
40  */
41 sealed trait DB {
42   /** The [[reactivemongo.api.MongoConnection]] that will be used to query this database. */
43   def connection: MongoConnection
44 
45   /** This database name. */
46   def name: String
47 
48   /** A failover strategy for sending requests. */
49   def failoverStrategy: FailoverStrategy
50 
51   /**
52    * Returns a [[reactivemongo.api.Collection]] from this database
53    * (alias for the `collection` method).
54    *
55    * @param name the name of the collection to open
56    */
57   def apply[C <: Collection](name: String, failoverStrategy: FailoverStrategy = failoverStrategy)(implicit producer: CollectionProducer[C] = collections.bson.BSONCollectionProducer): C = collection(name, failoverStrategy)
58 
59   /**
60    * Returns a [[reactivemongo.api.Collection]] from this database.
61    *
62    * @param name the name of the collection to open
63    */
64   def collection[C <: Collection](name: String, failoverStrategy: FailoverStrategy = failoverStrategy)(implicit producer: CollectionProducer[C] = collections.bson.BSONCollectionProducer): C = producer(this, name, failoverStrategy)
65 
66   /**
67    * Returns a [[reactivemongo.api.Collection]] from this database, when ready.
68    *
69    * @param name the name of the collection to open
70    */
71   def coll[C <: Collection](name: String, failoverStrategy: FailoverStrategy = failoverStrategy)(implicit ec: ExecutionContext, producer: CollectionProducer[C] = collections.bson.BSONCollectionProducer): Future[C] = connection.waitIsAvailable(failoverStrategy).map(_ => producer(this, name, failoverStrategy))
72 
73   @inline def defaultReadPreference: ReadPreference =
74     connection.options.readPreference
75 
76   /**
77    * Sends a command and get the future result of the command.
78    *
79    * @param command The command to send.
80    * @param readPreference The ReadPreference to use for this command (defaults to [[MongoConnectionOptions.readPreference]]).
81    *
82    * @return a future containing the result of the command.
83    */
84   @deprecated("Consider using reactivemongo.api.commands along with `GenericDB.runCommand` methods", "0.11.0")
85   def command[T](command: Command[T], readPreference: ReadPreference = defaultReadPreference)(implicit ec: ExecutionContext): Future[T] =
86     Failover(command.apply(name).maker(readPreference),
87       connection, failoverStrategy).future.mapEither(command.ResultMaker(_))
88 
89   /** Authenticates the connection on this database. */
90   def authenticate(user: String, password: String)(implicit timeout: FiniteDuration): Future[SuccessfulAuthentication] = connection.authenticate(name, user, password)
91 
92   /**
93    * Returns the database of the given name on the same MongoConnection.
94    * @see [[sibling1]]
95    */
96   def sibling(name: String, failoverStrategy: FailoverStrategy = failoverStrategy)(implicit ec: ExecutionContext): DefaultDB = connection(name, failoverStrategy)
97 
98   /** Returns the database of the given name on the same MongoConnection. */
99   def sibling1(name: String, failoverStrategy: FailoverStrategy = failoverStrategy)(implicit ec: ExecutionContext): Future[DefaultDB] = connection.database(name, failoverStrategy)
100 
101 }

103 trait GenericDB[P <: SerializationPack with Singleton] { self: DB =>
104   val pack: P
105 
106   import reactivemongo.api.commands._
107 
108   def runner = Command.run(pack)
109 
110   def runCommand[R, C <: Command with CommandWithResult[R]](command: C with CommandWithResult[R])(implicit writer: pack.Writer[C], reader: pack.Reader[R], ec: ExecutionContext): Future[R] = runner(self, command)
111 
112   def runCommand[C <: Command](command: C)(implicit writer: pack.Writer[C]): CursorFetcher[pack.type, Cursor] = runner(self, command)
113 
114   def runValueCommand[A <: AnyVal, R <: BoxedAnyVal[A], C <: Command with CommandWithResult[R]](command: C with CommandWithResult[R with BoxedAnyVal[A]])(implicit writer: pack.Writer[C], reader: pack.Reader[R], ec: ExecutionContext): Future[A] = runner.unboxed(self, command)

117 /** A mixin that provides commands about this database itself. */
118 trait DBMetaCommands { self: DB =>
119   import reactivemongo.core.protocol.MongoWireVersion
120   import reactivemongo.api.commands.{
121     Command,
122     DropDatabase,
123     ListCollectionNames,
124     ServerStatus,
125     ServerStatusResult
126   }
127   import reactivemongo.api.commands.bson.{
128     CommonImplicits,
129     BSONDropDatabaseImplicits,
130     BSONServerStatusImplicits
131   }
132   import reactivemongo.api.commands.bson.BSONListCollectionNamesImplicits._
133   import CommonImplicits._
134   import BSONDropDatabaseImplicits._
135   import BSONServerStatusImplicits._
136 
137   /** Drops this database. */
138   def drop()(implicit ec: ExecutionContext): Future[Unit] =
139     Command.run(BSONSerializationPack).unboxed(self, DropDatabase)
140 
141   /** Returns an index manager for this database. */
142   def indexesManager(implicit ec: ExecutionContext) = IndexesManager(self)
143 
144   private lazy val collectionNameReader = new BSONDocumentReader[String] {
145     val prefixLength = name.size + 1
146 
147     def read(bson: BSONDocument) = bson.get("name").collect {
148       case BSONString(value) => value.substring(prefixLength)
149     }.getOrElse(throw new Exception(
150       "name is expected on system.namespaces query"))
151 
152   }
153 
154   /** Returns the names of the collections in this database. */
155   def collectionNames(implicit ec: ExecutionContext): Future[List[String]] = {
156     val wireVer = connection.metadata.map(_.maxWireVersion)
157 
158     if (wireVer.exists(_ >= MongoWireVersion.V30)) {
159       Command.run(BSONSerializationPack)(self, ListCollectionNames).map(_.names)
160     } else collection("system.namespaces").as[BSONCollection]().
161       find(BSONDocument(
162         "name" -> BSONRegex("^[^\\$]+$", "") // strip off any indexes
163         )).cursor(defaultReadPreference)(
164         collectionNameReader, ec, CursorProducer.defaultCursorProducer).
165         collect[List]()
166   }
167 
168   /** Returns the server status. */
169   def serverStatus(implicit ec: ExecutionContext): Future[ServerStatusResult] =
170     Command.run(BSONSerializationPack)(self, ServerStatus)
171 }
