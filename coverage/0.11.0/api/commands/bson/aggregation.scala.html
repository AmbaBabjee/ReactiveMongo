52   implicit object AggregateWriter
53       extends BSONDocumentWriter[ResolvedCollectionCommand[Aggregate]] {
54     def write(agg: ResolvedCollectionCommand[Aggregate]) = {
55       val cmd = BSONDocument(
56         "aggregate" -> BSONString(agg.collection),
57         "pipeline" -> BSONArray(
58           { for (pipe <- agg.command.pipeline) yield pipe.makePipe }.toStream),
59         "explain" -> BSONBoolean(agg.command.explain),
60         "allowDiskUse" -> BSONBoolean(agg.command.allowDiskUse),
61         "cursor" -> agg.command.cursor.map(CursorWriter.write(_)))
62 
63       if (agg.command.wireVersion < MongoWireVersion.V32) cmd else {
64         cmd ++ ("bypassDocumentValidation" -> BSONBoolean(
65           agg.command.bypassDocumentValidation),
66           "readConcern" -> agg.command.readConcern)
67       }
68     }
69   }

71   implicit object AggregationResultReader
72       extends DealingWithGenericCommandErrorsReader[AggregationResult] {
73     def readResult(doc: BSONDocument): AggregationResult =
74       if (doc contains "stages") AggregationResult(List(doc), None) else {
75         doc.getAs[List[BSONDocument]]("result") match {
76           case Some(docs) => AggregationResult(docs, None)
77           case _ => (for {

79             id <- cursor.getAsTry[BSONNumberLike]("id").map(_.toLong)
80             ns <- cursor.getAsTry[String]("ns")
81             firstBatch <- cursor.getAsTry[List[BSONDocument]]("firstBatch")
82           } yield AggregationResult(firstBatch, Some(ResultCursor(id, ns)))).get
83         }
84       }
