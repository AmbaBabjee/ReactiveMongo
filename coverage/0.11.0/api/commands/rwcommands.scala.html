54 case class MultiBulkWriteResult(
55     ok: Boolean,
56     n: Int,
57     nModified: Int,
58     upserted: Seq[Upserted],
59     writeErrors: Seq[WriteError],
60     writeConcernError: Option[WriteConcernError], // TODO ?
61     code: Option[Int],
62     errmsg: Option[String],
63     totalN: Int) {
64   def merge(wr: WriteResult): MultiBulkWriteResult = wr match {
65     case wr: UpdateWriteResult => MultiBulkWriteResult(
66       ok = ok && wr.ok,
67       n = n + wr.n,
68       writeErrors = writeErrors ++ wr.writeErrors.map(e => e.copy(index = e.index + totalN)),
69       writeConcernError = writeConcernError.orElse(wr.writeConcernError),
70       code = code.orElse(wr.code),
71       errmsg = errmsg.orElse(wr.errmsg),
72       nModified = wr.nModified,
73       upserted = wr.upserted,
74       totalN = totalN + wr.n + wr.writeErrors.size)
75     case _ =>
76       MultiBulkWriteResult(
77         ok = ok && wr.ok,
78         n = n + wr.n,
79         writeErrors = writeErrors ++ wr.writeErrors.map(e => e.copy(index = e.index + totalN)),
80         writeConcernError = writeConcernError.orElse(wr.writeConcernError),
81         code = code.orElse(wr.code),
82         errmsg = errmsg.orElse(wr.errmsg),
83         nModified = nModified,
84         upserted = upserted,
85         totalN = totalN + wr.n + wr.writeErrors.size)
86   }
