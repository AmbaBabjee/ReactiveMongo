286     val makePipe = makeDocument(Seq(
287       elementProducer("$sort", makeDocument(fields.map {
288         case Ascending(field)  => elementProducer(field, intValue(1))
289         case Descending(field) => elementProducer(field, intValue(-1))
290         case MetadataSort(field, keyword) => {
291           val meta = makeDocument(Seq(
292             elementProducer("$meta", stringValue(keyword.name))))

295         }
296       }))))

299   /**
300    * Outputs documents in order of nearest to farthest from a specified point.
301    * http://docs.mongodb.org/manual/reference/operator/aggregation/geoNear/#pipe._S_geoNear
302    * @param spherical if using a 2dsphere index
303    * @param limit the maximum number of documents to return
304    * @param maxDistance the maximum distance from the center point that the documents can be
305    * @param selector limits the results to the matching documents
306    * @param distanceMultiplier the factor to multiply all distances returned by the query
307    * @param uniqueDocs if this value is true, the query returns a matching document once
308    * @param near the point for which to find the closest documents
309    * @param distanceField the output field that contains the calculated distance
310    * @param includeLocs this specifies the output field that identifies the location used to calculate the distance
311    */
312   case class GeoNear(spherical: Boolean = false, limit: Long = 100, maxDistance: Option[Long] = None, selector: Option[pack.Document] = None, distanceMultiplier: Option[Double] = None, uniqueDocs: Boolean = false, near: Option[pack.Value] = None, distanceField: Option[String] = None, includeLocs: Option[String] = None) extends PipelineOperator {
313     def makePipe =
314       makeDocument(Seq(elementProducer("$geoNear", makeDocument(Seq(
315         elementProducer("spherical", booleanValue(spherical)),
316         elementProducer("limit", longValue(limit))) ++ Seq(
317           maxDistance.map(l => elementProducer("maxDistance", longValue(l))),
318           selector.map(s => elementProducer("query", s)),
319           distanceMultiplier.map(d => elementProducer(
320             "distanceMultiplier", doubleValue(d))),
321           Some(elementProducer("uniqueDocs", booleanValue(uniqueDocs))),
322           near.map(n => elementProducer("near", n)),
323           distanceField.map(s =>
324             elementProducer("distanceField", stringValue(s))),
325           includeLocs.map(s =>
326             elementProducer("includeLocs", stringValue(s)))).
327           flatten))))
328 
329   }
