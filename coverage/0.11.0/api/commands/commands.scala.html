106   def defaultCursorFetcher[P <: SerializationPack, A](db: DB, p: P, command: A, failover: FailoverStrategy)(implicit writer: p.Writer[A]): CursorFetcher[p.type, DefaultCursor.Impl] = new CursorFetcher[p.type, DefaultCursor.Impl] {
107     val pack: p.type = p
108 
109     @inline private def defaultReadPreference: ReadPreference =
110       db.connection.options.readPreference
111 
112     def one[A](readPreference: ReadPreference)(implicit reader: pack.Reader[A], ec: ExecutionContext): Future[A] = {
113       val (requestMaker, m26WriteCommand) =
114         buildRequestMaker(pack)(command, writer, readPreference, db.name)
115 
116       Failover2(db.connection, failover) { () =>
117         db.connection.sendExpectingResponse(requestMaker, m26WriteCommand)
118       }.future.map { response =>
119         pack.readAndDeserialize(
120           LoweLevelDocumentIterator(ChannelBufferReadableBuffer(
121             response.documents)).next, reader)
122 
123       }
124     }
125 
126     def one[A](implicit reader: pack.Reader[A], ec: ExecutionContext): Future[A] = one[A](defaultReadPreference)
127 
128     def cursor[A](readPreference: ReadPreference)(implicit reader: pack.Reader[A]): DefaultCursor.Impl[A] = {
129       val buffer = ChannelBufferWritableBuffer()
130       pack.serializeAndWrite(buffer, command, writer)
131 
132       val bs = BufferSequence(buffer.buffer)
133       val flags = if (readPreference.slaveOk) QueryFlags.SlaveOk else 0
134       val op = Query(flags, db.name + ".$cmd", 0, 1)
135       val mongo26WriteCommand = command match {
136         case _: Mongo26WriteCommand => true
137         case _                      => false
138       }
139 
140       DefaultCursor.query(pack, op, bs,
141         if (mongo26WriteCommand) ReadPreference.primary else readPreference,
142         db.connection, failover, mongo26WriteCommand)
143 
144     }
145 
146     def cursor[A](implicit reader: pack.Reader[A]): DefaultCursor.Impl[A] =
147       cursor(defaultReadPreference)
148   }
