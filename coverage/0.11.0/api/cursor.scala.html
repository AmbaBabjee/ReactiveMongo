29 trait Cursor[T] {
30   /**
31    * Produces an Enumerator of documents.
32    * Given the `stopOnError` parameter, this Enumerator may stop on any non-fatal exception, or skip and continue.
33    * The returned enumerator may process up to `maxDocs`. If `stopOnError` is false, then documents that cause error
34    * are dropped, so the enumerator may emit a little less than `maxDocs` even if it processes `maxDocs` documents.
35    *
36    * @param maxDocs Enumerate up to `maxDocs` documents.
37    * @param stopOnError States if the produced Enumerator may stop on non-fatal exception.
38    *
39    * @return an Enumerator of documents.
40    */
41   def enumerate(maxDocs: Int = Int.MaxValue, stopOnError: Boolean = false)(implicit ctx: ExecutionContext): Enumerator[T]
42 
43   /**
44    * Produces an Enumerator of Iterator of documents.
45    * Given the `stopOnError` parameter, this Enumerator may stop on any non-fatal exception, or skip and continue.
46    *
47    * @param maxDocs Enumerate up to `maxDocs` documents.
48    * @param stopOnError States if the produced Enumerator may stop on non-fatal exception.
49    *
50    * @return an Enumerator of Iterators of documents.
51    */
52   def enumerateBulks(maxDocs: Int = Int.MaxValue, stopOnError: Boolean = false)(implicit ctx: ExecutionContext): Enumerator[Iterator[T]]
53 
54   /**
55    * Produces an Enumerator of responses from the database.
56    * Given the `stopOnError` parameter, this Enumerator may stop on any non-fatal exception, or skip and continue.
57    *
58    * @param maxDocs Enumerate up to `maxDocs` documents.
59    * @param stopOnError States if the produced Enumerator may stop on non-fatal exception.
60    *
61    * @return an Enumerator of Responses.
62    */
63   def enumerateResponses(maxDocs: Int = Int.MaxValue, stopOnError: Boolean = false)(implicit ctx: ExecutionContext): Enumerator[Response]
64 
65   /**
66    * Collects all the documents into a collection of type `M[T]`.
67    * Given the `stopOnError` parameter (which defaults to true), the resulting Future may fail if any
68    * non-fatal exception occurs. If set to false, all the documents that caused exceptions are skipped.
69    * Up to `maxDocs` returned by the database may be processed. If `stopOnError` is false, then documents that cause error
70    * are dropped, so the result may contain a little less than `maxDocs` even if `maxDocs` documents were processed.
71    *
72    * @param upTo Collect up to `upTo` documents.
73    * @param stopOnError States if the Future should fail if any non-fatal exception occurs.
74    *
75    * Example:
76    * {{{
77    * val cursor = collection.find(query, filter).cursor[BSONDocument]
78    * // return the 3 first documents in a Vector[BSONDocument].
79    * val vector = cursor.collect[Vector](3)
80    * }}}
81    */
82   def collect[M[_]](upTo: Int = Int.MaxValue, stopOnError: Boolean = true)(implicit cbf: CanBuildFrom[M[_], T, M[T]], ec: ExecutionContext): Future[M[T]]
83 
84   /**
85    * Applies a binary operator to a start value and all responses handled  
86    * by this cursor, going first to last.
87    * 
88    * @tparam A the result type of the binary operator.
89    * @param z the start value.
90    * @param maxDocs the maximum number of documents to be read.
91    * @param suc the binary operator to be applied when the next response is successfully read.
92    * @param err the binary operator to be applied when failing to get the next response.
93    */
94   def foldResponses[A](z: => A, maxDocs: Int = Int.MaxValue)(suc: (A, Response) => Cursor.State[A], err: (A, Throwable) => Cursor.State[A])(implicit ctx: ExecutionContext): Future[A]
95 
96   /**
97    * Applies a binary operator to a start value and all bulks of documents
98    * retrieved by this cursor, going first to last.
99    * 
100    * @tparam A the result type of the binary operator.
101    * @param z the start value.
102    * @param maxDocs the maximum number of documents to be read.
103    * @param suc the binary operator to be applied when the next response is successfully read.
104    * @param err the binary operator to be applied when failing to get the next response.
105     */
106   def foldBulks[A](z: => A, maxDocs: Int = Int.MaxValue)(suc: (A, Iterator[T]) => Cursor.State[A], err: (A, Throwable) => Cursor.State[A])(implicit ctx: ExecutionContext): Future[A]
107 
108   /**
109    * Applies a binary operator to a start value and all elements retrieved 
110    * by this cursor, going first to last.
111    * 
112    * @tparam A the result type of the binary operator.
113    * @param z the start value.
114    * @param maxDocs the maximum number of documents to be read.
115    * @param suc the binary operator to be applied when the next document is successfully read.
116    * @param err the binary operator to be applied when failing to read the next document.
117    * 
118    * {{{
119    * cursor.foldWhile(Nil: Seq[Person])((s, p) => Cursor.Cont(s :+ p), 
120    *   { (l, e) => println("last valid value: " + l); Cursor.Fail(e) })
121    * }}}
122    */
123   def foldWhile[A](z: => A, maxDocs: Int = Int.MaxValue)(suc: (A, T) => Cursor.State[A], err: (A, Throwable) => Cursor.State[A])(implicit ctx: ExecutionContext): Future[A]
124 
125   /**
126    * Collects all the documents into a `List[T]`.
127    * Given the `stopOnError` parameter (which defaults to true), the resulting Future may fail if any
128    * non-fatal exception occurs. If set to false, all the documents that caused exceptions are skipped.
129    *
130    * @param upTo Collect up to `maxDocs` documents.
131    * @param stopOnError States if the Future should fail if any non-fatal exception occurs.
132    *
133    * Example:
134    * {{{
135    * val cursor = collection.find(query, filter).cursor[BSONDocument]
136    * // return the 3 first documents in a list.
137    * val list = cursor.toList(3)
138    * }}}
139    */
140   @deprecated("consider using collect[List] instead", "0.10.0")
141   def toList(upTo: Int = Int.MaxValue, stopOnError: Boolean = true)(implicit ctx: ExecutionContext): Future[List[T]] = collect[List](upTo, stopOnError)
142 
143   /**
144    * Gets the first document matching the query, if any.
145    * The resulting Future may fail if any exception occurs (for example, while deserializing the document).
146    *
147    * Example:
148    * {{{
149    * val cursor = collection.find(query, filter).cursor[BSONDocument]
150    * // return option of the first element.
151    * val first: Future[Option[BSONDocument]] = cursor.headOption
152    * }}}
153    */
154   def headOption(implicit ctx: ExecutionContext): Future[Option[T]] = collect[Iterable](1, true).map(_.headOption)
155 
156   /**
157    * Produces an Enumerator of responses from the database.
158    * An Enumeratee for error handling should be used to prevent silent failures.
159    * Consider using `enumerateResponses` instead.
160    *
161    * @param maxDocs Enumerate up to `maxDocs` documents.
162    *
163    * @return an Enumerator of Responses.
164    */
165   def rawEnumerateResponses(maxDocs: Int = Int.MaxValue)(implicit ctx: ExecutionContext): Enumerator[Response]
166 }

221 object Cursor {
222   private[api] val logger = LazyLogger("reactivemongo.api.Cursor")
223 
224   /**
225    * Flattens the given future [[reactivemongo.api.Cursor]] to a [[reactivemongo.api.FlattenedCursor]].
226    */
227   def flatten[T, C[_] <: Cursor[_]](future: Future[C[T]])(implicit fs: CursorFlattener[C]): C[T] = fs.flatten(future)
228 
229   sealed trait State[A]
230 
231   /** Continue with given value */
232   case class Cont[T](value: T) extends State[T]
233 
234   /** Successfully stop processing with given value */
235   case class Done[T](value: T) extends State[T]
236 
237   /** Ends processing due to failure of given `cause` */
238   case class Fail[T](cause: Throwable) extends State[T]
239 }

241 object DefaultCursor {
242   import Cursor.{ State, Cont, Done, Fail, logger }
243 
244   def apply[P <: SerializationPack, A](
245     pack: P,
246     query: Query,
247     documents: BufferSequence,
248     readPreference: ReadPreference,
249     mongoConnection: MongoConnection,
250     failoverStrategy: FailoverStrategy,
251     isMongo26WriteOp: Boolean)(implicit reader: pack.Reader[A]): Cursor[A] = new Cursor[A] {
252       private def next(response: Response)(implicit ctx: ExecutionContext): Future[Option[Response]] = {
253         if (response.reply.cursorID != 0) {
254           val op = GetMore(query.fullCollectionName, query.numberToReturn, response.reply.cursorID)
255           logger.trace("[Cursor] Calling next on " + response.reply.cursorID + ", op=" + op)
256           Failover2(mongoConnection, failoverStrategy) { () =>
257             mongoConnection.sendExpectingResponse(RequestMaker(op).copy(channelIdHint = Some(response.info.channelId)), isMongo26WriteOp)
258           }.future.map(Some(_))
259         } else {
260           logger.error("[Cursor] Call to next() but cursorID is 0, there is probably a bug")
261           Future.successful(Option.empty[Response])
262         }
263       }
264 
265       @inline
266       private def hasNext(response: Response): Boolean = response.reply.cursorID != 0
267 
268       @inline
269       private def hasNext(response: Response, maxDocs: Int): Boolean =
270         hasNext(response) && (response.reply.numberReturned + response.reply.startingFrom) < maxDocs
271 
272       @inline
273       private def makeIterator(response: Response) = ReplyDocumentIterator(pack)(response.reply, response.documents)
274 
275     @inline
276     private def makeRequest(implicit ctx: ExecutionContext): Future[Response] =
277       Failover2(mongoConnection, failoverStrategy) { () =>
278         mongoConnection.sendExpectingResponse(RequestMaker(query, documents, readPreference), isMongo26WriteOp)
279       }.future
280 
281     @inline
282     private def isTailable =
283       (query.flags & QueryFlags.TailableCursor) == QueryFlags.TailableCursor
284 
285     /** Returns next response using tailable mode */
286     private def tailResponse(current: Response, maxDocs: Int)(implicit context: ExecutionContext): Future[Option[Response]] = mongoConnection.killed flatMap {
287       case true =>
288         logger.warn("[tailResponse] Connection is killed")
289         Future.successful(Option.empty[Response])
290 
291       case _ =>
292         if (hasNext(current)) next(current)
293         else {
294           logger.debug("[tailResponse] Current cursor exhausted, renewing...")
295           DelayedFuture(500, mongoConnection.actorSystem).
296             flatMap(_ => makeRequest.map(Some(_)))
297         }
298     }
299 
300     @inline
301     private def killCursors(cursorID: Long, logCat: String): Unit =
302       if (cursorID != 0) {
303         logger.debug(s"[$logCat] Clean up ${cursorID}, sending KillCursor")
304         mongoConnection.send(RequestMaker(KillCursors(Set(cursorID))))
305       } else logger.trace(s"[$logCat] Cursor exhausted (${cursorID})")
306 
307     def foldResponses[T](z: => T, maxDocs: Int = Int.MaxValue)(suc: (T, Response) => State[T], err: (T, Throwable) => State[T])(implicit ctx: ExecutionContext): Future[T] = new FoldResponses(z, maxDocs, suc, err)(ctx)()    
308 
309     def foldBulks[T](z: => T, maxDocs: Int = Int.MaxValue)(suc: (T, Iterator[A]) => State[T], err: (T, Throwable) => State[T])(implicit ctx: ExecutionContext): Future[T] = foldResponses(z, maxDocs)({ (s, r) =>
310       Try(makeIterator(r)) match {
311         case Success(it) => suc(s, it)
312         case Failure(e) => Fail(e)
313       }
314     }, err)
315 
316     def foldWhile[T](z: => T, maxDocs: Int = Int.MaxValue)(suc: (T, A) => State[T], err: (T, Throwable) => State[T])(implicit ctx: ExecutionContext): Future[T] = {
317       def process(cur: T, st: State[T])(op: T => State[T]): State[T] =
318         st match {
319           case Cont(v)  => op(v)
320           case f @ Fail(_) => f
321           case _ => st
322         }
323 
324       def go(it: Iterator[A])(v: T): State[T] =
325         if (!it.hasNext) Cont(v)
326         else Try(it.next) match {
327           case Failure(x @ ReplyDocumentIteratorExhaustedException(_)) =>
328             Fail(x)
329           case Failure(e) => process(v, err(v, e))(go(it))
330           case Success(a) => process(v, suc(v, a))(go(it))
331         }
332 
333       foldBulks(z, maxDocs)({ (cur, it) => go(it)(cur) }, err)
334     }
335 
336     def simpleCursorEnumerateResponses(maxDocs: Int = Int.MaxValue)(implicit ctx: ExecutionContext): Enumerator[Response] =
337       Enumerator.flatten(makeRequest.map(new CustomEnumerator.SEnumerator(_)(
338         next = response => {
339           if (hasNext(response, maxDocs)) next(response)
340           else Future.successful(Option.empty[Response])
341         }, cleanUp = { resp => killCursors(resp.reply.cursorID, "Cursor") })))
342 
343     def tailableCursorEnumerateResponses(maxDocs: Int = Int.MaxValue)(implicit ctx: ExecutionContext): Enumerator[Response] =
344       Enumerator.flatten(makeRequest.map { response =>
345         new CustomEnumerator.SEnumerator(response -> 0)(
346           next = { current =>
347             val (r, c) = current
348             if (c < maxDocs) {
349               tailResponse(r, maxDocs).map(_.map((_, c+r.reply.numberReturned)))
350             } else Future.successful(Option.empty[(Response, Int)])
351           },
352           cleanUp = { current =>
353             val (r, _) = current
354             killCursors(r.reply.cursorID, "Tailable Cursor")
355           }).map(_._1)
356       })
357 
358       def rawEnumerateResponses(maxDocs: Int = Int.MaxValue)(implicit ctx: ExecutionContext): Enumerator[Response] =
359         if (isTailable) tailableCursorEnumerateResponses(maxDocs) else simpleCursorEnumerateResponses(maxDocs)
360 
361       def enumerateResponses(maxDocs: Int = Int.MaxValue, stopOnError: Boolean = false)(implicit ctx: ExecutionContext): Enumerator[Response] =
362         rawEnumerateResponses(maxDocs) &> {
363           if (stopOnError)
364             CustomEnumeratee.stopOnError
365           else CustomEnumeratee.recover {
366             new CustomEnumeratee.RecoverFromErrorFunction {
367               def apply[E, A](throwable: Throwable, input: Input[E], continue: () => Iteratee[E, A]): Iteratee[E, A] = throwable match {
368                 case e: ReplyDocumentIteratorExhaustedException =>
369                   val errstr = "ReplyDocumentIterator exhausted! " +
370                     "Was this enumerator applied to many iteratees concurrently? " +
371                     "Stopping to prevent infinite recovery."
372                   logger.error(errstr, e)
373                   Error(errstr, input)
374                 case e =>
375                   logger.debug("There was an exception during the stream, dropping it since stopOnError is false", e)
376                   continue()
377               }
378             }
379           }
380         }
381 
382       def enumerateBulks(maxDocs: Int = Int.MaxValue, stopOnError: Boolean = false)(implicit ctx: ExecutionContext): Enumerator[Iterator[A]] =
383         enumerateResponses(maxDocs, stopOnError) &> Enumeratee.map(makeIterator)
384 
385       def enumerate(maxDocs: Int = Int.MaxValue, stopOnError: Boolean = false)(implicit ctx: ExecutionContext): Enumerator[A] = {
386         @tailrec
387         def next(it: Iterator[A], stopOnError: Boolean): Option[Try[A]] = {
388           if (it.hasNext) {
389             val tried = Try(it.next)
390             if (tried.isFailure && !stopOnError)
391               next(it, stopOnError)
392             else Some(tried)
393           } else None
394         }
395         enumerateResponses(maxDocs, stopOnError) &> Enumeratee.mapFlatten { response =>
396           val iterator = ReplyDocumentIterator(pack)(response.reply, response.documents)
397           if (!iterator.hasNext) Enumerator.empty
398           else CustomEnumerator.SEnumerator(iterator.next) { _ =>
399             next(iterator, stopOnError).
400               fold(Future.successful(Option.empty[A])) {
401                 case Success(mt) => Future.successful(Some(mt))
402                 case Failure(e)  => Future.failed(e)
403               }
404           }
405         }
406       }
407 
408       def collect[M[_]](upTo: Int = Int.MaxValue, stopOnError: Boolean = true)(implicit cbf: CanBuildFrom[M[_], A, M[A]], ctx: ExecutionContext): Future[M[A]] = {
409         (enumerateResponses(upTo, stopOnError) |>>> Iteratee.fold(cbf.apply) { (builder, response) =>
410 
411           def tried[A](it: Iterator[A]) = new Iterator[Try[A]] { def hasNext = it.hasNext; def next = Try(it.next) }
412 
413           logger.trace(s"[collect] got response $response")
414 
415           val filteredIterator =
416             if (!stopOnError)
417               tried(makeIterator(response)).filter(_.isSuccess).map(_.get)
418             else makeIterator(response)
419           val iterator =
420             if (upTo < response.reply.numberReturned + response.reply.startingFrom)
421               filteredIterator.take(upTo - response.reply.startingFrom)
422             else filteredIterator
423           builder ++= iterator
424         }).map(_.result)
425       }
426 
427     private class FoldResponses[T](z: => T, maxDocs: Int,
428       suc: (T, Response) => State[T], err: (T, Throwable) => State[T])(
429       implicit ctx: ExecutionContext) {
430 
431       val nextResponse: (Response, Int) => Future[Option[Response]] =
432         if (!isTailable) { (r: Response, maxDocs: Int) =>
433           if (!hasNext(r, maxDocs)) Future.successful(Option.empty[Response])
434           else next(r)
435         } else (r: Response, maxDocs: Int) => tailResponse(r, maxDocs)
436 
437       def process(cur: T, st: State[T])(op: T => Future[T]): Future[T] =
438         st match {
439           case Done(v) => Future.successful(v)
440           case Cont(v) => op(v)
441           case Fail(x) => Future.failed[T](x)
442         }
443 
444       def go(r: Response, c: Int)(v: T): Future[T] = nextResponse(r, c).flatMap(
445         _.fold(Future successful v)(x =>
446           procResponses(Future.successful(x), v, c)))
447 
448       def procResponses(done: Future[Response], cur: T, c: Int): Future[T] =
449         done flatMap { r =>
450           val st = suc(cur, r) match {
451             case end @ (Done(_) | Fail(_)) => {
452               // Releases cursor before ending
453               killCursors(r.reply.cursorID, "FoldResponses")
454               end
455             }
456             case Cont(v) if (c == maxDocs) => Done(v)
457             case state @ _ => state
458           }
459 
460           process(cur, st)(go(r, c + r.reply.numberReturned))
461         }
462 
463       def apply(): Future[T] = procResponses(makeRequest, z, 0)
464     }
465   }
