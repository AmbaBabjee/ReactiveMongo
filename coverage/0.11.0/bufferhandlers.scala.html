21 trait BufferHandler {
22   def serialize(bson: BSONValue, buffer: WritableBuffer): WritableBuffer
23   def deserialize(buffer: ReadableBuffer): Try[(String, BSONValue)]
24 
25   def write(buffer: WritableBuffer, document: BSONDocument) = {
26     serialize(document, buffer)
27   }
28 
29   def write(buffer: WritableBuffer, arr: BSONArray) = {
30     serialize(arr, buffer)
31   }
32 
33   def readDocument(buffer: ReadableBuffer): Try[BSONDocument]
34 
35   def writeDocument(document: BSONDocument, buffer: WritableBuffer): WritableBuffer
36 
37   def stream(buffer: ReadableBuffer): Stream[(String, BSONValue)] = {
38     val elem = deserialize(buffer)
39     if (elem.isSuccess)
40       elem.get #:: stream(buffer)
41     else Stream.empty
42   }
43 }

45 object DefaultBufferHandler extends BufferHandler {
46   sealed trait BufferWriter[B <: BSONValue] {
47     def write(value: B, buffer: WritableBuffer): WritableBuffer
48   }
49 
50   sealed trait BufferReader[B <: BSONValue] {
51     def read(buffer: ReadableBuffer): B
52   }

54   sealed trait BufferRW[B <: BSONValue] extends BufferWriter[B] with BufferReader[B]
55 
56   val handlersByCode: Map[Byte, BufferRW[_ <: BSONValue]] = Map(
57     0x01.toByte -> BSONDoubleBufferHandler,
58     0x02.toByte -> BSONStringBufferHandler,
59     0x03.toByte -> BSONDocumentBufferHandler,
60     0x04.toByte -> BSONArrayBufferHandler, // array
61     0x05.toByte -> BSONBinaryBufferHandler, // binary TODO
62     0x06.toByte -> BSONUndefinedBufferHandler, // undefined,
63     0x07.toByte -> BSONObjectIDBufferHandler, // objectid,
64     0x08.toByte -> BSONBooleanBufferHandler, // boolean
65     0x09.toByte -> BSONDateTimeBufferHandler, // datetime
66     0x0A.toByte -> BSONNullBufferHandler, // null
67     0x0B.toByte -> BSONRegexBufferHandler, // regex
68     0x0C.toByte -> BSONDBPointerBufferHandler, // dbpointer
69     0x0D.toByte -> BSONJavaScriptBufferHandler, // JS
70     0x0E.toByte -> BSONSymbolBufferHandler, // symbol
71     0x0F.toByte -> BSONJavaScriptWSBufferHandler, // JS with scope
72     0x10.toByte -> BSONIntegerBufferHandler,
73     0x11.toByte -> BSONTimestampBufferHandler, // timestamp,
74     0x12.toByte -> BSONLongBufferHandler, // long,
75     0xFF.toByte -> BSONMinKeyBufferHandler, // min
76     0x7F.toByte -> BSONMaxKeyBufferHandler
77   ) // max

79   object BSONDoubleBufferHandler extends BufferRW[BSONDouble] {
80     def write(value: BSONDouble, buffer: WritableBuffer): WritableBuffer = buffer.writeDouble(value.value)
81     def read(buffer: ReadableBuffer): BSONDouble = BSONDouble(buffer.readDouble)
82   }
83   object BSONStringBufferHandler extends BufferRW[BSONString] {
84     def write(value: BSONString, buffer: WritableBuffer): WritableBuffer = buffer.writeString(value.value)
85     def read(buffer: ReadableBuffer): BSONString = BSONString(buffer.readString)
86   }
87   object BSONDocumentBufferHandler extends BufferRW[BSONDocument] {
88     def write(doc: BSONDocument, buffer: WritableBuffer) = {
89       val now = buffer.index
90       buffer.writeInt(0)
91       doc.elements.foreach { e =>
92         buffer.writeByte(e.value.code.toByte)
93         buffer.writeCString(e.name)
94         serialize(e.value, buffer)
95       }
96       buffer.setInt(now, (buffer.index - now + 1))
97       buffer.writeByte(0)
98       buffer
99     }
100     def read(b: ReadableBuffer) = {
101       val startIndex = b.index
102       val length = b.readInt
103       val buffer = b.slice(length - 4)
104       b.discard(length - 4)
105       def makeStream(): Stream[Try[BSONElement]] = {
106         if (buffer.readable > 1) { // last is 0
107           val code = buffer.readByte
108           val name = buffer.readCString
109           val elem = Try(BSONElement(name, DefaultBufferHandler.handlersByCode.get(code).map(_.read(buffer)).get))
110           elem #:: makeStream
111         } else Stream.empty
112       }
113       val stream = makeStream
114       stream.force // TODO remove
115       new BSONDocument(stream)
116     }
117   }
118   object BSONArrayBufferHandler extends BufferRW[BSONArray] {
119     def write(array: BSONArray, buffer: WritableBuffer) = {
120       val now = buffer.index
121       buffer.writeInt(0)
122       array.values.zipWithIndex.foreach { e =>
123         buffer.writeByte(e._1.code.toByte)
124         buffer.writeCString(e._2.toString)
125         serialize(e._1, buffer)
126       }
127       buffer.setInt(now, (buffer.index - now + 1))
128       buffer.writeByte(0)
129       buffer
130     }
131     def read(b: ReadableBuffer) = {
132       val startIndex = b.index
133       val length = b.readInt
134       val buffer = b.slice(length - 4)
135       b.discard(length - 4)
136       def makeStream(): Stream[Try[BSONValue]] = {
137         if (buffer.readable > 1) { // last is 0
138           val code = buffer.readByte
139           val name = buffer.readCString
140           val elem = Try(DefaultBufferHandler.handlersByCode.get(code).map(_.read(buffer)).get)
141           elem #:: makeStream
142         } else Stream.empty
143       }
144       val stream = makeStream
145       stream.force // TODO remove
146       new BSONArray(stream)
147     }
148   }
149   object BSONBinaryBufferHandler extends BufferRW[BSONBinary] {
150     def write(binary: BSONBinary, buffer: WritableBuffer) = {
151       buffer.writeInt(binary.value.readable)
152       buffer.writeByte(binary.subtype.value.toByte)
153       val bin = binary.value.slice(binary.value.readable)
154       buffer.writeBytes(bin.readArray(bin.readable)) // TODO
155       buffer
156     }
157     def read(buffer: ReadableBuffer) = {
158       val length = buffer.readInt
159       val subtype = Subtype.apply(buffer.readByte)
160       val bin = buffer.slice(length)
161       buffer.discard(length)
162       BSONBinary(bin, subtype)
163     }
164   }

226   def serialize(bson: BSONValue, buffer: WritableBuffer): WritableBuffer = {
227     handlersByCode.get(bson.code).get.asInstanceOf[BufferRW[BSONValue]].write(bson, buffer)
228   }
229 
230   def deserialize(buffer: ReadableBuffer): Try[(String, BSONValue)] = Try {
231     if (buffer.readable > 0) {
232       val code = buffer.readByte
233       buffer.readString -> handlersByCode.get(code).map(_.read(buffer)).get
234     } else throw new NoSuchElementException("buffer can not be read, end of buffer reached")
235   }
236 
237   def readDocument(buffer: ReadableBuffer): Try[BSONDocument] = Try {
238     BSONDocumentBufferHandler.read(buffer)
239   }
240 
241   def writeDocument(document: BSONDocument, buffer: WritableBuffer): WritableBuffer =
242     serialize(document, buffer)

245 sealed trait BSONIterator extends Iterator[(String, BSONValue)] {
246   val buffer: ReadableBuffer
247 
248   val startIndex = buffer.index
249   val documentSize = buffer.readInt
250 
251   def next: (String, BSONValue) = {
252     val code = buffer.readByte
253     buffer.readString -> DefaultBufferHandler.handlersByCode.get(code).map(_.read(buffer)).get
254   }
255 
256   def hasNext = buffer.index - startIndex + 1 < documentSize
257 
258   def mapped: Map[String, BSONElement] =
259     (for (el <- this) yield (el._1, BSONElement(el._1, el._2))).toMap
260 }

262 object BSONIterator {
263   private[bson] def pretty(i: Int, it: Iterator[Try[BSONElement]]): String = {
264     val prefix = (0 to i).map { i => "  " }.mkString("")
265     (for (tryElem <- it) yield {
266       tryElem match {
267         case Success(elem) => elem.value match {
268           case array: BSONArray  => prefix + elem.name + ": [\n" + pretty(i + 1, array.elements.map(Success(_)).iterator) + "\n" + prefix + "]"
269 
270           case doc: BSONDocument => prefix + elem.name + ": {\n" + pretty(i + 1, doc.stream.iterator) + "\n" + prefix + "}"
271 
272           case BSONString(s) =>
273             prefix + elem.name + ": \"" + s.replaceAll("\"", "\\\"") + '"'
274 
275           case _ => prefix + elem.name + ": " + elem.value.toString
276         }
277         case Failure(e) => s"${prefix}ERROR[${e.getMessage()}]"
278       }
279     }).mkString(",\n")
280   }
281   /** Makes a pretty String representation of the given iterator of BSON elements. */
282   def pretty(it: Iterator[Try[BSONElement]]): String = "{\n" + pretty(0, it) + "\n}"
