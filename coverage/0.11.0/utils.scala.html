16 package reactivemongo.bson.utils

18 /** Common functions */
19 object Converters {
20   private val HEX_CHARS: Array[Char] = "0123456789abcdef".toCharArray
21 
22   /** Turns an array of Byte into a String representation in hexadecimal. */
23   def hex2Str(bytes: Array[Byte]): String = {
24     val hex = new Array[Char](2 * bytes.length)
25     var i = 0
26     while (i < bytes.length) {
27       hex(2 * i) = HEX_CHARS((bytes(i) & 0xF0) >>> 4)
28       hex(2 * i + 1) = HEX_CHARS(bytes(i) & 0x0F)
29       i = i + 1
30     }
31     new String(hex)
32   }
33 
34   /** Turns a hexadecimal String into an array of Byte. */
35   def str2Hex(str: String): Array[Byte] = {
36     val bytes = new Array[Byte](str.length / 2)
37     var i = 0
38     while (i < bytes.length) {
39       bytes(i) = Integer.parseInt(str.substring(2 * i, 2 * i + 2), 16).toByte
40       i += 1
41     }
42     bytes
43   }
44 
45   /** Computes the MD5 hash of the given String. */
46   def md5(s: String) = java.security.MessageDigest.getInstance("MD5").digest(s.getBytes)
47 
48   /** Computes the MD5 hash of the given Array of Bytes. */
49   def md5(array: Array[Byte]) = java.security.MessageDigest.getInstance("MD5").digest(array)
50 
51   /** Computes the MD5 hash of the given String and turns it into a hexadecimal String representation. */
52   def md5Hex(s: String): String = hex2Str(md5(s))
53 }
