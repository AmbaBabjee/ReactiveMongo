120 /**
121  * A `BSONArray` structure (BSON type `0x04`).
122  *
123  * A `BSONArray` is a straightforward `BSONDocument` where keys are a sequence of positive integers.
124  *
125  * A `BSONArray` is basically a stream of tuples `(String, BSONValue)` where the first member is a string representation of an index.
126  * It is completely lazy. The stream it wraps is a `Stream[Try[(String, BSONValue)]]` since
127  * we cannot be sure that a not yet deserialized value will be processed without error.
128  */
129 case class BSONArray(stream: Stream[Try[BSONValue]])
130     extends BSONValue with BSONElementSet {
131 
132   val code = 0x04.toByte
133 
134   type SetType = BSONArray

136   /**
137    * Returns the [[BSONValue]] at the given `index`.
138    *
139    * If there is no such `index` or the matching value cannot be deserialized, returns `None`.
140    */
141   def get(index: Int): Option[BSONValue] = getTry(index).toOption
142 
143   /**
144    * Returns the [[BSONValue]] matching the given `name`,
145    * provided it is a valid string representation of a valid index.
146    */
147   def get(name: String): Option[BSONValue] = try {
148     get(name.toInt)
149   } catch {
150     case _: Throwable => None
151   }
152 
153   /** Returns true if the given `name` corresponds to a valid index. */
154   def contains(name: String): Boolean = get(name).isDefined

156   def headOption: Option[BSONElement] = stream.collectFirst {
157     case Success(v) => BSONElement("0", v)
158   }
159 
160   /**
161    * Returns the [[BSONValue]] at the given `index`.
162    *
163    * If there is no such `index` or the matching value cannot be deserialized, returns a `Failure`.
164    * The `Failure` holds a [[exceptions.DocumentKeyNotFound]] if the key could not be found.
165    */
166   def getTry(index: Int): Try[BSONValue] = stream.drop(index).headOption.
167     getOrElse(Failure(DocumentKeyNotFound(index.toString)))
168 
169   /**
170    * Returns the [[BSONValue]] at the given `index`.
171    *
172    * If there is no such `index`, the resulting option will be `None`.
173    * If the matching value could not be deserialized, returns a `Failure`.
174    */
175   def getUnflattenedTry(index: Int): Try[Option[BSONValue]] = getTry(index) match {
176     case Failure(e: DocumentKeyNotFound) => Success(None)
177     case Failure(e)                      => Failure(e)
178     case Success(e)                      => Success(Some(e))
179   }
180 
181   /**
182    * Gets the [[BSONValue]] at the given `index`, and converts it with the given implicit [[BSONReader]].
183    *
184    * If there is no matching value, or the value could not be deserialized or converted, returns a `None`.
185    */
186   def getAs[T](index: Int)(implicit reader: BSONReader[_ <: BSONValue, T]): Option[T] = {
187     getTry(index).toOption.flatMap { element =>
188       Try(reader.asInstanceOf[BSONReader[BSONValue, T]].read(element)).toOption
189     }
190   }
191 
192   /**
193    * Gets the [[BSONValue]] at the given `index`,
194    * and converts it with the given implicit [[BSONReader]].
195    *
196    * If there is no matching value, or the value could not be deserialized or converted, returns a `Failure`.
197    * The `Failure` holds a [[exceptions.DocumentKeyNotFound]] if the key could not be found.
198    */
199   def getAsTry[T](index: Int)(implicit reader: BSONReader[_ <: BSONValue, T]): Try[T] = {
200     val tt = getTry(index)
201     tt.flatMap { element => Try(reader.asInstanceOf[BSONReader[BSONValue, T]].read(element)) }
202   }
203 
204   /**
205    * Gets the [[BSONValue]] at the given `index`,
206    * and converts it with the given implicit [[BSONReader]].
207    *
208    * If there is no matching value, returns a `Success` holding `None`.
209    * If the value could not be deserialized or converted, returns a `Failure`.
210    */
211   def getAsUnflattenedTry[T](index: Int)(implicit reader: BSONReader[_ <: BSONValue, T]): Try[Option[T]] = getAsTry(index)(reader) match {
212     case Failure(e: DocumentKeyNotFound) => Success(None)
213     case Failure(e)                      => Failure(e)
214     case Success(e)                      => Success(Some(e))

369 object BSONObjectID {
370   private val maxCounterValue = 16777216
371   private val increment = new java.util.concurrent.atomic.AtomicInteger(scala.util.Random.nextInt(maxCounterValue))
372 
373   private def counter = (increment.getAndIncrement + maxCounterValue) % maxCounterValue
374 
375   /**
376    * The following implemtation of machineId work around openjdk limitations in
377    * version 6 and 7
378    *
379    * Openjdk fails to parse /proc/net/if_inet6 correctly to determine macaddress
380    * resulting in SocketException thrown.
381    *
382    * Please see:
383    * * https://github.com/openjdk-mirror/jdk7u-jdk/blob/feeaec0647609a1e6266f902de426f1201f77c55/src/solaris/native/java/net/NetworkInterface.c#L1130
384    * * http://lxr.free-electrons.com/source/net/ipv6/addrconf.c?v=3.11#L3442
385    * * http://lxr.free-electrons.com/source/include/linux/netdevice.h?v=3.11#L1130
386    * * http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=7078386
387    *
388    * and fix in openjdk8:
389    * * http://hg.openjdk.java.net/jdk8/tl/jdk/rev/b1814b3ea6d3
390    */
391 
392   private val machineId = {
393     import java.net._
394     val validPlatform = Try {
395       val correctVersion = System.getProperty("java.version").substring(0, 3).toFloat >= 1.8
396       val noIpv6 = System.getProperty("java.net.preferIPv4Stack") == true
397       val isLinux = System.getProperty("os.name") == "Linux"
398 
399       !isLinux || correctVersion || noIpv6
400     }.getOrElse(false)
401 
402     // Check java policies
403     val permitted = {
404       val sec = System.getSecurityManager();
405       Try { sec.checkPermission(new NetPermission("getNetworkInformation")) }.toOption.map(_ => true).getOrElse(false);
406     }
407 
408     if (validPlatform && permitted) {
409       val networkInterfacesEnum = NetworkInterface.getNetworkInterfaces
410       val networkInterfaces = scala.collection.JavaConverters.enumerationAsScalaIteratorConverter(networkInterfacesEnum).asScala
411       val ha = networkInterfaces.find(ha => Try(ha.getHardwareAddress).isSuccess && ha.getHardwareAddress != null && ha.getHardwareAddress.length == 6)
412         .map(_.getHardwareAddress)
413         .getOrElse(InetAddress.getLocalHost.getHostName.getBytes)
414       Converters.md5(ha).take(3)
415     } else {
416       val threadId = Thread.currentThread.getId.toInt
417       val arr = new Array[Byte](3)
418 
419       arr(0) = (threadId & 0xFF).toByte
420       arr(1) = (threadId >> 8 & 0xFF).toByte
421       arr(2) = (threadId >> 16 & 0xFF).toByte
422 
423       arr
424     }
425   }
426 
427   /**
428    * Constructs a BSON ObjectId element from a hexadecimal String representation.
429    * Throws an exception if the given argument is not a valid ObjectID.
430    *
431    * `parse(str: String): Try[BSONObjectID]` should be considered instead of this method.
432    */
433   def apply(id: String): BSONObjectID = {
434     if (id.length != 24)
435       throw new IllegalArgumentException(s"wrong ObjectId: '$id'")
436     /** Constructs a BSON ObjectId element from a hexadecimal String representation */
437     new BSONObjectID(Converters.str2Hex(id))
438   }
439 
440   def apply(array: Array[Byte]): BSONObjectID = {
441     if (array.length != 12)
442       throw new IllegalArgumentException(s"wrong byte array for an ObjectId (size ${array.length})")
443     new BSONObjectID(java.util.Arrays.copyOf(array, 12))
444   }
445 
446   def unapply(id: BSONObjectID): Option[Array[Byte]] = Some(id.valueAsArray)
447 
448   /** Tries to make a BSON ObjectId element from a hexadecimal String representation. */
449   def parse(str: String): Try[BSONObjectID] = Try(apply(str))
450 
451   /**
452    * Generates a new BSON ObjectID.
453    *
454    * +------------------------+------------------------+------------------------+------------------------+
455    * + timestamp (in seconds) +   machine identifier   +    thread identifier   +        increment       +
456    * +        (4 bytes)       +        (3 bytes)       +        (2 bytes)       +        (3 bytes)       +
457    * +------------------------+------------------------+------------------------+------------------------+
458    *
459    * The returned BSONObjectID contains a timestamp set to the current time (in seconds),
460    * with the `machine identifier`, `thread identifier` and `increment` properly set.
461    */
462   def generate(): BSONObjectID = fromTime(System.currentTimeMillis, false)
463 
464   /**
465    * Generates a new BSON ObjectID from the given timestamp in milliseconds.
466    *
467    * +------------------------+------------------------+------------------------+------------------------+
468    * + timestamp (in seconds) +   machine identifier   +    thread identifier   +        increment       +
469    * +        (4 bytes)       +        (3 bytes)       +        (2 bytes)       +        (3 bytes)       +
470    * +------------------------+------------------------+------------------------+------------------------+
471    *
472    * The included timestamp is the number of seconds since epoch, so a BSONObjectID time part has only
473    * a precision up to the second. To get a reasonably unique ID, you _must_ set `onlyTimestamp` to false.
474    *
475    * Crafting a BSONObjectID from a timestamp with `fillOnlyTimestamp` set to true is helpful for range queries,
476    * eg if you want of find documents an _id field which timestamp part is greater than or lesser than
477    * the one of another id.
478    *
479    * If you do not intend to use the produced BSONObjectID for range queries, then you'd rather use
480    * the `generate` method instead.
481    *
482    * @param fillOnlyTimestamp if true, the returned BSONObjectID will only have the timestamp bytes set; the other will be set to zero.
483    */
484   def fromTime(timeMillis: Long, fillOnlyTimestamp: Boolean = true): BSONObjectID = {
485     // n of seconds since epoch. Big endian
486     val timestamp = (timeMillis / 1000).toInt
487     val id = new Array[Byte](12)
488 
489     id(0) = (timestamp >>> 24).toByte
490     id(1) = (timestamp >> 16 & 0xFF).toByte
491     id(2) = (timestamp >> 8 & 0xFF).toByte
492     id(3) = (timestamp & 0xFF).toByte
493 
494     if (!fillOnlyTimestamp) {
495       // machine id, 3 first bytes of md5(macadress or hostname)
496       id(4) = machineId(0)
497       id(5) = machineId(1)
498       id(6) = machineId(2)
499 
500       // 2 bytes of the pid or thread id. Thread id in our case. Low endian
501       val threadId = Thread.currentThread.getId.toInt
502       id(7) = (threadId & 0xFF).toByte
503       id(8) = (threadId >> 8 & 0xFF).toByte
504 
505       // 3 bytes of counter sequence, which start is randomized. Big endian
506       val c = counter
507       id(9) = (c >> 16 & 0xFF).toByte
508       id(10) = (c >> 8 & 0xFF).toByte
509       id(11) = (c & 0xFF).toByte
510     }
511 
512     BSONObjectID(id)
513   }
514 }
