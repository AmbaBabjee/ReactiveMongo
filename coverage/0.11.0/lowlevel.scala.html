38 class LowLevelBsonDocReader[A <: ReadableBuffer](rbuf: A) {
39   private val start = rbuf.index
40   private val length = {
41     val res = rbuf.readInt
42     rbuf.index = start
43     res
44   }
45   private val end = start + length
46   private def slice = rbuf.slice(length)
47 
48   type -> [A, B] = (A, B)
49 
50   def lookup(name: String): Option[Field] =
51     fieldStream.find(_.name == name)
52 
53   def fieldStream(): Stream[Field] = {
54     val buf = slice
55     buf.index = buf.index + 4
56 
57     def skipCString(): Unit =
58       while (buf.readByte != 0x00) {}
59 
60     @inline def move(diff: Int): Unit =
61       buf.index = buf.index + diff
62 
63     def stream(): Stream[Field] = {
64       val tpe = buf.readByte
65       val name = buf.readCString
66       val field = tpe match {
67         case 0x01 =>
68           DoubleField(name, buf.readDouble)
69         case 0x09 | 0x11 | 0x12 =>
70           LongField(tpe, name, buf.readLong)
71         case 0x08 =>
72           BooleanField(name, buf.readByte == 0x01)
73         case 0x10 =>
74           IntField(name, buf.readInt)
75         case 0x03 | 0x04 =>
76           val length = buf.readInt
77           buf.index = buf.index - 4
78           val res = StructureField(tpe, name, new LowLevelBsonDocReader(buf.slice(length)))
79           buf.index += length
80           res
81         case 0x05 =>
82           val length = buf.readInt + 4 + 1
83           buf.index = buf.index - 4
84           val res = LazyField(tpe, name, buf.slice(length))
85           buf.index = buf.index + length
86           res
87         //buf.index = buf.index + buf.readInt + 5
88         case 0x0B =>
89           val now = buf.index
90           skipCString(); skipCString()
91           val length = buf.index - now
92           buf.index = now
93           val res = LazyField(tpe, name, buf.slice(length))
94           buf.index = buf.index + length
95           res
96         case 0x02 | 0x0D | 0x0E =>
97           val length = buf.readInt + 4
98           buf.index = buf.index - 4
99           val res = LazyField(tpe, name, buf.slice(length))
100           buf.index = buf.index + length
101           res
102         case 0x0C =>
103           val length = buf.readInt + 12 + 4
104           buf.index = buf.index - 4
105           val res = LazyField(tpe, name, buf.slice(length))
106           buf.index = buf.index + length
107           res
108         case 0x07 =>
109           val res = LazyField(tpe, name, buf.slice(12))
110           buf.index = buf.index + 12
111           res
112         case 0x0F =>
113           // TODO
114           ???
115         case 0x06 | 0x0A | 0xFF | 0x7F =>
116           NoValue(tpe, name)
117         case x => throw new RuntimeException(s"unexpected type $x")
118       }
119       if (buf.readable > 1)
120         field #:: stream()
121       else Stream(field)
122     }
123 
124     stream
125   }
