20 /**
21  * A writable buffer.
22  *
23  * The implementation '''MUST''' ensure it stores data in little endian when needed.
24  */
25 trait WritableBuffer { self =>
26   /** Returns the current write index of this buffer. */
27   def index: Int
28 
29   /** Replaces 4 bytes at the given `index` by the given `value` */
30   def setInt(index: Int, value: Int): WritableBuffer
31 
32   /** Writes the bytes stored in the given `array` into this buffer. */
33   def writeBytes(array: Array[Byte]): self.type
34 
35   /** Writes the bytes stored in the given `buffer` into this buffer. */
36   def writeBytes(buffer: ReadableBuffer): self.type = {
37     @annotation.tailrec
38     def write(buffer: ReadableBuffer): self.type = {
39       if (buffer.readable > 1024) {
40         writeBytes(buffer.readArray(1024))
41         write(buffer)
42       } else writeBytes(buffer.readArray(buffer.readable))
43     }
44 
45     write(buffer.slice(buffer.readable))
46   }
47 
48   /** Writes the given `Byte` into this buffer. */
49   def writeByte(byte: Byte): WritableBuffer
50 
51   /** Writes the given `Int` into this buffer. */
52   def writeInt(int: Int): WritableBuffer
53 
54   /** Writes the given `Long` into this buffer. */
55   def writeLong(long: Long): WritableBuffer
56 
57   /** Writes the given `Double` into this buffer. */
58   def writeDouble(double: Double): WritableBuffer
59 
60   def toReadableBuffer(): ReadableBuffer
61 
62   /** Write a UTF-8 encoded C-Style String. */
63   def writeCString(s: String): WritableBuffer = {
64     val bytes = s.getBytes("utf-8")
65     writeBytes(bytes).writeByte(0)
66   }
67 
68   /** Write a UTF-8 encoded String. */
69   def writeString(s: String): WritableBuffer = {
70     val bytes = s.getBytes("utf-8")
71     writeInt(bytes.size + 1).writeBytes(bytes).writeByte(0)
72   }
73 }
74 
75 /**
76  * A readable buffer.
77  *
78  * The implementation '''MUST''' ensure it reads data in little endian when needed.
79  */
80 trait ReadableBuffer {
81   /** Returns the current read index of this buffer. */
82   def index: Int
83 
84   def index_=(i: Int): Unit
85 
86   /** Sets the read index to `index + n` (in other words, skips `n` bytes). */
87   def discard(n: Int): Unit
88 
89   /** Fills the given array with the bytes read from this buffer. */
90   def readBytes(bytes: Array[Byte]): Unit
91 
92   /** Reads a `Byte` from this buffer. */
93   def readByte(): Byte
94 
95   /** Reads an `Int` from this buffer. */
96   def readInt(): Int
97 
98   /** Reads a `Long` from this buffer. */
99   def readLong(): Long
100 
101   /** Reads a `Double` from this buffer. */
102   def readDouble(): Double
103 
104   /** Returns the number of readable remaining bytes of this buffer. */
105   def readable(): Int
106 
107   def toWritableBuffer: WritableBuffer
108 
109   /**
110    * Returns a new instance of ReadableBuffer which starts at the current index and contains `n` bytes.
111    *
112    * This method does not update the read index of the original buffer.
113    */
114   def slice(n: Int): ReadableBuffer
115 
116   /** Returns the buffer size. */
117   def size: Int
118 
119   /** Reads a UTF-8 String. */
120   def readString(): String = {
121     val bytes = new Array[Byte](this.readInt - 1)
122     this.readBytes(bytes)
123     this.readByte
124     new String(bytes, "UTF-8")
125   }
126 
127   /**
128    * Reads an array of Byte of the given length.
129    *
130    * @param length Length of the newly created array.
131    */
132   def readArray(length: Int): Array[Byte] = {
133     val bytes = new Array[Byte](length)
134     this.readBytes(bytes)
135     bytes
136   }
137 
138   /** Reads a UTF-8 C-Style String. */
139   def readCString(): String = readCString(new ArrayBuffer[Byte](16))
140 
141   @scala.annotation.tailrec
142   private def readCString(array: ArrayBuffer[Byte]): String = {
143     val byte = this.readByte
144 
145     if (byte == 0x00) {
146       new String(array.toArray, "UTF-8")
147     } else readCString(array += byte)
148   }
149 
150   def duplicate(): ReadableBuffer
151 }

158 case class ArrayReadableBuffer private (
