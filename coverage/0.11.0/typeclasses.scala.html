18 import scala.math.Numeric

20 sealed trait BSONNumberLikeClass[B <: BSONValue] extends BSONNumberLike
21 
22 /**
23  * A BSON value that can be seen as a number.
24  *
25  * Conversions:
26  *   - [[BSONLong]]
27  *   - [[BSONInteger]]
28  *   - [[BSONDouble]]
29  */
30 sealed trait BSONNumberLike { self: BSONNumberLikeClass[_] =>
31   private[bson] def underlying: BSONValue
32 
33   /** Converts this number into an `Int`. */
34   def toInt: Int
35 
36   /** Converts this number into a `Long`. */
37   def toLong: Long
38 
39   /** Converts this number into a `Float`. */
40   def toFloat: Float
41 
42   /** Converts this number into a `Double`. */
43   def toDouble: Double
44 }
45 
46 private[bson] sealed trait HasNumeric[A] {
47   private[bson] def number: ExtendedNumeric[A]
48 }
49 
50 private[bson] sealed trait IsNumeric[A] extends HasNumeric[A] {
51   def toInt = number.numeric.toInt(number.value)
52   def toLong = number.numeric.toLong(number.value)
53   def toFloat = number.numeric.toFloat(number.value)
54   def toDouble = number.numeric.toDouble(number.value)
55 }
56 
57 private[bson] sealed trait IsBooleanLike[A] extends HasNumeric[A] {
58   def toBoolean: Boolean = number.numeric.compare(number.value, number.numeric.zero) != 0
59 }
60 
61 private[bson] case class ExtendedNumeric[A](value: A)(implicit val numeric: Numeric[A]) {}

81 /**
82  * A BSON value that can be seen as a boolean.
83  *
84  * Conversions:
85  *   - `number = 0 ~> false`
86  *   - `number != 0 ~> true`
87  *   - `boolean`
88  *   - `undefined ~> false`
89  *   - `null ~> false`
90  */
91 sealed trait BSONBooleanLike { self: BSONBooleanLikeClass[_] =>
92   private[bson] def underlying: BSONValue
93   /** Returns the boolean equivalent value */
94   def toBoolean: Boolean
95 }
