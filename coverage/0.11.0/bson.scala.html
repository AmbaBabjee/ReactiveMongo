59 object Producer {
60   case class NameOptionValueProducer(private val element: (String, Option[BSONValue])) extends Producer[(String, BSONValue)] {
61     private[bson] def produce = element._2.map(value => element._1 -> value)
62   }
63 
64   case class OptionValueProducer(private val element: Option[BSONValue]) extends Producer[BSONValue] {
65     private[bson] def produce = element
66   }
67 
68   implicit def nameValue2Producer[T](element: (String, T))(implicit writer: BSONWriter[T, _ <: BSONValue]) =
69     NameOptionValueProducer(element._1, Some(writer.write(element._2)))
70 
71   implicit def nameOptionValue2Producer[T](element: (String, Option[T]))(implicit writer: BSONWriter[T, _ <: BSONValue]) =
72     NameOptionValueProducer(element._1, element._2.map(value => writer.write(value)))
73 
74   implicit def noneOptionValue2Producer(element: (String, None.type)) =
75     NameOptionValueProducer(element._1, None)
76 
77   implicit def valueProducer[T](element: T)(implicit writer: BSONWriter[T, _ <: BSONValue]) =
78     OptionValueProducer(Some(writer.write(element)))

80   implicit def optionValueProducer[T](element: Option[T])(implicit writer: BSONWriter[T, _ <: BSONValue]) =

115 object BSON {
116   /**
117    * Produces a `T` instance of the given BSON value, if there is an implicit `BSONReader[B, T]` in the scope.
118    *
119    * Prefer `readDocument` over this one if you want to deserialize `BSONDocuments`.
120    */
121   def read[B <: BSONValue, T](bson: B)(implicit reader: BSONReader[B, T]): T = reader.read(bson)
122   /**
123    * Produces a `BSONValue` instance of the given `T` value, if there is an implicit `BSONWriter[T, B]` in the scope.
124    *
125    * Prefer `writeDocument` over this one if you want to serialize `T` instances.
126    */
127   def write[T, B <: BSONValue](t: T)(implicit writer: BSONWriter[T, B]): B = writer.write(t)
128 
129   /** Produces a `T` instance of the given `BSONDocument`, if there is an implicit `BSONReader[BSONDocument, T]` in the scope. */
130   def readDocument[T](doc: BSONDocument)(implicit reader: BSONReader[BSONDocument, T]): T = reader.read(doc)
131 
132   /** Produces a `BSONDocument` of the given `T` instance, if there is an implicit `BSONWriter[T, BSONDocument]` in the scope. */
133   def writeDocument[T](t: T)(implicit writer: BSONWriter[T, BSONDocument]): BSONDocument = writer.write(t)
