5  * at compile time. Invoking these macros is equivalent to writing anonymous

7  *
8  * {{{

70     /**
71      * Use type parameter `A` as static type but use pattern matching to handle
72      * different possible subtypes. This makes it easy to persist algebraic
73      * data types(pattern where you have a sealed trait and several implementing
74      * case classes). When writing a case class into BSON its dynamic type
75      * will be pattern matched, when reading from BSON the pattern matching
76      * will be done on the `className` string. This option extends
77      * [[reactivemongo.bson.Macros.Options.SaveClassName]] in to ensure class
78      * names are always serialized.
79      *
80      * @tparam Types to use in pattern matching. Listed in a "type list" \/
81      */
82     trait UnionType[Types <: \/[_, _]] extends SaveClassName with Default
83 
84     /**
85      * Same as [[UnionType]] but saving the classâ€™ simple name io. the
86      * fully-qualified name.
87      * @tparam Types to use in pattern matching. Listed in a "type list" \/
88      */
89     trait SimpleUnionType[Types <: \/[_, _]] extends UnionType[Types] with SaveSimpleName with Default
90 
91     /**
92      * Type for making type-level lists for UnionType.
93      * If second parameter is another \/ it will be flattend out into a list
94      * and so on. Using infix notation makes much more sense since it then

96      *
97      * `Foo \/ Bar \/ Baz` is interpreted as type Foo or type Bar or type Baz

116   /** Annotations to use on case classes that are being processed by macros. */
117   object Annotations {
118     import scala.annotation.{ StaticAnnotation, meta }
119 
120     /**
121      * Specify a key different from field name in your case class.
122      * Convenient to use when you'd like to leverage mongo's `_id` index
123      * but don't want to actually use `_id` in your code.
124      *
125      * {{{
126      * case class Website(@Key("_id") url: String)
127      * }}}
128      *
129      * Generated handler will map the `url` field in your code
130      * to `_id` field in BSON
131      *
132      * @param key the desired key to use in BSON
133      */
134     @meta.param
135     case class Key(key: String) extends StaticAnnotation
136 
137     /** Ignores a field */
138     @meta.param
139     case class Ignore() extends StaticAnnotation
140   }
