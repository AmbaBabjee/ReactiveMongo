54 object BSON {
55   /**
56    * Produces a `T` instance of the given BSON value, if there is an implicit `BSONReader[B, T]` in the scope.
57    *
58    * Prefer `readDocument` over this one if you want to deserialize `BSONDocuments`.
59    */
60   def read[B <: BSONValue, T](bson: B)(implicit reader: BSONReader[B, T]): T = reader.read(bson)
61   /**
62    * Produces a `BSONValue` instance of the given `T` value, if there is an implicit `BSONWriter[T, B]` in the scope.
63    *
64    * Prefer `writeDocument` over this one if you want to serialize `T` instances.
65    */
66   def write[T, B <: BSONValue](t: T)(implicit writer: BSONWriter[T, B]): B = writer.write(t)
67 
68   /** Produces a `T` instance of the given `BSONDocument`, if there is an implicit `BSONReader[BSONDocument, T]` in the scope. */
69   def readDocument[T](doc: BSONDocument)(implicit reader: BSONReader[BSONDocument, T]): T = reader.read(doc)
70 
71   /** Produces a `BSONDocument` of the given `T` instance, if there is an implicit `BSONWriter[T, BSONDocument]` in the scope. */
72   def writeDocument[T](t: T)(implicit writer: BSONWriter[T, BSONDocument]): BSONDocument = writer.write(t)
