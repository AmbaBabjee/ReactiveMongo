23 object CustomEnumeratee {
24   trait RecoverFromErrorFunction {
25     def apply[E, A](throwable: Throwable, input: Input[E], continue: () => Iteratee[E, A]): Iteratee[E, A]
26   }
27   object RecoverFromErrorFunction {
28     object StopOnError extends RecoverFromErrorFunction {
29       def apply[E, A](throwable: Throwable, input: Input[E], continue: () => Iteratee[E, A]): Iteratee[E, A] =
30         Error(throwable.getMessage(), input)
31     }
32     object ContinueOnError extends RecoverFromErrorFunction {
33       def apply[E, A](throwable: Throwable, input: Input[E], continue: () => Iteratee[E, A]): Iteratee[E, A] =
34         continue()
35     }
36   }
37   def stopOnError[E]: Enumeratee[E, E] = recover(RecoverFromErrorFunction.StopOnError)
38   def continueOnError[E]: Enumeratee[E, E] = recover(RecoverFromErrorFunction.ContinueOnError)
39   def recover[E](ƒ: RecoverFromErrorFunction)(implicit ec: ExecutionContext): Enumeratee[E, E] = {
40     new Enumeratee[E, E] {
41       def applyOn[A](it: Iteratee[E, A]): Iteratee[E, Iteratee[E, A]] = {
42         def step(it: Iteratee[E, A])(input: Input[E]): Iteratee[E, Iteratee[E, A]] = input match {
43           case in @ (Input.El(_) | Input.Empty) =>
44             val next: Future[Iteratee[E, Iteratee[E, A]]] = it.pureFlatFold[E, Iteratee[E, A]] {
45               case Step.Cont(k) =>
46                 val n = k(in)
47                 n.pureFlatFold[E, Iteratee[E, A]] {
48                   case Step.Cont(k) => Cont(step(n))
49                   case _            => Done(n, Input.Empty)
50                 }
51               case other => Done(other.it, in)
52             }.unflatten.map({ s =>
53               s.it
54             }).recover({
55               case e: Throwable =>
56                 ƒ(e, in, () => Cont(step(it)))
57             })
58             Iteratee.flatten(next)
59           case Input.EOF =>
60             Done(it, Input.Empty)
61         }
62         Cont(step(it))
63       }
64     }
65   }
66 }

68 object CustomEnumerator {
69   private[iteratees] def intermediatePromise[A](future: Future[A])(implicit ec: ExecutionContext) = {
70     val promise = Promise[A]()
71     future.onComplete {
72       case Success(s) =>
73         promise.success(s)
74       case Failure(f) =>
75         promise.failure(f)
76     }
77     promise.future
78   }

80   class SEnumerator[C](zero: C)(next: C => Future[Option[C]], cleanUp: C => Unit)(implicit ec: ExecutionContext) extends Enumerator[C] {
81 
82     def apply[A](iteratee: Iteratee[C, A]): Future[Iteratee[C, A]] = {
83 
84       def loop(current: C, iteratee: Iteratee[C, A]): Future[Iteratee[C, A]] =
85         iteratee.fold {
86           case step @ Step.Cont(ƒ) => next(current) flatMap {
87             case Some(nnx) => loop(nnx, ƒ(Input.El(nnx)))
88             case _ =>
89               val it = ƒ(Input.Empty)
90               cleanUp(current)
91               Future.successful(it)
92           }
93           case Step.Done(a, e) =>
94             val done = Done(a, e)
95             cleanUp(current)
96             Future.successful(done)
97 
98           case Step.Error(msg, e) =>
99             val error = Error(msg, e)
100             cleanUp(current)
101             Future.successful(error)
102         }
103 
104       iteratee fold {
105         case Step.Cont(ƒ) =>
106           loop(zero, ƒ(Input.El(zero)))
107         case Step.Done(a, e) =>
108           val done = Done(a, e)
109           Future.successful(done)
110         case Step.Error(msg, e) =>
111           val error = Error(msg, e)
112           Future.successful(error)
113       }
114 
115     }
116   }

118   object SEnumerator {
119     def apply[C](zero: C)(next: C => Future[Option[C]])(implicit ec: ExecutionContext) = new SEnumerator(zero)(next, _ => ())

129   class StateEnumerator[Chunk](zero: Future[Option[Chunk]])(nextChunk: Chunk => Future[Option[Chunk]])(implicit ec: ExecutionContext) extends Enumerator[Chunk] {
130     def apply[A](i: Iteratee[Chunk, A]): Future[Iteratee[Chunk, A]] = {
131       val promise = Promise[Iteratee[Chunk, A]]
132       def inloop(step: Future[Option[Chunk]], iteratee: Iteratee[Chunk, A]): Unit = {
133         iteratee.fold {
134           case Step.Cont(f) => {
135             val future = step.map {
136               case Some(state) =>
137                 (Some(state), f(Input.El(state)))
138               case None =>
139                 (None, f(Input.EOF))
140             }
141             future.onSuccess {
142               case (state, iteratee) =>
143                 if (state.isDefined)
144                   inloop(nextChunk(state.get), iteratee)
145                 else
146                   promise.success(iteratee)
147             }
148             future.onFailure {
149               case e => promise.failure(e)
150             }
151             future.map(_._2)
152           }
153           case Step.Done(a, e) => {
154             val finalIteratee = Done(a, e)
155             promise.success(finalIteratee)
156             Future.successful(finalIteratee)
157           }
158           case Step.Error(msg, e) => {
159             val finalIteratee = Error(msg, e)
160             promise.success(finalIteratee)
161             Future.successful(finalIteratee)
162           }
163         }
164       }
165       inloop(zero, i)
166       promise.future
167     }
168   }
169 
170   object StateEnumerator {
171     def apply[Chunk](zero: Future[Option[Chunk]])(nextChunk: Chunk => Future[Option[Chunk]])(implicit ec: ExecutionContext): Enumerator[Chunk] =
172       new StateEnumerator(zero)(nextChunk)
173   }
