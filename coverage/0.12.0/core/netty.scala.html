29 class ChannelBufferReadableBuffer(
30     protected[netty] val buffer: ChannelBuffer
31 ) extends ReadableBuffer {
32 
33   def size = buffer.capacity()
34 
35   def index = buffer.readerIndex()
36 
37   def index_=(i: Int) = buffer.readerIndex(i)
38 
39   def discard(n: Int) = buffer.readerIndex(buffer.readerIndex + n)
40 
41   def slice(n: Int) =
42     new ChannelBufferReadableBuffer(buffer.slice(buffer.readerIndex(), n))
43 
44   def readBytes(array: Array[Byte]): Unit = buffer.readBytes(array)
45 
46   def readByte() = buffer.readByte()
47 
48   def readInt() = buffer.readInt()
49 
50   def readLong() = buffer.readLong()
51 
52   def readDouble() = buffer.readDouble()
53 
54   def readable() = buffer.readableBytes()
55 
56   def toWritableBuffer: ChannelBufferWritableBuffer = {
57     val buf = new ChannelBufferWritableBuffer
58     buf.writeBytes(buffer)
59   }
60 
61   def duplicate() = new ChannelBufferReadableBuffer(buffer.duplicate())
62 }
63 
64 object ChannelBufferReadableBuffer {
65   def apply(buffer: ChannelBuffer) = new ChannelBufferReadableBuffer(buffer)
66 
67   def document(buffer: ChannelBuffer): BSONDocument =
68     BSONDocument.read(ChannelBufferReadableBuffer(buffer))
69 
70 }

72 class ChannelBufferWritableBuffer(val buffer: ChannelBuffer = ChannelBuffers.dynamicBuffer(LITTLE_ENDIAN, 32)) extends WritableBuffer {
73   def index = buffer.writerIndex
74 
75   def setInt(index: Int, value: Int) = {
76     buffer.setInt(index, value)
77     this
78   }
79 
80   def toReadableBuffer =
81     ChannelBufferReadableBuffer(buffer.duplicate())
82 
83   def writeBytes(array: Array[Byte]): this.type = {
84     buffer writeBytes array
85     this
86   }
87 
88   override def writeBytes(buf: ReadableBuffer): this.type = buf match {
89     case nettyBuffer: ChannelBufferReadableBuffer =>
90       val readable = nettyBuffer.buffer.slice()
91       buffer.writeBytes(readable)
92       this
93 
94     case _ => super.writeBytes(buf)
95   }
96 
97   def writeBytes(buf: ChannelBuffer) = {
98     val readable = buf.slice()
99     buffer.writeBytes(readable)
100     this
101   }
102 
103   def writeByte(byte: Byte): WritableBuffer = {
104     buffer writeByte byte
105     this
106   }
107 
108   def writeInt(int: Int): WritableBuffer = {
109     buffer writeInt int
110     this
111   }
112 
113   def writeLong(long: Long): WritableBuffer = {
114     buffer writeLong long
115     this
116   }
117 
118   def writeDouble(double: Double): WritableBuffer = {
119     buffer writeDouble double
120     this
121   }
122 }

124 object ChannelBufferWritableBuffer {
125   /** Returns a new writable channel buffer. */
126   def apply() = new ChannelBufferWritableBuffer()
127 
128   /** Returns a new channel buffer with the give `document` written on. */
129   private[reactivemongo] def single(document: BSONDocument): ChannelBuffer = {
130     val buffer = ChannelBufferWritableBuffer()
131     BSONDocument.write(document, buffer)
132     buffer.buffer
133   }
134 }

136 case class BufferSequence(private val head: ChannelBuffer, private val tail: ChannelBuffer*) {
137   def merged: ChannelBuffer = mergedBuffer.duplicate()
138 
139   private lazy val mergedBuffer =
140     ChannelBuffers.wrappedBuffer((head +: tail): _*)
