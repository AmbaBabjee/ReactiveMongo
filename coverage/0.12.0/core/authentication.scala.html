22   protected final def sendAuthenticate(connection: Connection, nextAuth: Authenticate): Connection = {
23     connection.send(GetCrNonce(nextAuth.db).maker(RequestId.getNonce.next))
24     connection.copy(authenticating = Some(
25       CrAuthenticating(nextAuth.db, nextAuth.user, nextAuth.password, None)

29   protected val authReceive: Receive = {
30     case response: Response if RequestId.getNonce accepts response =>
31       GetCrNonce.ResultMaker(response).fold(
32         e =>
33           logger.warn(s"error while processing getNonce response #${response.header.responseTo}", e),
34         nonce => {
35           logger.debug(s"AUTH: got nonce for channel ${response.info.channelId}: $nonce")
36           whenAuthenticating(response.info.channelId) {
37             case (connection, a @ CrAuthenticating(db, user, pass, _)) =>
38               connection.send(CrAuthenticate(user, pass, nonce)(db).
39                 maker(RequestId.authenticate.next))
40 
41               connection.copy(authenticating = Some(a.copy(
42                 nonce = Some(nonce)
43               )))
44 
45             case (connection, auth) => {
46               val msg = s"unexpected authentication: $auth"
47 
48               logger.warn(s"AUTH: $msg")
49               authenticationResponse(response)(
50                 _ => Left(FailedAuthentication(msg))
51               )
52 
53               connection
54             }
55           }
56         }
57       )
58 
59     case response: Response if RequestId.authenticate accepts response => {
60       logger.debug(s"AUTH: got authenticated response! ${response.info.channelId}")
61       authenticationResponse(response)(CrAuthenticate.parseResponse(_))
62     }
63   }

78   }

80   protected final def sendAuthenticate(connection: Connection, nextAuth: Authenticate): Connection = {
81     val start = ScramSha1Initiate(nextAuth.user)
82 
83     connection.send(start(nextAuth.db).maker(RequestId.getNonce.next))
84 
85     connection.copy(authenticating = Some(
86       ScramSha1Authenticating(nextAuth.db, nextAuth.user, nextAuth.password,

91   protected val authReceive: Receive = {
92     case response: Response if RequestId.getNonce accepts response => {
93       ScramSha1Initiate.parseResponse(response).fold(
94         { e =>
95           val msg = s"error while processing getNonce response #${response.header.responseTo}"
96 
97           logger.warn(s"AUTH: $msg")
98           logger.debug("SCRAM-SHA1 getNonce failure", e)
99 
100           authenticationResponse(response)(_ => Left(FailedAuthentication(msg)))
101         }, { challenge =>
102           logger.debug(s"AUTH: got challenge for channel ${response.info.channelId}: $challenge")
103 
104           whenAuthenticating(response.info.channelId) {
105             case (con, a @ ScramSha1Authenticating(
106               db, user, pwd, rand, msg, _, _, step)) => {
107               val negociation = ScramSha1StartNegociation(user, pwd,
108                 challenge.conversationId, challenge.payload, rand, msg)
109 
110               negociation.serverSignature.fold[Connection](
111                 { e => authenticationResponse(response)(_ => Left(e)); con },
112                 { sig =>
113                   con.send(negociation(db).maker(RequestId.authenticate.next))
114 
115                   con.copy(authenticating = Some(a.copy(
116                     conversationId = Some(challenge.conversationId),
117                     serverSignature = Some(sig),
118                     step = step + 1
119                   )))
120                 }
121               )
122             }
123 
124             case (con, auth) => {
125               val msg = s"unexpected authentication: $auth"
126 
127               logger.warn(s"AUTH: $msg")
128               authenticationResponse(response)(
129                 _ => Left(FailedAuthentication(msg))
130               )
131 
132               con
133             }
134           }
135         }
136       )
137     }
138 
139     case response: Response if RequestId.authenticate accepts response => {
140       logger.debug(s"AUTH: got authenticated response! ${response.info.channelId}")
141 
142       @inline def resp: Either[Either[CommandError, SuccessfulAuthentication], Array[Byte]] = ScramSha1StartNegociation.parseResponse(response) match {
143         case Left(err)             => Left(Left(err))
144         case Right(Left(authed))   => Left(Right(authed))
145         case Right(Right(payload)) => Right(payload)
146       }
147 
148       resp.fold(
149         { r => authenticationResponse(response)(_ => r) },
150         { payload: Array[Byte] =>
151           logger.debug("2-phase SCRAM-SHA1 negociation")
152 
153           whenAuthenticating(response.info.channelId) {
154             case (con, a @ ScramSha1Authenticating(
155               db, user, pwd, rand, msg, Some(cid), Some(sig),
156               1 /* step; TODO: more retry? */ )) => {
157 
158               val serverSig: Option[String] =
159                 ScramSha1Negociation.parsePayload(payload).get("v")
160 
161               if (!serverSig.exists(_ == Base64.encodeBase64String(sig))) {
162                 val msg = "the SCRAM-SHA1 server signature is invalid"
163 
164                 logger.warn(s"AUTH: $msg")
165                 authenticationResponse(response)(
166                   _ => Left(FailedAuthentication(msg))
167                 )
168 
169                 con
170               } else {
171                 val negociation = ScramSha1FinalNegociation(cid, payload)
172 
173                 con.send(negociation(db).maker(RequestId.authenticate.next))
174                 con.copy(authenticating = Some(a.copy(step = 2)))
175               }
176             }
177 
178             case (con, auth) => {
179               val msg = s"unexpected authentication: $auth"
180 
181               logger.warn(s"AUTH: msg")
182               authenticationResponse(response)(
183                 _ => Left(FailedAuthentication(msg))
184               )
185 
186               con
187             }
188           }
189         }
190       )
191     }
192   }
