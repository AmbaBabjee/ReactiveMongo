35   def updateFirst[A, M[T] <: Iterable[T]](coll: M[A])(f: A => Option[A])(implicit cbf: CanBuildFrom[M[_], A, M[A]]): M[A] = {
36     val builder = cbf.apply
37 
38     def run(iterator: Iterator[A]): Unit = {
39       while (iterator.hasNext) {
40         val e = iterator.next
41         val updated = f(e)
42         if (updated.isDefined) {
43           builder += updated.get
44           builder ++= iterator
45         } else builder += e
46       }
47     }
48 
49     builder.result
50   }

52   def update[A, M[T] <: Iterable[T]](coll: M[A])(f: PartialFunction[A, A])(implicit cbf: CanBuildFrom[M[_], A, M[A]]): (M[A], Boolean) = {
53     val builder = cbf.apply
54     val (head, tail) = coll.span(!f.isDefinedAt(_))
55     builder ++= head
56 
57     if (!tail.isEmpty) {
58       builder += f(tail.head)
59       builder ++= tail.drop(1)
60     }
61 
62     builder.result -> !tail.isEmpty
63   }
