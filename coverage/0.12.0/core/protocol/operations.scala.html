20 /**
21  * Helper methods to write tuples of supported types into a [[http://static.netty.io/3.5/api/org/jboss/netty/buffer/ChannelBuffer.html ChannelBuffer]].
22  */
23 private[protocol] object BufferAccessors {
24   /**
25    * Typeclass for types that can be written into a [[http://static.netty.io/3.5/api/org/jboss/netty/buffer/ChannelBuffer.html ChannelBuffer]]
26    * via writeTupleToBufferN methods.
27    *
28    * @tparam T type to be written via BufferAccessors.writeTupleToBufferN(...) methods.
29    */
30   sealed trait BufferInteroperable[T] {
31     def apply(buffer: ChannelBuffer, t: T): Unit
32   }
33 
34   implicit object IntChannelInteroperable extends BufferInteroperable[Int] {
35     def apply(buffer: ChannelBuffer, i: Int) = buffer writeInt i
36   }
37 
38   implicit object LongChannelInteroperable extends BufferInteroperable[Long] {
39     def apply(buffer: ChannelBuffer, l: Long) = buffer writeLong l
40   }
41 
42   implicit object StringChannelInteroperable extends BufferInteroperable[String] {
43     def apply(buffer: ChannelBuffer, s: String) = buffer writeCString s
44   }
45 
46   /**
47    * Write the given tuple into the given [[http://static.netty.io/3.5/api/org/jboss/netty/buffer/ChannelBuffer.html ChannelBuffer]].
48    *
49    * @tparam A type that have an implicit typeclass [[reactivemongo.core.protocol.BufferAccessors.BufferInteroperable]].
50    * @tparam B type that have an implicit typeclass [[reactivemongo.core.protocol.BufferAccessors.BufferInteroperable]].
51    */
52   def writeTupleToBuffer2[A, B](t: (A, B))(buffer: ChannelBuffer)(implicit i1: BufferInteroperable[A], i2: BufferInteroperable[B]): Unit = {
53     i1(buffer, t._1)
54     i2(buffer, t._2)
55   }
56 
57   /**
58    * Write the given tuple into the given [[http://static.netty.io/3.5/api/org/jboss/netty/buffer/ChannelBuffer.html ChannelBuffer]].
59    *
60    * @tparam A type that have an implicit typeclass [[reactivemongo.core.protocol.BufferAccessors.BufferInteroperable]].
61    * @tparam B type that have an implicit typeclass [[reactivemongo.core.protocol.BufferAccessors.BufferInteroperable]].
62    * @tparam C type that have an implicit typeclass [[reactivemongo.core.protocol.BufferAccessors.BufferInteroperable]].
63    */
64   def writeTupleToBuffer3[A, B, C](t: (A, B, C))(buffer: ChannelBuffer)(implicit i1: BufferInteroperable[A], i2: BufferInteroperable[B], i3: BufferInteroperable[C]): Unit = {
65     i1(buffer, t._1)
66     i2(buffer, t._2)
67     i3(buffer, t._3)
68   }
69 
70   /**
71    * Write the given tuple into the given [[http://static.netty.io/3.5/api/org/jboss/netty/buffer/ChannelBuffer.html ChannelBuffer]].
72    *
73    * @tparam A type that have an implicit typeclass [[reactivemongo.core.protocol.BufferAccessors.BufferInteroperable]].
74    * @tparam B type that have an implicit typeclass [[reactivemongo.core.protocol.BufferAccessors.BufferInteroperable]].
75    * @tparam C type that have an implicit typeclass [[reactivemongo.core.protocol.BufferAccessors.BufferInteroperable]].
76    * @tparam D type that have an implicit typeclass [[reactivemongo.core.protocol.BufferAccessors.BufferInteroperable]].
77    */
78   def writeTupleToBuffer4[A, B, C, D](t: (A, B, C, D))(buffer: ChannelBuffer)(implicit i1: BufferInteroperable[A], i2: BufferInteroperable[B], i3: BufferInteroperable[C], i4: BufferInteroperable[D]): Unit = {
79     i1(buffer, t._1)
80     i2(buffer, t._2)
81     i3(buffer, t._3)
82     i4(buffer, t._4)
83   }
84 }

88 /** A Mongo Wire Protocol operation */
89 sealed trait Op {
90   /** operation code */
91   val code: Int
92 }
93 
94 /**
95  * A Mongo Wire Protocol request operation.
96  *
97  * Actually, all operations excepted Reply are requests.
98  */
99 sealed trait RequestOp extends Op with ChannelBufferWritable {
100   /** States if this request expects a response. */
101   val expectsResponse: Boolean = false
102 
103   /** States if this request has to be run on a primary. */
104   val requiresPrimary: Boolean = false
105 }
106 
107 /** A request that needs to know the full collection name. */
108 sealed trait CollectionAwareRequestOp extends RequestOp {
109   /** The full collection name ('''') */
110   val fullCollectionName: String
111 
112   /** Database and collection name */
113   lazy val (db: String, collectionName: String) = fullCollectionName.span(_ != '.')
114 }
115 
116 /** A request that will perform a write on the database */
117 sealed trait WriteRequestOp extends CollectionAwareRequestOp
118 
119 /**
120  * Reply operation.
121  *
122  * @param flags The flags of this response.
123  * @param cursorID The cursor id. Strictly positive if a cursor has been created server side, 0 if none or exhausted.
124  * @param startingFrom The index the returned documents start from.
125  * @param numberReturned The number of documents that are present in this reply.
126  */
127 case class Reply(
128     flags: Int,
129     cursorID: Long,
130     startingFrom: Int,
131     numberReturned: Int
132 ) extends Op {
133   override val code = 1
134 
135   /** States whether the cursor given in the request was found */
136   lazy val cursorNotFound = (flags & 0x01) != 0
137 
138   /** States if the request encountered an error */
139   lazy val queryFailure = (flags & 0x02) != 0
140 
141   /** States if the answering server supports the AwaitData query option */
142   lazy val awaitCapable = (flags & 0x08) != 0
143 
144   private def str(b: Boolean, s: String) = if (b) s else ""
145 
146   /** States if this reply is in error */
147   lazy val inError = cursorNotFound || queryFailure
148 
149   lazy val stringify = toString + " [" + str(cursorNotFound, "CursorNotFound;") + str(queryFailure, "QueryFailure;") + str(awaitCapable, "AwaitCapable") + "]"
150 }
151 
152 object Reply extends ChannelBufferReadable[Reply] {
153   def readFrom(buffer: ChannelBuffer) = Reply(
154     buffer.readInt,
155     buffer.readLong,
156     buffer.readInt,
157     buffer.readInt
158   )
159 }

161 /**
162  * Update operation.
163  *
164  * @param flags Operation flags.
165  */
166 case class Update(
167     fullCollectionName: String,
168     flags: Int
169 ) extends WriteRequestOp {
170   override val code = 2001
171   override val writeTo = writeTupleToBuffer3((0, fullCollectionName, flags)) _
172   override def size = 4 /* int32 = ZERO */ + 4 + fullCollectionName.length + 1
173   override val requiresPrimary = true
174 }
175 
176 object UpdateFlags {
177   /** If set, the database will insert the supplied object into the collection if no matching document is found. */
178   val Upsert = 0x01
179 
180   /** If set, the database will update all matching objects in the collection. Otherwise only updates first matching doc. */
181   val MultiUpdate = 0x02
182 }

199 /**
200  * Query operation.
201  *
202  * @param flags Operation flags.
203  * @param numberToSkip number of documents to skip in the response.
204  * @param numberToReturn number of documents to return in the response. 0 means the server will choose.
205  */
206 case class Query(
207     flags: Int,
208     fullCollectionName: String,
209     numberToSkip: Int,
210     numberToReturn: Int
211 ) extends CollectionAwareRequestOp {
212   override val expectsResponse = true
213   override val code = 2004
214   override val writeTo = writeTupleToBuffer4((flags, fullCollectionName, numberToSkip, numberToReturn)) _
215   override def size = 4 + fullCollectionName.length + 1 + 4 + 4
216 }
217 
218 /**
219  * Query flags.
220  */
221 object QueryFlags {
222   /** Makes the cursor not to close after all the data is consumed. */
223   val TailableCursor = 0x02
224 
225   /** The query is might be run on a secondary. */
226   val SlaveOk = 0x04
227 
228   /** OplogReplay */
229   val OplogReplay = 0x08
230 
231   /** The cursor will not expire automatically */
232   val NoCursorTimeout = 0x10
233 
234   /**
235    * Block a little while waiting for more data instead of returning immediately if no data.
236    * Use along with TailableCursor.
237    */
238   val AwaitData = 0x20
239 
240   /** Exhaust */
241   val Exhaust = 0x40
242 
243   /** The response can be partial - if a shard is down, no error will be thrown. */
244   val Partial = 0x80
245 }

280 /**
281  * KillCursors operation.
282  *
283  * @param cursorIDs ids of the cursors to kill. Should not be empty.
284  */
285 case class KillCursors(
286     cursorIDs: Set[Long]
287 ) extends RequestOp {
288   override val code = 2007
289   override val writeTo = { buffer: ChannelBuffer =>
290     buffer writeInt 0
291     buffer writeInt cursorIDs.size
292     for (cursorID <- cursorIDs) {
293       buffer writeLong cursorID
294     }
295   }
296   override def size = 4 /* int32 ZERO */ + 4 + cursorIDs.size * 8
297 }
