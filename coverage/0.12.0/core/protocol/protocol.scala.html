45   implicit class RichBuffer(val buffer: ChannelBuffer) extends AnyVal {
46     import scala.collection.mutable.ArrayBuffer
47     /** Write a UTF-8 encoded C-Style String. */
48     def writeCString(s: String): ChannelBuffer = {
49       val bytes = s.getBytes("utf-8")
50       buffer writeBytes bytes
51       buffer writeByte 0
52       buffer
53     }
54 
55     /** Write a UTF-8 encoded String. */
56     def writeString(s: String): ChannelBuffer = {
57       val bytes = s.getBytes("utf-8")
58       buffer writeInt (bytes.size + 1)
59       buffer writeBytes bytes
60       buffer writeByte 0
61       buffer
62     }
63 
64     /** Write the contents of the given [[reactivemongo.core.protocol.ChannelBufferWritable]]. */
65     def write(writable: ChannelBufferWritable) = writable writeTo buffer
66 
67     /** Reads a UTF-8 String. */
68     def readString(): String = {
69       val bytes = new Array[Byte](buffer.readInt - 1)
70       buffer.readBytes(bytes)
71       buffer.readByte
72       new String(bytes, "UTF-8")
73     }
74 
75     /**
76      * Reads an array of Byte of the given length.
77      *
78      * @param length Length of the newly created array.
79      */
80     def readArray(length: Int): Array[Byte] = {
81       val bytes = new Array[Byte](length)
82       buffer.readBytes(bytes)
83       bytes
84     }
85 
86     /** Reads a UTF-8 C-Style String. */
87     def readCString(): String = {
88       @annotation.tailrec
89       def readCString(array: ArrayBuffer[Byte]): String = {
90         val byte = buffer.readByte
91         if (byte == 0x00)
92           new String(array.toArray, "UTF-8")
93         else readCString(array += byte)
94       }
95       readCString(new ArrayBuffer[Byte](16))
96     }
97 
98   }

116  * @tparam T type which instances can be constructed with this.
117  */

128  * Header of a Mongo Wire Protocol message.
129  *
130  * @param messageLength length of this message.
131  * @param requestID id of this request (> 0 for request operations, else 0).
132  * @param responseTo id of the request that the message including this a response to (> 0 for reply operation, else 0).

139     opCode: Int

162  * Request message.
163  *
164  * @param requestID id of this request, so that the response may be identifiable. Should be strictly positive.
165  * @param op request operation.
166  * @param documents body of this request, a [[http://static.netty.io/3.5/api/org/jboss/netty/buffer/ChannelBuffer.html ChannelBuffer]] containing 0, 1, or many documents.
167  * @param channelIdHint a hint for sending this request on a particular channel.

169 case class Request(
170     requestID: Int,
171     responseTo: Int, // TODO remove, nothing to do here.
172     op: RequestOp,
173     documents: BufferSequence,
174     readPreference: ReadPreference = ReadPreference.primary,
175     channelIdHint: Option[Int] = None
176 ) extends ChannelBufferWritable {
177   override val writeTo = { buffer: ChannelBuffer =>
178     buffer write header
179     buffer write op
180     buffer writeBytes documents.merged
181   }
182   override def size = 16 + op.size + documents.merged.writerIndex
183 
184   /** Header of this request */
185   lazy val header = MessageHeader(size, requestID, responseTo, op.code)
186 
187   override def toString =
188     s"Request($requestID, $responseTo, $op, $readPreference, $channelIdHint)"

313 sealed trait MongoWireVersion extends Ordered[MongoWireVersion] {
314   /** The numeric representation */
315   def value: Int
316 
317   final def compare(x: MongoWireVersion): Int =
318     if (value == x.value) 0
319     else if (value < x.value) -1
320     else 1
321 }
322 
323 object MongoWireVersion {
324   /*
325    * Original meaning of MongoWireVersion is more about protocol features.
326    *
327    * - RELEASE_2_4_AND_BEFORE (0)
328    * - AGG_RETURNS_CURSORS (1)
329    * - BATCH_COMMANDS (2)
330    *
331    * But wireProtocol=1 is virtually non-existent; Mongo 2.4 was 0 and Mongo 2.6 is 2.
332    */
333   @deprecated(message = "No longer supported", since = "0.12.0")
334   object V24AndBefore extends MongoWireVersion {
335     val value = 0
336     override val toString = "<=2.4"
337   }
338 
339   object V26 extends MongoWireVersion {
340     val value = 2
341     override val toString = "2.6"
342   }
343 
344   object V30 extends MongoWireVersion {
345     val value = 3
346     override val toString = "3.0"
347   }
348 
349   object V32 extends MongoWireVersion {
350     val value = 4
351     override val toString = "3.2"
352   }
353 
354   def apply(v: Int): MongoWireVersion = {
355     if (v <= V26.value) V26
356     else if (v >= V32.value) V32
357     else V30
358   }
359 
360   def unapply(v: MongoWireVersion): Option[Int] = Some(v.value)
361 }

365   import RequestEncoder._
366   def encode(ctx: ChannelHandlerContext, channel: Channel, obj: Object) =
367     obj match {
368       case message: Request => {
369         val buffer: ChannelBuffer = ChannelBuffers.buffer(ByteOrder.LITTLE_ENDIAN, message.size) //ChannelBuffers.dynamicBuffer(ByteOrder.LITTLE_ENDIAN, 1000)
370         message writeTo buffer
371         buffer
372       }
373       case _ => {
374         logger.error("weird... do not know how to encode this object: " + obj)
375         obj
376       }
377     }

381 object ReplyDocumentIterator {
382   def apply[P <: SerializationPack, A](pack: P)(reply: Reply, buffer: ChannelBuffer)(implicit reader: pack.Reader[A]): Iterator[A] = new Iterator[A] {
383     override val isTraversableAgain = false // TODO: Add test
384     override def hasNext = buffer.readable
385     override def next =
386       try {
387         val cbrb = ChannelBufferReadableBuffer(buffer.readBytes(buffer.getInt(buffer.readerIndex)))
388         pack.readAndDeserialize(cbrb, reader)
389       } catch {
390         case e: IndexOutOfBoundsException =>
391           /*
392            * If this happens, the buffer is exhausted, and there is probably a bug.
393            * It may happen if an enumerator relying on it is concurrently applied to many iteratees – which should not be done!
394            */
395           throw ReplyDocumentIteratorExhaustedException(e)
396       }
397   }
398 }

401   val cause: Exception
402 ) extends Exception(cause)

408 private[reactivemongo] class ResponseFrameDecoder extends FrameDecoder {
409   override def decode(context: ChannelHandlerContext, channel: Channel, buffer: ChannelBuffer) = {
410     val readableBytes = buffer.readableBytes
411     if (readableBytes < 4) null
412     else {
413       buffer.markReaderIndex
414       val length = buffer.readInt
415       buffer.resetReaderIndex
416       if (length <= readableBytes && length > 0)
417         buffer.readBytes(length)
418       else null
419     }
420   }
421 }

433 private[reactivemongo] class MongoHandler(
434   supervisor: String, connection: String, receiver: ActorRef
435 )
436     extends IdleStateAwareChannelHandler {
437 
438   override def messageReceived(ctx: ChannelHandlerContext, e: MessageEvent) {
439     val response = e.getMessage.asInstanceOf[Response]
440     log(e, s"messageReceived: $response will be send to $receiver")
441     receiver ! response
442     super.messageReceived(ctx, e)
443   }
444 
445   override def writeComplete(ctx: ChannelHandlerContext, e: WriteCompletionEvent) {
446     log(e, "A write is complete!")
447     super.writeComplete(ctx, e)
448   }
449 
450   override def writeRequested(ctx: ChannelHandlerContext, e: MessageEvent) {
451     log(e, "A write is requested!")
452     super.writeRequested(ctx, e)
453   }
454 
455   override def channelConnected(ctx: ChannelHandlerContext, e: ChannelStateEvent) {
456     log(e, "Channel is connected")
457     receiver ! ChannelConnected(e.getChannel.getId)
458     super.channelConnected(ctx, e)
459   }
460 
461   override def channelDisconnected(ctx: ChannelHandlerContext, e: ChannelStateEvent) {
462     log(e, "Channel is disconnected")
463     receiver ! ChannelDisconnected(e.getChannel.getId)
464     super.channelDisconnected(ctx, e)
465   }
466 
467   override def channelClosed(ctx: ChannelHandlerContext, e: ChannelStateEvent) {
468     if (e.getChannel.getRemoteAddress != null) log(e, "Channel is closed")
469 
470     receiver ! ChannelClosed(e.getChannel.getId)
471 
472     super.channelClosed(ctx, e)
473   }
474 
475   override def exceptionCaught(ctx: ChannelHandlerContext, e: shaded.netty.channel.ExceptionEvent) = log(e, "Channel error", e.getCause)
476 
477   override def channelIdle(ctx: ChannelHandlerContext, e: IdleStateEvent) = {
478     val now = System.currentTimeMillis()
479     val last = e.getLastActivityTimeMillis
480 
481     log(e, s"Channel has been inactive for ${now - last} (last = $last)")
482     e.getChannel.close()
483 
484     super.channelIdle(ctx, e)
485   }
486 
487   @inline def log(e: ChannelEvent, s: String) = MongoHandler.
488     logger.trace(s"[$supervisor/$connection @ ${e.getChannel}] $s")
489 
490   @inline def log(e: ChannelEvent, s: String, cause: Throwable) = MongoHandler.
491     logger.trace(s"[$supervisor/$connection @ ${e.getChannel}] $s", cause)
492 }
