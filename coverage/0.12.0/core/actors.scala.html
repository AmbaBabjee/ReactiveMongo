261   private def close(): NodeSet = {
262     logger.debug(s"[$lnm] Closing MongoDBSystem")
263 
264     // cancel all jobs
265     connectAllJob.cancel()
266     refreshAllJob.cancel()
267 
268     // close all connections
269     val listener = new ChannelGroupFutureListener {
270       val factory = channelFactory
271       def operationComplete(future: ChannelGroupFuture): Unit = {
272         logger.debug(s"[$lnm] Netty says all channels are closed.")
273 
274         try {
275           factory.channelFactory.releaseExternalResources()
276         } catch {
277           case err: Throwable =>
278             logger.debug(s"[$lnm] Fails to release channel resources", err)
279         }
280       }
281     }
282 
283     val ns = updateNodeSet("Close")(_.updateAll { node =>
284       node._copy(connections = node.connected)
285       // Only keep already connected connection:
286       // - prevent to activate other connection
287       // - know which connections to be closed
288     })
289 
290     allChannelGroup(ns).close().addListener(listener)
291 
292     // fail all requests waiting for a response
293     val istate = internalState
294 
295     awaitingResponses.foreach {
296       case (_, r) if (!r.promise.isCompleted) =>
297         // fail all requests waiting for a response
298         r.promise.failure(new ClosedException(supervisor, name, istate))
299 
300       case _ => ( /* already completed */ )
301     }
302 
303     awaitingResponses.clear()
304 
305     ns
306   }

366   protected final def authenticateConnection(connection: Connection, auths: Seq[Authenticate]): Connection = {
367     if (!connection.authenticating.isEmpty) connection
368     else auths.headOption match {
369       case Some(nextAuth) =>
370         if (connection.isAuthenticated(nextAuth.db, nextAuth.user)) {
371           authenticateConnection(connection, auths.tail)
372         } else sendAuthenticate(connection, nextAuth)
373 
374       case _ => connection
375     }
376   }

394     val connections = node.connections.collect(collector)

410   private def stopWhenDisconnected[T](state: String, msg: T): Unit = {
411     val remainingConnections = _nodeSet.nodes.foldLeft(0) {
412       _ + _.connected.size
413     }
414 
415     if (logger.isDebugEnabled) {
416       val disconnected = _nodeSet.nodes.foldLeft(0) { (open, node) =>
417         open + node.connections.count(_.status == ConnectionStatus.Disconnected)
418       }
419 
420       logger.debug(s"[$lnm$$$state] Received $msg remainingConnections = $remainingConnections, disconnected = $disconnected, connected = ${remainingConnections - disconnected}")
421     }
422 
423     if (remainingConnections == 0) {
424       monitors.foreach(_ ! Closed)
425       logger.debug(s"[$lnm] Stopping $self")
426       context.stop(self)
427     }
428   }

460   private def retry(req: AwaitingResponse): Option[AwaitingResponse] = {
461     val onError = failureOrLog(req.promise, _: Throwable) { cause =>
462       logger.error(s"[$lnm] Fails to retry '${req.request.op}' (channel #${req.channelID})", cause)
463     }
464 
465     req.retriable(requestRetries).flatMap { newReq =>
466       foldNodeConnection(req.request)({ error =>
467         onError(error)
468         None
469       }, { (node, con) =>
470         val reqId = req.requestID
471         val awaiting = newReq(con.channel.getId)
472 
473         awaitingResponses += reqId -> awaiting
474 
475         awaiting.getWriteConcern.fold(con.send(awaiting.request)) { wc =>
476           con.send(awaiting.request, wc)
477         }
478 
479         Some(awaiting)
480       })
481     }
482   }

488   private val processing: Receive = {
489     case RegisterMonitor => monitors += sender
490 
491     case Close => {
492       logger.debug(s"[$lnm] Received Close message, going to close connections and moving on state Closing")
493 
494       val ns = close()
495       val connectedCon = ns.nodes.foldLeft(0) { _ + _.connected.size }
496 
497       // moving to closing state
498       context become closing
499 
500       if (connectedCon == 0) stopWhenDisconnected("Processing", Close)
501 
502       listener.foreach(_.poolShutdown(supervisor, name))
503     }
504 
505     case req @ RequestMaker(_, _, _, _) => {
506       logger.trace(s"[$lnm] Transmiting a request: $req")
507 
508       val r = req(RequestId.common.next)
509       pickChannel(r).map(_._2.send(r))
510     }
511 
512     case req @ RequestMakerExpectingResponse(maker, _) => {
513       val reqId = RequestId.common.next
514 
515       logger.trace(
516         s"[$lnm] Received a request expecting a response ($reqId): $req"
517       )
518 
519       val request = maker(reqId)
520 
521       foldNodeConnection(request)(req.promise.failure(_), { (node, con) =>
522         if (request.op.expectsResponse) {
523           awaitingResponses += reqId -> AwaitingResponse(
524             request, con.channel.getId, req.promise,
525             isGetLastError = false,
526             isMongo26WriteOp = req.isMongo26WriteOp
527           )
528 
529           logger.trace(s"[$lnm] Registering awaiting response for requestID $reqId, awaitingResponses: $awaitingResponses")
530         } else logger.trace(s"[$lnm] NOT registering awaiting response for requestID $reqId")
531 
532         con.send(request)
533       })
534     }
535 
536     case req @ CheckedWriteRequestExpectingResponse(_) => {
537       logger.debug(s"[$lnm] Received a checked write request")
538 
539       val checkedWriteRequest = req.checkedWriteRequest
540       val reqId = RequestId.common.next
541       val (request, writeConcern) = {
542         val tuple = checkedWriteRequest()
543         tuple._1(reqId) -> tuple._2(reqId)
544       }
545       val onError = failureOrLog(req.promise, _: Throwable) { cause =>
546         logger.error(
547           s"[$lnm] Fails to register a request: ${request.op}", cause
548         )
549       }
550 
551       foldNodeConnection(request)(onError, { (node, con) =>
552         awaitingResponses += reqId -> AwaitingResponse(
553           request, con.channel.getId, req.promise,
554           isGetLastError = true, isMongo26WriteOp = false
555         ).withWriteConcern(writeConcern)
556 
557         logger.trace(s"[$lnm] Registering awaiting response for requestID $reqId, awaitingResponses: $awaitingResponses")
558 
559         con.send(request, writeConcern)
560       })
561     }
562 
563     case ConnectAll => { // monitor
564       val statusInfo = _nodeSet.toShortString
565 
566       updateNodeSet(s"ConnectAll($statusInfo)")(
567         _.createNeededChannels(channelFactory, self, options.nbChannelsPerNode)
568       )
569 
570       logger.debug(s"[$lnm] ConnectAll Job running... Status: $statusInfo")
571 
572       connectAll(_nodeSet)
573     }
574 
575     case RefreshAll => {
576       val statusInfo = _nodeSet.toShortString
577 
578       updateNodeSet(s"RefreshAll($statusInfo)")(_.updateAll { node =>
579         logger.trace(s"[$lnm] Try to refresh ${node.name}")
580 
581         sendIsMaster(node, RequestId.isMaster.next)
582       })
583 
584       logger.debug(s"[$lnm] RefreshAll Job running... Status: $statusInfo")
585     }
586 
587     case ChannelConnected(channelId) => {
588       val statusInfo = _nodeSet.toShortString
589 
590       updateNodeSet(s"ChannelConnected($channelId, $statusInfo)")(
591         _.updateByChannelId(channelId)(
592           _.copy(status = ConnectionStatus.Connected)
593         ) { node =>
594             sendIsMaster(node, RequestId.isMaster.next)
595           }
596       )
597 
598       logger.trace(
599         s"[$lnm] Channel #$channelId connected. NodeSet status: $statusInfo"
600       )
601     }
602 
603     case channelUnavailable @ ChannelUnavailable(channelId) => {
604       logger.trace(
605         s"[$lnm] Channel #$channelId is unavailable ($channelUnavailable)."
606       )
607 
608       val nodeSetWasReachable = _nodeSet.isReachable
609       val primaryWasAvailable = _nodeSet.primary.isDefined
610 
611       channelUnavailable match {
612         case ChannelClosed(_) => {
613           @inline def event =
614             s"ChannelClosed($channelId, ${_nodeSet.toShortString})"
615 
616           updateNodeSet(event)(_.updateNodeByChannelId(channelId) { node =>
617             collectConnections(node) {
618               case other if (other.channel.getId != channelId) => other
619             }
620           })
621         }
622 
623         case ChannelDisconnected(_) => updateNodeSetOnDisconnect(channelId)
624       }
625 
626       val retried = Map.newBuilder[Int, AwaitingResponse]
627 
628       awaitingResponses.retain { (_, awaitingResponse) =>
629         if (awaitingResponse.channelID == channelId) {
630           retry(awaitingResponse) match {
631             case Some(awaiting) => {
632               logger.trace(s"[$lnm] Retrying to await response for requestID ${awaiting.requestID}: $awaiting")
633               retried += channelId -> awaiting
634             }
635 
636             case _ => {
637               logger.debug(s"[$lnm] Completing response for '${awaitingResponse.request.op}' with error='socket disconnected' (channel #$channelId)")
638 
639               failureOrLog(awaitingResponse.promise, SocketDisconnected)(
640                 err => logger.warn(
641                   s"[$lnm] Socket disconnected (channel #$channelId)", err
642                 )
643               )
644             }
645           }
646 
647           false
648         } else true
649       }
650 
651       awaitingResponses ++= retried.result()
652 
653       if (!_nodeSet.isReachable) {
654         if (nodeSetWasReachable) {
655           logger.warn(s"[$lnm] The entire node set is unreachable, is there a network problem?")
656 
657           broadcastMonitors(SetUnavailable)
658         } else logger.debug(s"[$lnm] The entire node set is still unreachable, is there a network problem?")
659       } else if (!_nodeSet.primary.isDefined) {
660         if (primaryWasAvailable) {
661           logger.warn(s"[$lnm] The primary is unavailable, is there a network problem?")
662           broadcastMonitors(PrimaryUnavailable)
663         } else logger.debug(
664           s"[$lnm] The primary is still unavailable, is there a network problem?"
665         )
666       }
667 
668       logger.debug(s"[$lnm] Channel #$channelId is released")
669     }
670 
671     case IsMasterResponse(response) => {
672       logger.trace(s"[$lnm] IsMaster response: $response")
673 
674       import reactivemongo.api.BSONSerializationPack
675       import reactivemongo.api.commands.bson.BSONIsMasterCommandImplicits
676       import reactivemongo.api.commands.Command
677 
678       val isMaster = Command.deserialize(BSONSerializationPack, response)(
679         BSONIsMasterCommandImplicits.IsMasterResultReader
680       )
681 
682       val updated = {
683         val respTo = response.header.responseTo
684         @inline def event =
685           s"IsMaster(${respTo}, ${_nodeSet.toShortString}"
686 
687         updateNodeSet(event) { nodeSet =>
688           val nodeSetWasReachable = nodeSet.isReachable
689           val wasPrimary = nodeSet.primary.toSeq.flatMap(_.names)
690           @volatile var chanNode = Option.empty[Node]
691 
692           val prepared =
693             nodeSet.updateNodeByChannelId(response.info.channelId) { node =>
694               val pingInfo =
695                 if (node.pingInfo.lastIsMasterId == respTo) {
696                   node.pingInfo.copy(
697                     ping =
698                     System.currentTimeMillis() - node.pingInfo.lastIsMasterTime,
699                     lastIsMasterTime = 0, lastIsMasterId = -1
700                   )
701 
702                 } else node.pingInfo
703 
704               val authenticating =
705                 if (!isMaster.status.queryable || nodeSet.authenticates.isEmpty) {
706                   node
707                 } else authenticateNode(node, nodeSet.authenticates.toSeq)
708 
709               val meta = ProtocolMetadata(
710                 MongoWireVersion(isMaster.minWireVersion),
711                 MongoWireVersion(isMaster.maxWireVersion),
712                 isMaster.maxBsonObjectSize,
713                 isMaster.maxMessageSizeBytes,
714                 isMaster.maxWriteBatchSize
715               )
716 
717               val an = authenticating._copy(
718                 status = isMaster.status,
719                 pingInfo = pingInfo,
720                 tags = isMaster.replicaSet.flatMap(_.tags),
721                 protocolMetadata = meta,
722                 isMongos = isMaster.isMongos
723               )
724 
725               val n = isMaster.replicaSet.fold(an)(rs => an.withAlias(rs.me))
726               chanNode = Some(n)
727 
728               n
729             }
730 
731           val discoveredNodes = isMaster.replicaSet.toSeq.flatMap { rs =>
732             rs.hosts.collect {
733               case host if (!prepared.nodes.exists(_.names contains host)) =>
734                 // Prepare node for newly discovered host in the RS
735                 Node(host, NodeStatus.Uninitialized,
736                   Vector.empty, Set.empty, None, ProtocolMetadata.Default)
737             }
738           }
739

881               case BooleanField(_, v) => v
882               case IntField(_, v)     => v != 0
883               case DoubleField(_, v)  => v != 0

904               }
905 
906               promise.failure(GenericDriverException("not ok"))
907             }
908           } else {
909             logger.trace(s"[$lnm] {${response.header.responseTo}} [MongoDB26 Write Op response] sending a success!")
910             promise.success(response)
911           }
912         }
913 
914         case _ => logger.error(s"[$lnm] Oups. ${response.header.responseTo} not found! complete message is $response")
