5 /**
6  * Implements the "aggregation" command, otherwise known as the "Aggregation Framework."
7  * http://docs.mongodb.org/manual/applications/aggregation/
8  *
9  * @param collectionName Collection to aggregate against
10  * @param pipeline Sequence of MongoDB aggregation operations.
11  */
12 @deprecated(
13   message = "Use [[reactivemongo.api.collections.bson.BSONCollection.aggregate]]",
14   since = "0.11.4"
15 )
16 case class Aggregate(
17     collectionName: String,
18     pipeline: Seq[PipelineOperator]
19 ) extends Command[Stream[BSONDocument]] {
20   override def makeDocuments =
21     BSONDocument(
22       "aggregate" -> BSONString(collectionName),
23       "pipeline" -> BSONArray(
24         { for (pipe <- pipeline) yield pipe.makePipe }.toStream
25       )
26     )
27 
28   val ResultMaker = Aggregate
29 }
30 
31 object Aggregate extends BSONCommandResultMaker[Stream[BSONDocument]] {
32   def apply(document: BSONDocument) =
33     CommandError.checkOk(document, Some("aggregate")).toLeft(document.get("result").get.asInstanceOf[BSONArray].values.map(_.asInstanceOf[BSONDocument]))
34 }

44 /**
45  * Reshapes a document stream by renaming, adding, or removing fields.
46  * Also use "Project" to create computed values or sub-objects.
47  * http://docs.mongodb.org/manual/reference/aggregation/project/#_S_project
48  * @param fields Fields to include. The resulting objects will contain only these fields
49  */
50 case class Project(fields: (String, BSONValue)*) extends PipelineOperator {
51   override val makePipe = BSONDocument("$project" -> BSONDocument(
52     { for (field <- fields) yield field._1 -> field._2 }.toStream
53   ))
54 }

93 /**
94  * Groups documents together to calulate aggregates on document collections. This command
95  * aggregates on one field.
96  * http://docs.mongodb.org/manual/reference/aggregation/group/#_S_group
97  * @param idField Name of the field to aggregate on.
98  * @param ops Sequence of operators specifying aggregate calculation.
99  */
100 case class GroupField(idField: String)(ops: (String, GroupFunction)*) extends PipelineOperator {

102 }

104 /**
105  * Groups documents together to calulate aggregates on document collections. This command
106  * aggregates on multiple fields, and they must be named.
107  * http://docs.mongodb.org/manual/reference/aggregation/group/#_S_group
108  * @param idField Fields to aggregate on, and the names they should be aggregated under.
109  * @param ops Sequence of operators specifying aggregate calculation.
110  */
111 case class GroupMulti(idField: (String, String)*)(ops: (String, GroupFunction)*) extends PipelineOperator {
112   override val makePipe = Group(BSONDocument(
113     idField.map {
114     case (alias, attribute) => alias -> BSONString("$" + attribute)
115   }.toStream
116   ))(ops: _*).makePipe
117 }

119 /**
120  * Groups documents together to calulate aggregates on document collections. This command
121  * aggregates on arbitrary identifiers. Document fields identifier must be prefixed with `$`.
122  * http://docs.mongodb.org/manual/reference/aggregation/group/#_S_group
123  * @param identifiers Any BSON value acceptable by mongodb as identifier
124  * @param ops Sequence of operators specifying aggregate calculation.
125  */
126 case class Group(identifiers: BSONValue)(ops: (String, GroupFunction)*) extends PipelineOperator {
127   override val makePipe = BSONDocument(
128     "$group" -> BSONDocument(
129       {
130         "_id" -> identifiers
131       } +:
132         {
133           ops.map {
134             case (field, operator) => field -> operator.makeFunction
135           }
136         }.toStream
137     )
138   )
139 }

141 /**
142  * Sorts the stream based on the given fields.
143  * http://docs.mongodb.org/manual/reference/aggregation/sort/#_S_sort
144  * @param fields Fields to sort by.
145  */
146 case class Sort(fields: Seq[SortOrder]) extends PipelineOperator {
147   override val makePipe = BSONDocument("$sort" -> BSONDocument(fields.map {
148     case Ascending(field)  => field -> BSONInteger(1)
149     case Descending(field) => field -> BSONInteger(-1)
150   }.toStream))
151 }

169 /** Factory to declare custom call to a group function. */
170 object GroupFunction {
171   /**
172    * Creates a call to specified group function with given argument.
173    *
174    * @param name The name of the group function (e.g. `\$sum`)
175    * @param arg The group function argument
176    * @return A group function call defined as `{ name: arg }`
177    */
178   def apply(name: String, arg: BSONValue): GroupFunction = new GroupFunction {
179     val makeFunction = BSONDocument(name -> arg)
180   }
181 }
