32 /**
33  * Creates a collection on the database.
34  *
35  * Unless you want to give some extra parameters (like capped collections, or autoIndexId), it is not necessary as MongoDB creates regular collections on the fly.
36  *
37  * @param name The collection name.
38  * @param capped The collection to be created will be a capped collection if some CappedOptions instance is supplied.
39  * @param autoIndexId States if should automatically add an index on the _id field. By default, regular collections will have an indexed _id field, in contrast to capped collections.
40  */
41 @deprecated("consider using reactivemongo.api.commands.Create instead", "0.11.0")
42 class CreateCollection(
43     name: String,
44     capped: Option[CappedOptions] = None,
45     autoIndexId: Option[Boolean] = None
46 ) extends Command[Boolean] {
47   def makeDocuments = {
48     val doc = BSONDocument(
49       "create" -> BSONString(name),
50       "autoIndexId" -> autoIndexId.map(BSONBoolean(_))
51     )
52     if (capped.isDefined)
53       doc ++ capped.get.toDocument
54     else doc
55   }
56 
57   object ResultMaker extends BSONCommandResultMaker[Boolean] {
58     def apply(doc: BSONDocument) = {
59       CommandError.checkOk(doc, Some("createCollection")).toLeft(true)
60     }
61   }
62 }

107 /**
108  * Various information about a collection.
109  *
110  * @param ns The fully qualified collection name.
111  * @param count The number of documents in this collection.
112  * @param size The size in bytes (or in bytes / scale, if any).
113  * @param averageObjectSize The average object size in bytes (or in bytes / scale, if any).
114  * @param storageSize Preallocated space for the collection.
115  * @param numExtents Number of extents (contiguously allocated chunks of datafile space).
116  * @param nindexes Number of indexes.
117  * @param lastExtentSize Size of the most recently created extent.
118  * @param paddingFactor Padding can speed up updates if documents grow.
119  * @param systemFlags System flags.
120  * @param userFlags User flags.
121  * @param indexSizes Size of specific indexes in bytes.
122  * @param capped States if this collection is capped.
123  * @param max The maximum number of documents of this collection, if capped.
124  */
125 case class CollStatsResult(
126   ns: String,
127   count: Int,
128   size: Double,
129   averageObjectSize: Option[Double],
130   storageSize: Double,
131   numExtents: Int,
132   nindexes: Int,
133   lastExtentSize: Option[Int],
134   paddingFactor: Option[Double],
135   systemFlags: Option[Int],
136   userFlags: Option[Int],
137   totalIndexSize: Int,
138   indexSizes: Array[(String, Int)],
139   capped: Boolean,
140   max: Option[Long]
141 )
142 
143 object CollStatsResult extends BSONCommandResultMaker[CollStatsResult] {
144   def apply(doc: BSONDocument): Either[CommandError, CollStatsResult] = {
145     // TODO support sharding info
146     CommandError.checkOk(doc, Some("collStats")).toLeft {
147       CollStatsResult(
148         doc.getAs[BSONString]("ns").get.value,
149         doc.getAs[BSONInteger]("count").get.value,
150         doc.getAs[BSONNumberLike]("size").get.toDouble,
151         doc.getAs[BSONDouble]("avgObjSize").map(_.value),
152         doc.getAs[BSONNumberLike]("storageSize").get.toDouble,
153         doc.getAs[BSONInteger]("numExtents").get.value,
154         doc.getAs[BSONInteger]("nindexes").get.value,
155         doc.getAs[BSONInteger]("lastExtentSize").map(_.value),
156         doc.getAs[BSONDouble]("paddingFactor").map(_.value),
157         doc.getAs[BSONInteger]("systemFlags").map(_.value),
158         doc.getAs[BSONInteger]("userFlags").map(_.value),
159         doc.getAs[BSONInteger]("totalIndexSize").get.value,
160         {
161           val indexSizes = doc.getAs[BSONDocument]("indexSizes").get
162           (for (kv <- indexSizes.elements) yield kv._1 -> kv._2.asInstanceOf[BSONInteger].value).toArray
163         },
164         doc.getAs[BSONBooleanLike]("capped").map(_.toBoolean).getOrElse(false),
165         doc.getAs[BSONDouble]("max").map(_.value.toLong)
166       )
167     }
168   }
169 }
