50 object ScramSha1Initiate extends BSONCommandResultMaker[ScramSha1Challenge] {
51   def parseResponse(response: Response): Either[CommandError, ScramSha1Challenge] = apply(response)
52 
53   def apply(bson: BSONDocument) =
54     CommandError.checkOk(bson, Some("authenticate"), (doc, name) => {
55       FailedAuthentication(doc.getAs[BSONString]("errmsg").
56         map(_.value).getOrElse(""), Some(doc))
57     }).map[Either[CommandError, ScramSha1Challenge]](Left(_)).getOrElse {
58       (for {
59         cid <- bson.getAs[BSONNumberLike]("conversationId").map(_.toInt)
60         pay <- bson.getAs[Array[Byte]]("payload")
61       } yield (cid, pay)).fold[Either[CommandError, ScramSha1Challenge]](Left(FailedAuthentication(s"invalid SCRAM-SHA1 challenge response: ${BSONDocument pretty bson}", Some(bson)))) {
62         case (conversationId, payload) =>
63           Right(ScramSha1Challenge(conversationId, payload))
64       }
65     }
66 
67   // Request utility
68   private val authChars: Stream[Char] = new Iterator[Char] {
69     val rand = new scala.util.Random(this.hashCode)
70     val chars = rand.shuffle("""!"#$%&'()*+-./0123456789:;<=>?@ABCDEFGHIJKLMNOPQRSTUVWXYZ[\]^_`abcdefghijklmnopqrstuvwxyz{|}~""".toList)
71 
72     def hasNext = true
73 
74     var i = 0
75     def next(): Char = {
76       if (i == chars.length) {
77         i = 1
78         chars(0)
79       } else {
80         i += 1
81         chars(i - 1)
82       }
83     }
84   }.toStream
85 
86   def randomPrefix(seed: Int): String = {
87     val pos = (System.currentTimeMillis % 100).toInt // temporal position
88     val slice = authChars.slice(pos, pos + 24 /* random str length */ )
89     val rand = new scala.util.Random(seed)
90 
91     rand.shuffle(slice.toList).mkString
92   }
93 }

122 /**
123  * Command to continue with Mongo SCRAM-SHA1 authentication.
124  *
125  * @param user the name of the user
126  * @param password the user password
127  * @param conversationId
128  * @param payload the initial payload
129  * @param randomPrefix
130  * @param startMessage
131  */
132 case class ScramSha1StartNegociation(
133   user: String,
134   password: String,
135   conversationId: Int,
136   payload: Array[Byte],
137   randomPrefix: String,
138   startMessage: String
139 )
140     extends Command[Either[SuccessfulAuthentication, Array[Byte]]] {
141 
142   import javax.crypto.spec.PBEKeySpec
143   import org.apache.commons.codec.binary.Base64
144   import org.apache.commons.codec.digest.{ DigestUtils, HmacUtils }
145   import akka.util.ByteString
146   import reactivemongo.bson.buffer.ArrayReadableBuffer
147 
148   val data: Either[CommandError, ScramSha1Negociation] = {
149     val challenge = new String(payload, "UTF-8")
150     val response = ScramSha1Negociation.parsePayload(challenge)
151 
152     for {
153       rand <- response.get("r").filter(_ startsWith randomPrefix).
154         toRight(CommandError("invalid SCRAM-SHA1 random prefix")).right
155 
156       salt <- response.get("s").flatMap[Array[Byte]] { s =>
157         try { Some(Base64 decodeBase64 s) } catch { case _: Throwable => None }
158       }.toRight(CommandError("invalid SCRAM-SHA1 password salt")).right
159 
160       iter <- response.get("i").flatMap[Int] { i =>
161         try { Some(i.toInt) } catch { case _: Throwable => None }
162       }.toRight(CommandError("invalid SCRAM-SHA1 iteration count")).right
163 
164       nego <- try {
165         import HmacUtils.hmacSha1
166 
167         val nonce = s"c=biws,r=$rand" // biws = base64("n,,")
168         val saltedPassword: Array[Byte] = {
169           val digest = DigestUtils.md5Hex(s"$user:mongo:$password")
170           val spec = new PBEKeySpec(digest.toCharArray, salt, iter, 160 /* 20 * 8 = 20 bytes */ )
171           ScramSha1StartNegociation.keyFactory.generateSecret(spec).getEncoded
172         }
173         val authMsg =
174           s"${startMessage.drop(3)},$challenge,$nonce".getBytes("UTF-8")
175 
176         val clientKey =
177           hmacSha1(saltedPassword, ScramSha1StartNegociation.ClientKeySeed)
178         val clientSig = hmacSha1(DigestUtils.sha1(clientKey), authMsg)
179         val clientProof: Array[Byte] = (clientKey, clientSig).
180           zipped.map((a, b) => (a ^ b).toByte)
181 
182         val message = s"$nonce,p=${Base64.encodeBase64String(clientProof)}"
183         val serverKey =
184           hmacSha1(saltedPassword, ScramSha1StartNegociation.ServerKeySeed)
185 
186         Right(ScramSha1Negociation(
187           serverSignature = hmacSha1(serverKey, authMsg),
188           request = BSONDocument(
189             "saslContinue" -> 1, "conversationId" -> conversationId,
190             "payload" -> BSONBinary(
191               ArrayReadableBuffer(ByteString(message).toArray[Byte]),
192               Subtype.GenericBinarySubtype
193             )
194           )
195         )).right
196 
197       } catch {
198         case err: Throwable => Left(CommandError(
199           s"fails to negociate SCRAM-SHA1: ${err.getMessage}"
200         )).right
201       }
202     } yield nego
203   }
204 
205   def serverSignature: Either[CommandError, Array[Byte]] =
206     data.right.map(_.serverSignature)
207 
208   override def makeDocuments = data.right.map(_.request).right.get
209 
210   val ResultMaker = ScramSha1StartNegociation
211 }
