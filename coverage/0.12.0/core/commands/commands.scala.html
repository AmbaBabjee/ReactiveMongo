43 /**
44  * A MongoDB Command.
45  *
46  * Basically, it's as query that is performed on any db.\$cmd collection
47  * and gives back one document as a result.
48  *
49  * @param Result This command's result type.
50  */
51 trait Command[Result] {
52   /**
53    * Deserializer for this command's result.
54    */
55   val ResultMaker: CommandResultMaker[Result]
56 
57   /**
58    * States if this command can be run on secondaries.
59    */
60   def slaveOk: Boolean = false
61 
62   /**
63    * Makes the `BSONDocument` for documents that will be send as body of this command's query.
64    */
65   def makeDocuments: BSONDocument
66 
67   /**
68    * Produces a [[reactivemongo.core.commands.MakableCommand]] instance of this command.
69    *
70    * @param db name of the target database.
71    */
72   def apply(db: String): MakableCommand = new MakableCommand(db, this)
73 }
74 
75 /**
76  * Handler for deserializing commands results.
77  *
78  * @tparam Result The result type of this command.
79  */
80 trait CommandResultMaker[Result] {
81   /**
82    * Deserializes the given response into an instance of Result.
83    */
84   def apply(response: Response): Either[CommandError, Result]
85 }
86 
87 trait BSONCommandResultMaker[Result] extends CommandResultMaker[Result] {
88   final def apply(response: Response): Either[CommandError, Result] = {
89     val document = Response.parse(response).next()
90     try {
91       apply(document)
92     } catch {
93       case e: CommandError => Left(e)
94       case e: Throwable =>
95         val error = CommandError("exception while deserializing this command's result!", Some(document))
96         error.initCause(e);
97         Left(error)
98     }
99   }
100 
101   /**
102    * Deserializes the given document into an instance of Result.
103    */
104   def apply(document: BSONDocument): Either[CommandError, Result]
105 }

140 object CommandError {
141   /**
142    * Makes a 'DefaultCommandError'.
143    *
144    * @param message The error message.
145    * @param originalDocument The original document contained in the response.
146    * @param code The code of the error, if any.
147    */
148   def apply(message: String, originalDocument: Option[BSONDocument] = None, code: Option[Int] = None): DefaultCommandError =
149     new DefaultCommandError(message, code, originalDocument)
150 
151   /**
152    * Checks if the given document contains a 'ok' field which value equals 1, and produces a command error if not.
153    *
154    * @param doc The document of the response.
155    * @param name The optional name of the command.
156    * @param error A function that takes the document of the response and the optional name of the command as arguments, and produces a command error.
157    */
158   def checkOk(
159     doc: BSONDocument, name: Option[String],
160     error: (BSONDocument, Option[String]) => CommandError = (doc, name) => CommandError("command " + name.map(_ + " ").getOrElse("") + "failed because the 'ok' field is missing or equals 0", Some(doc))
161   ): Option[CommandError] = {
162     doc.getAs[BSONNumberLike]("ok").map(_.toInt).orElse(Some(0)).flatMap {
163       case 1 => None
164       case _ => Some(error(doc, name))
165     }
166   }
167 }

182 /**
183  * A makable command, that can produce a request maker ready to be sent to a [[reactivemongo.core.actors.MongoDBSystem]] actor.
184  *
185  * @param db Database name.
186  * @param command Subject command.
187  */
188 class MakableCommand(val db: String, val command: Command[_]) {
189   /**
190    * Produces the [[reactivemongo.core.protocol.Query]] instance for the given command.
191    */
192   def makeQuery: Query = Query(if (command.slaveOk) QueryFlags.SlaveOk else 0, db + ".$cmd", 0, 1)
193 
194   /**
195    * Returns the [[reactivemongo.core.protocol.RequestMaker]] for the given command.
196    */
197   def maker = RequestMaker(
198     makeQuery,
199     BufferSequence.single(command.makeDocuments)
200   )
201 
202   /**
203    * Returns the [[reactivemongo.core.protocol.RequestMaker]] for the given command, using the given ReadPreference.
204    */
205   def maker(readPreference: ReadPreference) = {
206     val query = makeQuery
207     val flags = {
208       if (readPreference.slaveOk) query.flags | QueryFlags.SlaveOk
209       else query.flags
210     }
211 
212     RequestMaker(
213       query.copy(flags = flags),
214       BufferSequence.single(command.makeDocuments), readPreference
215     )
216   }
217 }

346 /**
347  * Deserializer for [[reactivemongo.core.commands.GetLastError GetLastError]] command result.
348  */
349 @deprecated("consider using reactivemongo.api.commands.LastError instead", "0.11.0")
350 object LastError extends BSONCommandResultMaker[LastError] {
351   def apply(document: BSONDocument) = {
352     Right(LastError(
353       document.getAs[BSONDouble]("ok").map(_.value == 1).getOrElse(true),
354       document.getAs[BSONString]("err").map(_.value),
355       document.getAs[BSONInteger]("code").map(_.value),
356       document.getAs[BSONString]("errmsg").map(_.value),
357       Some(document),
358       document.getAs[BSONInteger]("n").map(_.value).getOrElse(0),
359       document.getAs[BSONBoolean]("updatedExisting").map(_.value).getOrElse(false)
360     ))
361   }
362   def meaningful(response: Response) = {
363     apply(response) match {
364       case Left(e) => Left(e)
365       case Right(lastError) =>
366         if (lastError.inError)
367           Left(lastError)
368         else Right(lastError)
369     }
370   }
371 }
