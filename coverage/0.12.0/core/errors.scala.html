23 /** An error that can come from a MongoDB node or not. */
24 trait ReactiveMongoException extends Exception {
25   /** explanation message */
26   def message: String
27 
28   override def getMessage: String = s"MongoError['$message']"
29 }
30 
31 object ReactiveMongoException {
32   def apply(message: String): ReactiveMongoException =
33     GenericDriverException(message)
34 
35   def apply(doc: BSONDocument): DatabaseException =
36     new DetailedDatabaseException(doc)
37 }
38 
39 /** An error thrown by a MongoDB node. */
40 trait DatabaseException extends ReactiveMongoException {
41   /** original document of this error */
42   def originalDocument: Option[BSONDocument]
43 
44   /** error code */
45   def code: Option[Int]
46 
47   override def getMessage: String = s"DatabaseException['$message'" + code.map(c => s" (code = $c)").getOrElse("") + "]"
48 
49   /** Tells if this error is due to a write on a secondary node. */
50   def isNotAPrimaryError: Boolean = code.map {
51     case 10054 | 10056 | 10058 | 10107 | 13435 | 13436 => true
52     case _ => false
53   }.getOrElse(false)
54 
55   /** Tells if this error is related to authentication issues. */
56   def isUnauthorized: Boolean = code.map {
57     case 10057 | 15845 | 16550 => true
58     case _                     => false
59   }.getOrElse(false)
60 }
61 
62 /** A driver-specific error */
63 trait DriverException extends ReactiveMongoException {
64   protected def cause: Throwable = null
65   override def getCause = cause
66 }
67 
68 /** A generic driver error. */
69 case class GenericDriverException(
70   message: String
71 ) extends DriverException with NoStackTrace

73 sealed class ConnectionNotInitialized(
74   val message: String,
75   override val cause: Throwable
76 ) extends DriverException
77     with Product with java.io.Serializable with Serializable with Equals {
78 
79   @deprecated(message = "Use constructor with cause", since = "0.12-RC1")
80   def this(message: String) = this(message, null)
81 
82   override val productPrefix = "ConnectionNotInitialized"
83 
84   def productElement(i: Int): Any = i match {
85     case 0 => message
86     case 1 => cause
87     case _ => throw new NoSuchElementException
88   }
89 
90   override def productIterator: Iterator[Any] = Iterator(message, cause)
91 
92   val productArity = 2
93 
94   override lazy val hashCode = (message -> cause).hashCode
95 
96   override def equals(that: Any): Boolean = that match {
97     case x: ConnectionNotInitialized =>
98       (message -> cause) == (x.message -> x.cause)
99 
100     case _ => false
101   }
102 
103   def canEqual(other: Any): Boolean = other match {
104     case _: ConnectionNotInitialized => true
105     case _                           => false
106   }
107 
108   @deprecated(message = "Use constructor with cause", since = "0.12-RC1")
109   def copy(message: String): ConnectionNotInitialized =
110     new ConnectionNotInitialized(message, this.cause)
111 }

113 object ConnectionNotInitialized {
114   @deprecated(message = "Use constructor with cause", since = "0.12-RC1")

116     new ConnectionNotInitialized(message, null)
117 
118   @deprecated(message = "Use constructor with cause", since = "0.12-RC1")
119   def unapply(instance: ConnectionNotInitialized): Option[String] =

139 /** An error thrown by a MongoDB node (containing the original document of the error). */

143

146   lazy val code = doc.getAs[BSONInteger]("code").map(_.value)
147 }
