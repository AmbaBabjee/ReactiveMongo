115             case reason: Throwable =>

117           }

127             mbs.registerMBean(node, objName)

134           } catch {
135             case _: javax.management.InstanceAlreadyExistsException =>

137 
138             case reason: Throwable => logger.warn(
139               s"Fails to register the node MBean: $objName", reason
140             )
141           }
142       }
143 
144       // Updated nodes
145       (prev -- rmd.keys).foreach {
146         case (name, node) => try {
147           nodes.get(name).foreach { bean =>
148             bean.update(node)
149 
150             nodeSet.sendNotification("nodeUpdated", name,
151               s"The node properties have been updated: $name")
152 
153           }
154         } catch {

156             s"Fails to update the node MBean: $name", reason
