25 /**
26  * A reader that produces an instance of `T` from a subtype of [[BSONValue]].
27  */
28 trait BSONReader[B <: BSONValue, T] { self =>
29 
30   /**
31    * Reads a BSON value and produce an instance of `T`.
32    *
33    * This method may throw exceptions at runtime.
34    * If used outside a reader, one should consider `readTry(bson: B): Try[T]` or `readOpt(bson: B): Option[T]`.
35    */
36   def read(bson: B): T
37 
38   /** Tries to produce an instance of `T` from the `bson` value, returns `None` if an error occurred. */
39   def readOpt(bson: B): Option[T] = readTry(bson).toOption
40 
41   /** Tries to produce an instance of `T` from the `bson` value. */
42   def readTry(bson: B): Try[T] = Try(read(bson))
43 
44   /**
45    * Returns a BSON reader that returns the result of applying `f`
46    * on the result of this reader.
47    *
48    * @param f the function to apply
49    */
50   final def afterRead[U](f: T => U): BSONReader[B, U] =
51     BSONReader[B, U]((read _) andThen f)
52 
53   private[reactivemongo] def widenReader[U >: T]: UnsafeBSONReader[U] =
54     new UnsafeBSONReader[U] {
55       def readTry(value: BSONValue): Try[U] =
56         Try(value.asInstanceOf[B]) match {
57           case Failure(_) => Failure(exceptions.TypeDoesNotMatch(
58             s"Cannot convert $value: ${value.getClass} with ${self.getClass}"
59           ))
60 
61           case Success(bson) => self.readTry(bson)
62         }
63     }
64 }

78  * A writer that produces a subtype of [[BSONValue]] fron an instance of `T`.
79  */
80 trait BSONWriter[T, B <: BSONValue] {
81   /**
82    * Writes an instance of `T` as a BSON value.
83    *
84    * This method may throw exceptions at runtime.
85    * If used outside a reader, one should consider `writeTry(bson: B): Try[T]` or `writeOpt(bson: B): Option[T]`.
86    */
87   def write(t: T): B
88 
89   /** Tries to produce a BSON value from an instance of `T`, returns `None` if an error occurred. */
90   def writeOpt(t: T): Option[B] = writeTry(t).toOption
91 
92   /** Tries to produce a BSON value from an instance of `T`. */
93   def writeTry(t: T): Try[B] = Try(write(t))
94 
95   /**
96    * Returns a BSON writer that returns the result of applying `f`
97    * on the BSON value from this writer.
98    *
99    * @param f the function to apply
100    */
101   final def afterWrite[U <: BSONValue](f: B => U): BSONWriter[T, U] =
102     BSONWriter[T, U]((write _) andThen f)
103 
104   final def beforeWrite[U](f: U => T): BSONWriter[U, B] =
105     BSONWriter[U, B](f andThen (write _))
106 }

201 object BSONHandler {
202   private[bson] class MappedHandler[B <: BSONValue, T, U](
203       parent: BSONHandler[B, T],
204       to: T => U,
205       from: U => T
206   ) extends BSONHandler[B, U] {
207     def write(u: U) = parent.write(from(u))
208     def read(b: B) = to(parent.read(b))
209   }
210 
211   private[bson] class DefaultHandler[B <: BSONValue, T](r: B => T, w: T => B)
212       extends BSONHandler[B, T] {
213     def read(x: B): T = r(x)
214     def write(x: T): B = w(x)
215   }
216 
217   /**
218    * Handler factory.
219    *
220    * {{{
221    * import reactivemongo.bson.{ BSONHandler, BSONString }
222    *
223    * case class Foo(value: String)
224    *
225    * val foo: BSONHandler[BSONString, Foo] = BSONHandler(
226    *   { read: BSONString => Foo(read.value) },
227    *   { write: Foo => BSONString(write.value)
228    * )
229    * }}}
230    */
231   def apply[B <: BSONValue, T](read: B => T, write: T => B): BSONHandler[B, T] =
232     new DefaultHandler(read, write)
233 
234   /**
235    * Returns a BSON handler for a type `T`, provided there are
236    * a writer and a reader for it, both using the same kind of `BSONValue`.
237    */
238   implicit def provided[B <: BSONValue, T](implicit writer: BSONWriter[T, B], reader: BSONReader[B, T]): BSONHandler[B, T] = BSONHandler(reader.read _, writer.write _)
239 }

338   abstract class IdentityBSONConverter[T <: BSONValue](implicit m: Manifest[T]) extends BSONReader[T, T] with BSONWriter[T, T] {
339     override def write(t: T): T = m.runtimeClass.cast(t).asInstanceOf[T]
340     override def writeOpt(t: T): Option[T] = if (m.runtimeClass.isInstance(t)) Some(t.asInstanceOf[T]) else None
341     override def read(bson: T): T = m.runtimeClass.cast(bson).asInstanceOf[T]
342     override def readOpt(bson: T): Option[T] = if (m.runtimeClass.isInstance(bson)) Some(bson.asInstanceOf[T]) else None
343   }
