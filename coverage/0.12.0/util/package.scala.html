19   /** Makes an option of the value matching the condition. */
20   def option[T](cond: => Boolean, value: => T): Option[T] =
21     if (cond) Some(value) else None

23   import scala.concurrent.{
24     ExecutionContext,
25     Future,
26     Promise,
27     duration
28   }, duration.Duration
29 
30   case class EitherMappableFuture[A](future: Future[A]) {
31     def mapEither[E <: Throwable, B](f: A => Either[E, B])(implicit ec: ExecutionContext) = {
32       future.flatMap(
33         f(_) match {
34           case Left(e)  => Future.failed(e)
35           case Right(b) => Future.successful(b)
36         }
37       )
38     }
39   }
40   object EitherMappableFuture {
41     implicit def futureToEitherMappable[A](future: Future[A]): EitherMappableFuture[A] = EitherMappableFuture(future)
42   }
43 
44   object ExtendedFutures {
45     import akka.actor.ActorSystem
46 
47     // better way to this?
48     def DelayedFuture(millis: Long, system: ActorSystem): Future[Unit] = {
49       implicit val ec = system.dispatcher
50       val promise = Promise[Unit]()
51       system.scheduler.scheduleOnce(Duration.apply(millis, "millis"))(promise.success(()))
52       promise.future
53     }
54   }
