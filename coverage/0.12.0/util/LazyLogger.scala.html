16 package reactivemongo.util

18 object LazyLogger {
19   import org.slf4j.{ LoggerFactory, Logger }
20 
21   /**
22    * Returns the lazy logger matching the SLF4J `name`.
23    *
24    * @param name the logger name
25    */
26   def apply(name: String): LazyLogger =
27     new LazyLogger(LoggerFactory getLogger name)
28 
29   final class LazyLogger private[reactivemongo] (logger: Logger) {
30     /** Returns the corresponding SLF4J logger. */
31     def slf4j = logger
32 
33     def trace(s: => String) { if (logger.isTraceEnabled) logger.trace(s) }
34     def trace(s: => String, e: => Throwable) {
35       if (logger.isTraceEnabled) logger.trace(s, e)
36     }
37 
38     lazy val isDebugEnabled = logger.isDebugEnabled
39     def debug(s: => String) { if (isDebugEnabled) logger.debug(s) }
40     def debug(s: => String, e: => Throwable) {
41       if (isDebugEnabled) logger.debug(s, e)
42     }
43 
44     def info(s: => String) { if (logger.isInfoEnabled) logger.info(s) }
45     def info(s: => String, e: => Throwable) {
46       if (logger.isInfoEnabled) logger.info(s, e)
47     }
48 
49     def warn(s: => String) { if (logger.isWarnEnabled) logger.warn(s) }
50     def warn(s: => String, e: => Throwable) {
51       if (logger.isWarnEnabled) logger.warn(s, e)
52     }
53 
54     def error(s: => String) { if (logger.isErrorEnabled) logger.error(s) }
55     def error(s: => String, e: => Throwable) {
56       if (logger.isErrorEnabled) logger.error(s, e)
57     }
58   }
59 }
