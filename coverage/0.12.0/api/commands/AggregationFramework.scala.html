370   case class Sort(fields: SortOrder*) extends PipelineOperator {
371     val makePipe = makeDocument(Seq(
372       elementProducer("$sort", makeDocument(fields.map {
373         case Ascending(field)  => elementProducer(field, intValue(1))
374         case Descending(field) => elementProducer(field, intValue(-1))
375         case MetadataSort(field, keyword) => {
376           val meta = makeDocument(Seq(
377             elementProducer("$meta", stringValue(keyword.name))
378           ))
379 
380           elementProducer(field, meta)
381         }
382       }))
383     ))

387    * Outputs documents in order of nearest to farthest from a specified point.
388    *
389    * http://docs.mongodb.org/manual/reference/operator/aggregation/geoNear/#pipe._S_geoNear
390    * @param near the point for which to find the closest documents
391    * @param spherical if using a 2dsphere index
392    * @param limit the maximum number of documents to return
393    * @param maxDistance the maximum distance from the center point that the documents can be
394    * @param query limits the results to the matching documents
395    * @param distanceMultiplier the factor to multiply all distances returned by the query
396    * @param uniqueDocs if this value is true, the query returns a matching document once
397    * @param distanceField the output field that contains the calculated distance
398    * @param includeLocs this specifies the output field that identifies the location used to calculate the distance
399    */
400   case class GeoNear(near: pack.Value, spherical: Boolean = false, limit: Long = 100, minDistance: Option[Long] = None, maxDistance: Option[Long] = None, query: Option[pack.Document] = None, distanceMultiplier: Option[Double] = None, uniqueDocs: Boolean = false, distanceField: Option[String] = None, includeLocs: Option[String] = None) extends PipelineOperator {
401     def makePipe = makeDocument(
402       Seq(elementProducer("$geoNear", makeDocument(Seq(
403         elementProducer("near", near),
404         elementProducer("spherical", booleanValue(spherical)),
405         elementProducer("limit", longValue(limit))
406       ) ++ Seq(
407           minDistance.map(l => elementProducer("minDistance", longValue(l))),
408           maxDistance.map(l => elementProducer("maxDistance", longValue(l))),
409           query.map(s => elementProducer("query", s)),
410           distanceMultiplier.map(d => elementProducer(
411             "distanceMultiplier", doubleValue(d)
412           )),
413           Some(elementProducer("uniqueDocs", booleanValue(uniqueDocs))),
414           distanceField.map(s =>
415             elementProducer("distanceField", stringValue(s))),
416           includeLocs.map(s =>
417             elementProducer("includeLocs", stringValue(s)))
418         ).flatten)))
419     )
420   }
