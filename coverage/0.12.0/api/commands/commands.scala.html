138   def defaultCursorFetcher[P <: SerializationPack, A](db: DB, p: P, command: A, failover: FailoverStrategy)(implicit writer: p.Writer[A]): CursorFetcher[p.type, DefaultCursor.Impl] = new CursorFetcher[p.type, DefaultCursor.Impl] {
139     val pack: p.type = p
140 
141     @inline private def defaultReadPreference: ReadPreference =
142       db.connection.options.readPreference
143 
144     def one[A](readPreference: ReadPreference)(implicit reader: pack.Reader[A], ec: ExecutionContext): Future[A] = {
145       val (requestMaker, m26WriteCommand) =
146         buildRequestMaker(pack)(command, writer, readPreference, db.name)
147 
148       Failover2(db.connection, failover) { () =>
149         db.connection.sendExpectingResponse(requestMaker, m26WriteCommand)
150       }.future.map { response =>
151         pack.readAndDeserialize(
152           LoweLevelDocumentIterator(ChannelBufferReadableBuffer(
153           response.documents
154         )).next, reader
155         )
156 
157       }
158     }
159 
160     def one[A](implicit reader: pack.Reader[A], ec: ExecutionContext): Future[A] = one[A](defaultReadPreference)
161 
162     def cursor[A](readPreference: ReadPreference)(implicit reader: pack.Reader[A]): DefaultCursor.Impl[A] = {
163       val buffer = ChannelBufferWritableBuffer()
164       pack.serializeAndWrite(buffer, command, writer)
165 
166       val bs = BufferSequence(buffer.buffer)
167       val flags = if (readPreference.slaveOk) QueryFlags.SlaveOk else 0
168       val op = Query(flags, db.name + ".$cmd", 0, 1)
169       val mongo26WriteCommand = command match {
170         case _: Mongo26WriteCommand => true
171         case _                      => false
172       }
173 
174       DefaultCursor.query(pack, op, bs,
175         if (mongo26WriteCommand) ReadPreference.primary else readPreference,
176         db.connection, failover, mongo26WriteCommand)
177 
178     }
179 
180     def cursor[A](implicit reader: pack.Reader[A]): DefaultCursor.Impl[A] =
181       cursor(defaultReadPreference)
182   }
