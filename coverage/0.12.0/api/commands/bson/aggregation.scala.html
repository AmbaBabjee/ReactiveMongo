51   implicit object AggregateWriter
52       extends BSONDocumentWriter[ResolvedCollectionCommand[Aggregate]] {
53     def write(agg: ResolvedCollectionCommand[Aggregate]) = {
54       val cmd = BSONDocument(
55         "aggregate" -> BSONString(agg.collection),
56         "pipeline" -> BSONArray(
57           { for (pipe <- agg.command.pipeline) yield pipe.makePipe }.toStream
58         ),
59         "explain" -> BSONBoolean(agg.command.explain),
60         "allowDiskUse" -> BSONBoolean(agg.command.allowDiskUse),
61         "cursor" -> agg.command.cursor.map(CursorWriter.write(_))
62       )
63 
64       if (agg.command.wireVersion < MongoWireVersion.V32) cmd else {
65         cmd ++ ("bypassDocumentValidation" -> BSONBoolean(
66           agg.command.bypassDocumentValidation
67         ),
68           "readConcern" -> agg.command.readConcern)
69       }
70     }
71   }

73   implicit object AggregationResultReader
74       extends DealingWithGenericCommandErrorsReader[AggregationResult] {
75     def readResult(doc: BSONDocument): AggregationResult =
76       if (doc contains "stages") AggregationResult(List(doc), None) else {
77         doc.getAs[List[BSONDocument]]("result") match {
78           case Some(docs) => AggregationResult(docs, None)
79           case _ => (for {
80             cursor <- doc.getAsTry[BSONDocument]("cursor")
81             id <- cursor.getAsTry[BSONNumberLike]("id").map(_.toLong)
82             ns <- cursor.getAsTry[String]("ns")
83             firstBatch <- cursor.getAsTry[List[BSONDocument]]("firstBatch")
84           } yield AggregationResult(firstBatch, Some(ResultCursor(id, ns)))).get
85         }
86       }
87   }
