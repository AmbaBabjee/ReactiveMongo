52 case class MultiBulkWriteResult(
53     ok: Boolean,
54     n: Int,
55     nModified: Int,
56     upserted: Seq[Upserted],
57     writeErrors: Seq[WriteError],
58     writeConcernError: Option[WriteConcernError], // TODO ?
59     code: Option[Int],
60     errmsg: Option[String],
61     totalN: Int
62 ) {
63   def merge(wr: WriteResult): MultiBulkWriteResult = wr match {
64     case wr: UpdateWriteResult => MultiBulkWriteResult(
65       ok = ok && wr.ok,
66       n = n + wr.n,
67       writeErrors = writeErrors ++ wr.writeErrors.map(e => e.copy(index = e.index + totalN)),
68       writeConcernError = writeConcernError.orElse(wr.writeConcernError),
69       code = code.orElse(wr.code),
70       errmsg = errmsg.orElse(wr.errmsg),
71       nModified = wr.nModified,
72       upserted = wr.upserted,
73       totalN = totalN + wr.n + wr.writeErrors.size
74     )
75     case _ =>
76       MultiBulkWriteResult(
77         ok = ok && wr.ok,
78         n = n + wr.n,
79         writeErrors = writeErrors ++ wr.writeErrors.map(e => e.copy(index = e.index + totalN)),
80         writeConcernError = writeConcernError.orElse(wr.writeConcernError),
81         code = code.orElse(wr.code),
82         errmsg = errmsg.orElse(wr.errmsg),
83         nModified = nModified,
84         upserted = upserted,
85         totalN = totalN + wr.n + wr.writeErrors.size
86       )
87   }
