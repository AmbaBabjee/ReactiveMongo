5 /**
6  * A failover strategy for sending requests.
7  * The default uses 10 retries:
8  * 125ms, 250ms, 375ms, 500ms, 625ms, 750ms, 875ms, 1s, 1125ms, 1250ms
9  *
10  * @param initialDelay the initial delay between the first failed attempt and the next one.
11  * @param retries the number of retries to do before giving up.
12  * @param delayFactor a function that takes the current iteration and returns a factor to be applied to the initialDelay (default: [[FailoverStrategy.defaultFactor]])
13  */
14 case class FailoverStrategy(
15     initialDelay: FiniteDuration = FiniteDuration(100, "ms"),
16     retries: Int = 10,
17     delayFactor: Int => Double = FailoverStrategy.defaultFactor
18 ) {
19 
20   override lazy val toString = delayFactor match {
21     case fn @ FailoverStrategy.FactorFun(_) =>
22       s"FailoverStrategy($initialDelay, $retries, $fn)"
23 
24     case _ => s"FailoverStrategy($initialDelay, $retries)"
25   }
26 }
27 
28 object FailoverStrategy {
29   /** The default strategy */
30   val default = FailoverStrategy()
31 
32   /** The strategy when the MongoDB nodes are remote (with 16 retries) */
33   val remote = FailoverStrategy(retries = 16)
34 
35   /** A more strict strategy: same as default but with less retries (=5). */
36   val strict = FailoverStrategy(retries = 5)
37 
38   /** A factor function using simple multiplication. */
39   case class FactorFun(multiplier: Double) extends (Int => Double) {
40     /**
41      * Returns the factor by which the initial delay must be multiplied,
42      * for the current try.
43      *
44      * @param `try` the current number of tries
45      */
46     final def apply(`try`: Int): Double = `try` * multiplier
47 
48     override lazy val toString = s"× $multiplier"
49   }
50 
51   /** The factor function for the default strategy: × 1.25 */
52   @inline def defaultFactor = FactorFun(1.25)
53   // 'def' instead of 'val' as used in defaults of the case class
