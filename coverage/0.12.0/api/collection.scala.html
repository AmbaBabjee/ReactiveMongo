20 /**
21  * A MongoDB Collection, resolved from a [[reactivemongo.api.DefaultDB]].
22  *
23  * You should consider the generic API
24  * ([[reactivemongo.api.collections.GenericCollection]])
25  * and the default [[reactivemongo.api.collections.bson.BSONCollection]].
26  *
27  * {{{
28  * import reactivemongo.bson._
29  * import reactivemongo.api.collections.bson.BSONCollection
30  *
31  * object Samples {
32  *
33  *   val connection = MongoConnection(List("localhost"))
34  *
35  *   // Gets a reference to the database "plugin"
36  *   val db = connection("plugin")
37  *
38  *   // Gets a reference to the collection "acoll"
39  *   // By default, you get a BSONCollection.
40  *   val collection = db[BSONCollection]("acoll")
41  *
42  *   def listDocs() = {
43  *     // Select only the documents which field 'firstName' equals 'Jack'
44  *     val query = BSONDocument("firstName" -> "Jack")
45  *     // select only the field 'lastName'
46  *     val filter = BSONDocument(
47  *       "lastName" -> 1,
48  *       "_id" -> 0)
49  *
50  *     // Get a cursor of BSONDocuments
51  *     val cursor = collection.find(query, filter).cursor[BSONDocument]
52  *     // Let's enumerate this cursor and print a readable representation of each document in the response
53  *     cursor.enumerate().apply(Iteratee.foreach { doc =>
54  *       println("found document: " + BSONDocument.pretty(doc))
55  *     })
56  *
57  *     // Or, the same with getting a list
58  *     val cursor2 = collection.find(query, filter).cursor[BSONDocument]
59  *     val futureList = cursor.collect[List]()
60  *     futureList.map { list =>
61  *       list.foreach { doc =>
62  *         println("found document: " + BSONDocument.pretty(doc))
63  *       }
64  *     }
65  *   }
66  * }
67  * }}}
68  */
69 trait Collection {
70   /** The database which this collection belong to. */
71   def db: DB
72 
73   /** The name of the collection. */
74   def name: String
75 
76   /** The default failover strategy for the methods of this collection. */
77   def failoverStrategy: FailoverStrategy
78 
79   /** Gets the full qualified name of this collection. */
80   def fullCollectionName = db.name + "." + name
81 
82   /**
83    * Gets another implementation of this collection.
84    * An implicit CollectionProducer[C] must be present in the scope, or it will be the default implementation ([[reactivemongo.api.collections.bson.BSONCollection]]).
85    *
86    * @param failoverStrategy Overrides the default strategy.
87    */
88   def as[C <: Collection](failoverStrategy: FailoverStrategy = failoverStrategy)(implicit producer: CollectionProducer[C] = collections.bson.BSONCollectionProducer): C = {
89     producer.apply(db, name, failoverStrategy)
90   }
91 
92   /**
93    * Gets another collection in the current database.
94    * An implicit CollectionProducer[C] must be present in the scope, or it will be the default implementation ([[reactivemongo.api.collections.bson.BSONCollection]]).
95    *
96    * @param name The other collection name.
97    * @param failoverStrategy Overrides the default strategy.
98    */
99   @deprecated("Consider using `sibling` instead", "0.10")
100   def sister[C <: Collection](name: String, failoverStrategy: FailoverStrategy = failoverStrategy)(implicit producer: CollectionProducer[C] = collections.bson.BSONCollectionProducer): C =
101     sibling(name, failoverStrategy)
102 
103   /**
104    * Gets another collection in the current database.
105    * An implicit CollectionProducer[C] must be present in the scope, or it will be the default implementation ([[reactivemongo.api.collections.bson.BSONCollection]]).
106    *
107    * @param name The other collection name.
108    * @param failoverStrategy Overrides the default strategy.
109    */
110   def sibling[C <: Collection](name: String, failoverStrategy: FailoverStrategy = failoverStrategy)(implicit producer: CollectionProducer[C] = collections.bson.BSONCollectionProducer): C =
111     producer.apply(db, name, failoverStrategy)
112 }
113 
114 /**
115  * A Producer of [[Collection]] implementation.
116  *
117  * This is used to get an implementation implicitly when getting a reference of a [[Collection]].
118  */
119 trait CollectionProducer[+C <: Collection] {
120   def apply(db: DB, name: String, failoverStrategy: FailoverStrategy = FailoverStrategy()): C
121 }

123 /**
124  * A mixin that provides commands about this Collection itself.
125  *
126  * @define autoIndexIdParam If true should automatically add an index on the `_id` field. By default, regular collections will have an indexed `_id` field, in contrast to capped collections. This MongoDB option is deprecated and will be removed in a future release.
127  * @define cappedSizeParam the size of the collection (number of bytes)
128  * @define cappedMaxParam the maximum number of documents this capped collection can contain
129  */
130 trait CollectionMetaCommands { self: Collection =>
131   import scala.concurrent.{ ExecutionContext, Future }
132   import reactivemongo.api.commands._
133   import reactivemongo.api.commands.bson._
134   import CommonImplicits._
135   import BSONCreateImplicits._
136   import BSONEmptyCappedImplicits._
137   import BSONCollStatsImplicits._
138   import BSONRenameCollectionImplicits._
139   import BSONConvertToCappedImplicits._
140   import reactivemongo.api.indexes.CollectionIndexesManager
141 
142   /**
143    * Creates this collection.
144    *
145    * The returned future will be completed with an error if this collection already exists.
146    *
147    * @param autoIndexId $autoIndexIdParam
148    */
149   def create(autoIndexId: Boolean = true)(implicit ec: ExecutionContext): Future[Unit] = Command.run(BSONSerializationPack).unboxed(self, Create(None, autoIndexId))
150 
151   /**
152    * Creates this collection as a capped one.
153    *
154    * The returned future will be completed with an error if this collection already exists.
155    *
156    * @param size $cappedSizeParam
157    * @param maxDocuments $cappedMaxParam
158    * @param autoIndexId $autoIndexIdParam
159    */
160   def createCapped(size: Long, maxDocuments: Option[Int], autoIndexId: Boolean = false)(implicit ec: ExecutionContext): Future[Unit] =
161     Command.run(BSONSerializationPack).
162       unboxed(self, Create(Some(Capped(size, maxDocuments)), autoIndexId))
163 
164   /**
165    * Drops this collection.
166    *
167    * The returned future will be completed with an error
168    * if this collection does not exist.
169    */
170   @deprecated("Use `drop(Boolean)`", "0.12.0")
171   def drop()(implicit ec: ExecutionContext): Future[Unit] =
172     drop(true).map(_ => {})
173 
174   /**
175    * Drops this collection.
176    *
177    * If the collection existed and is successfully dropped,
178    * the returned future will be completed with true.
179    *
180    * If `failIfNotFound` is false and the collection doesn't exist,
181    * the returned future will be completed with false.
182    *
183    * Otherwise in case, the future will be completed with the encountered error.
184    */
185   def drop(failIfNotFound: Boolean)(implicit ec: ExecutionContext): Future[Boolean] = {
186     import BSONDropCollectionImplicits._
187 
188     Command.run(BSONSerializationPack)(self, DropCollection).flatMap {
189       case DropCollectionResult(false) if failIfNotFound =>
190         Future.failed[Boolean](GenericDatabaseException(
191           s"fails to drop collection: $name", Some(26)
192         ))
193 
194       case DropCollectionResult(dropped) => Future.successful(dropped)
195     }
196   }
197 
198   /**
199    * If this collection is capped, removes all the documents it contains.
200    *
201    * Deprecated because it became an internal command, unavailable by default.
202    */
203   @deprecated("Deprecated because emptyCapped became an internal command, unavailable by default.", "0.9")
204   def emptyCapped()(implicit ec: ExecutionContext): Future[Unit] =
205     Command.run(BSONSerializationPack).unboxed(self, EmptyCapped)
206 
207   /**
208    * Converts this collection to a capped one.
209    *
210    * @param size $cappedSizeParam
211    * @param maxDocuments $cappedMaxParam
212    */
213   def convertToCapped(size: Long, maxDocuments: Option[Int])(implicit ec: ExecutionContext): Future[Unit] = Command.run(BSONSerializationPack).unboxed(self, ConvertToCapped(Capped(size, maxDocuments)))
214 
215   /**
216    * Renames this collection.
217    *
218    * @param to The new name of this collection.
219    * @param dropExisting If a collection of name `to` already exists, then drops that collection before renaming this one.
220    *
221    * @return a failure if the dropExisting option is false and the target collection already exists
222    */
223   def rename(to: String, dropExisting: Boolean = false)(implicit ec: ExecutionContext): Future[Unit] =
224     Command.run(BSONSerializationPack).unboxed(self.db, RenameCollection(db.name + "." + name, db.name + "." + to, dropExisting))
225 
226   /**
227    * Returns various information about this collection.
228    */
229   def stats()(implicit ec: ExecutionContext): Future[CollStatsResult] =
230     Command.run(BSONSerializationPack)(self, CollStats(None))
231 
232   /**
233    * Returns various information about this collection.
234    *
235    * @param scale A scale factor (for example, to get all the sizes in kilobytes).
236    */
237   def stats(scale: Int)(implicit ec: ExecutionContext): Future[CollStatsResult] = Command.run(BSONSerializationPack)(self, CollStats(Some(scale)))
238 
239   /** Returns an index manager for this collection. */
240   def indexesManager(implicit ec: ExecutionContext): CollectionIndexesManager =
241     CollectionIndexesManager(self.db, name)
