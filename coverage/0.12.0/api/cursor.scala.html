43 /**
44  * Cursor over results from MongoDB.
45  *
46  * @tparam T the type parsed from each result document
47  * @define maxDocsParam The maximum number of documents to be retrieved
48  * @define maxDocsWarning The actual document count can exceed this, when this maximum devided by the batch size given a non-zero remainder
49  * @define stopOnErrorParam States if may stop on non-fatal exception (default: true). If set to false, the exceptions are skipped, trying to get the next result.
50  * @define errorHandlerParam The binary operator to be applied when failing to get the next response. Exception or [[reactivemongo.api.Cursor$.Fail Fail]] raised within the `suc` function cannot be recovered by this error handler.
51  * @define zeroParam the initial value
52  * @define getHead Returns the first document matching the query
53  * @define collect Collects all the documents into a collection of type `M[T]`
54  * @define resultTParam the result type of the binary operator
55  * @define sucRespParam The binary operator to be applied when the next response is successfully read
56  * @define foldResp Applies a binary operator to a start value and all responses handled by this cursor, going first to last.
57  * @define sucSafeWarning This must be safe, and any error must be returned as `Future.failed[State[A]]`
58  * @define foldBulks Applies a binary operator to a start value and all bulks of documents retrieved by this cursor, going first to last.
59  * @define foldWhile Applies a binary operator to a start value and all elements retrieved by this cursor, going first to last.
60  * @define sucDocParam The binary operator to be applied when the next document is successfully read
61  */
62 trait Cursor[T] {

71    * even if it processes `maxDocs` documents.

102   /**
103    * $collect.
104    *
105    * @param maxDocs $maxDocsParam.
106    * @param err $errorHandlerParam
107    *
108    * {{{
109    * val cursor = collection.find(query, filter).cursor[BSONDocument]
110    * // return the 3 first documents in a Vector[BSONDocument].
111    * val vector = cursor.collect[Vector](3, Cursor.FailOnError())
112    * }}}
113    */
114   def collect[M[_]](maxDocs: Int, err: ErrorHandler[M[T]])(implicit cbf: CanBuildFrom[M[_], T, M[T]], ec: ExecutionContext): Future[M[T]]

116   /**
117    * $collect.
118    *
119    * @param maxDocs $maxDocsParam.
120    * @param stopOnError $stopOnErrorParam.
121    *
122    * {{{
123    * val cursor = collection.find(query, filter).cursor[BSONDocument]
124    * // return the 3 first documents in a Vector[BSONDocument].
125    * val vector = cursor.collect[Vector](3)
126    * }}}
127    */
128   @deprecated(message = "Use `collect` with an [[Cursor.ErrorHandler]].", since = "0.12-RC1")
129   def collect[M[_]](maxDocs: Int = Int.MaxValue, stopOnError: Boolean = true)(implicit cbf: CanBuildFrom[M[_], T, M[T]], ec: ExecutionContext): Future[M[T]] = {
130     val err = if (stopOnError) FailOnError[M[T]]() else ContOnError[M[T]]()
131     collect[M](maxDocs, err)
132   }

176   def foldBulksM[A](z: => A, maxDocs: Int = Int.MaxValue)(suc: (A, Iterator[T]) => Future[Cursor.State[A]], err: ErrorHandler[A] = FailOnError[A]())(implicit ctx: ExecutionContext): Future[A]

178   /**
179    * $foldWhile
180    *
181    * @param z $zeroParam
182    * @param maxDocs $maxDocsParam.
183    * @param suc $sucDocParam.
184    * @param err $errorHandlerParam
185    * @tparam A $resultTParam
186    *
187    * {{{
188    * cursor.foldWhile(Nil: Seq[Person])((s, p) => Cursor.Cont(s :+ p),
189    *   { (l, e) => println("last valid value: " + l); Cursor.Fail(e) })
190    * }}}
191    */
192   def foldWhile[A](z: => A, maxDocs: Int = Int.MaxValue)(suc: (A, T) => Cursor.State[A], err: ErrorHandler[A] = FailOnError[A]())(implicit ctx: ExecutionContext): Future[A]
193 
194   /**
195    * $foldWhile
196    *
197    * @param z $zeroParam
198    * @param maxDocs $maxDocsParam.
199    * @param suc $sucDocParam. $sucSafeWarning.
200    * @param err $errorHandlerParam
201    * @tparam A $resultTParam
202    *
203    * {{{
204    * cursor.foldWhileM(Nil: Seq[Person])(
205    *   (s, p) => Future.successful(Cursor.Cont(s :+ p)),
206    *   { (l, e) => Future {
207    *     println("last valid value: " + l)
208    *     Cursor.Fail(e)
209    *   })
210    * }}}
211    */
212   def foldWhileM[A](z: => A, maxDocs: Int = Int.MaxValue)(suc: (A, T) => Future[Cursor.State[A]], err: ErrorHandler[A] = FailOnError[A]())(implicit ctx: ExecutionContext): Future[A]
213 
214   /**
215    * $foldWhile
216    *
217    * @param z $zeroParam
218    * @param maxDocs $maxDocsParam.
219    * @param suc $sucDocParam.
220    * @param err $errorHandlerParam
221    * @tparam A $resultTParam
222    *
223    * {{{
224    * cursor.foldWhile(Nil: Seq[Person])((s, p) => Cursor.Cont(s :+ p),
225    *   { (l, e) => println("last valid value: " + l); Cursor.Fail(e) })
226    * }}}
227    */
228   def fold[A](z: => A, maxDocs: Int = Int.MaxValue)(suc: (A, T) => A)(implicit ctx: ExecutionContext): Future[A] = foldWhile[A](z, maxDocs)(
229     { (st, v) => Cursor.Cont[A](suc(st, v)) }, FailOnError[A]()
230   )
231 
232   /** Returns the list of the matching documents. */
233   @deprecated("consider using collect[List] instead", "0.10.0")
234   def toList(maxDocs: Int = Int.MaxValue, stopOnError: Boolean = true)(implicit ctx: ExecutionContext): Future[List[T]] = collect[List](maxDocs, stopOnError)

396 /** Cursor companion object */
397 object Cursor {
398   private[api] val logger = LazyLogger("reactivemongo.api.Cursor")
399 
400   /**
401    * @tparam A the state type
402    * @see [[Cursor.foldWhile]]
403    */
404   type ErrorHandler[A] = (A, Throwable) => State[A]
405 
406   /**
407    * Error handler to fail on error (see [[Cursor.foldWhile]] and [[reactivemongo.api.Cursor$.Fail Fail]]).
408    *
409    * @param callback the callback function applied on last (possibily initial) value and the encountered error
410    */
411   def FailOnError[A](callback: (A, Throwable) => Unit = (_: A, _: Throwable) => {}): ErrorHandler[A] = (v: A, e: Throwable) => { callback(v, e); Fail(e): State[A] }
412 
413   /**
414    * Error handler to end on error (see [[Cursor.foldWhile]] and [[reactivemongo.api.Cursor$.Done Done]]).
415    *
416    * @param callback the callback function applied on last (possibily initial) value and the encountered error
417    */
418   def DoneOnError[A](callback: (A, Throwable) => Unit = (_: A, _: Throwable) => {}): ErrorHandler[A] = (v: A, e: Throwable) => { callback(v, e); Done(v): State[A] }
419 
420   /**
421    * Error handler to continue on error (see [[Cursor.foldWhile]] and [[reactivemongo.api.Cursor$.Cont Cont]]).
422    *
423    * @param callback the callback function applied on last (possibily initial) value and the encountered error
424    */
425   def ContOnError[A](callback: (A, Throwable) => Unit = (_: A, _: Throwable) => {}): ErrorHandler[A] = (v: A, e: Throwable) => { callback(v, e); Cont(v): State[A] }
426 
427   /**
428    * Value handler, ignoring the values (see [[Cursor.foldWhile]] and [[reactivemongo.api.Cursor$.Cont Cont]]).
429    *
430    * @param callback the callback function applied on each value.
431    */
432   def Ignore[A](callback: A => Unit = (_: A) => {}): (Unit, A) => State[Unit] = { (_, a) => Cont(callback(a)) }
433 
434   /**
435    * Flattens the given future [[reactivemongo.api.Cursor]] to a [[reactivemongo.api.FlattenedCursor]].
436    */
437   def flatten[T, C[_] <: Cursor[_]](future: Future[C[T]])(implicit fs: CursorFlattener[C]): C[T] = fs.flatten(future)
438 
439   /** A state of the cursor processing. */
440   sealed trait State[T] {
441     /**
442      * @param f the function applied on the statue value
443      */
444     def map[U](f: T => U): State[U]
445   }
446 
447   /** Continue with given value */
448   case class Cont[T](value: T) extends State[T] {
449     def map[U](f: T => U): State[U] = Cont(f(value))
450   }
451 
452   /** Successfully stop processing with given value */
453   case class Done[T](value: T) extends State[T] {
454     def map[U](f: T => U): State[U] = Done(f(value))
455   }
456 
457   /** Ends processing due to failure of given `cause` */
458   case class Fail[T](cause: Throwable) extends State[T] {
459     def map[U](f: T => U): State[U] = Fail[U](cause)
460   }
461 
462   /** Indicates that a required result cannot be found. */
463   case object NoSuchResultException
464     extends java.util.NoSuchElementException()
465     with scala.util.control.NoStackTrace
466 }

704     def foldWhileM[T](z: => T, maxDocs: Int = Int.MaxValue)(suc: (T, A) => Future[State[T]], err: (T, Throwable) => State[T])(implicit ctx: ExecutionContext): Future[T] = {
705       def go(v: T, it: Iterator[A]): Future[State[T]] =
706         if (!it.hasNext) Future.successful(Cont(v))
707         else Try(it.next) match {
708           case Failure(
709             x @ ReplyDocumentIteratorExhaustedException(_)) =>
710             Future.successful(Fail(x))
711 
712           case Failure(e) => err(v, e) match {
713             case Cont(cv) => go(cv, it)
714             case f @ Fail(Unrecoverable(_)) =>
715               /* already marked unrecoverable */ Future.successful(f)
716             case Fail(u) => Future.successful(Fail(Unrecoverable(u)))
717             case st      => Future.successful(st)
718           }
719 
720           case Success(a) => suc(v, a).flatMap {
721             case Cont(cv)    => go(cv, it)
722             case f @ Fail(_) => Future.successful(f)
723             case st          => Future.successful(st)
724           }
725         }
726 
727       foldBulksM(z, maxDocs)(go, err)

834     private class FoldResponses[T](
835         z: => T, maxDocs: Int,
836         suc: (T, Response) => Future[State[T]],
837         err: ErrorHandler[T]
838     )(implicit ctx: ExecutionContext) {
839 
840       private val nextResp: Response => Future[Option[Response]] =
841         nextResponse(maxDocs)(ctx, _: Response)
842 
843       @inline def ok(r: Response, v: T) = {
844         // Releases cursor before ending
845         killCursors(r.reply.cursorID, "FoldResponses")
846         Future.successful(v)
847       }
848 
849       @inline def kill(r: Response, f: Throwable) = {
850         killCursors(r.reply.cursorID, "FoldResponses")
851         Future.failed[T](f)
852       }
853 
854       def procResp(resp: Response, cur: T, c: Int): Future[T] = {
855         logger.trace(s"Process response: $resp")
856 
857         suc(cur, resp).transform(resp -> _, { error =>
858           killCursors(resp.reply.cursorID, "FoldResponses")
859           error
860         }).flatMap {
861           case (r, next) =>
862             val nc = c + resp.reply.numberReturned
863 
864             next match {
865               case Done(d)                    => ok(r, d)
866               case Cont(v) if (nc >= maxDocs) => ok(r, v)
867               case Fail(u @ Unrecoverable(_)) =>
868                 /* already marked recoverable */ kill(r, u)
869               case Fail(f) => kill(r, Unrecoverable(f))
870               case Cont(v) =>
871                 nextResp(r).flatMap(_.fold(Future successful v) { x =>
872                   procResponses(Future.successful(x), v, nc)
873                 })
874             }
875         }.recoverWith {
876           case Unrecoverable(e) => Future.failed(e)
877           case e =>
878             val nc = c + 1 // resp.reply.numberReturned
879 
880             err(cur, e) match {
881               case Done(d)                    => Future.successful(d)
882               case Cont(v) if (nc >= maxDocs) => Future.successful(v)
883               case Fail(f)                    => Future.failed[T](f)
884               case Cont(v) =>
885                 procResp(resp, cur, nc) // retry
886             }
887         }
888       }
889 
890       def procResponses(done: Future[Response], cur: T, c: Int): Future[T] =
891         done.map[Try[Response]](Success(_)).
892           recover { case err => Failure(err) }.flatMap {
893             case Success(r) => procResp(r, cur, c)
894             case Failure(error) => {
895               logger.error("fails to send request", error)
896 
897               err(cur, error) match {
898                 case Done(v) => Future.successful(v)
899                 case Fail(e) => Future.failed(e)
900                 case Cont(v) => {
901                   logger.warn(
902                     "cannot continue after fatal request error", error
903                   )
904 
905                   Future.successful(v)
906                 }
907               }
908             }
909           }
910 
911       def apply(): Future[T] =
912         Future(z).flatMap(v => procResponses(makeRequest(maxDocs), v, 0))
913     }
