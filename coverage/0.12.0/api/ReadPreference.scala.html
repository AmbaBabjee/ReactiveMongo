31     val matching = tagSet.find(_.foldLeft(Map.empty[String, String]) {
32       case (ms, (k, v)) =>
33         doc.getAs[String](k).filter(_ == v) match {

208   private implicit class BSONDocumentWrapper(val underlying: BSONDocument) extends AnyVal {
209     def contains(doc: BSONDocument): Boolean = {
210 
211       val els = underlying.elements
212       doc.elements.forall { element =>
213         els.find {
214           case BSONElement(name, value) =>
215             element._1 == name && ((element._2, value) match {
216               case (d1: BSONDocument, d2: BSONDocument) =>
217                 d1.elements == d2.elements
218 
219               case (a1: BSONArray, a2: BSONArray) =>
220                 a1.values == a2.values
221 
222               case (v1, v2) => v1 == v2
223             })
224         }.isDefined
225       }
226     }
227   }
