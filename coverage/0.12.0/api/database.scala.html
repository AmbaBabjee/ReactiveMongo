37 /**
38  * The reference to a MongoDB database, obtained from a [[reactivemongo.api.MongoConnection]].
39  *
40  * You should consider the provided [[reactivemongo.api.DefaultDB]] implementation.
41  *
42  * {{{
43  * import reactivemongo.api._
44  *
45  * val connection = MongoConnection(List("localhost:27017"))
46  * val db = connection.database("plugin")
47  * val collection = db.map(_("acoll"))
48  * }}}
49  *
50  * @define resolveDescription Returns a [[reactivemongo.api.Collection]] from this database
51  * @define nameParam the name of the collection to resolve
52  */
53 sealed trait DB {
54   /** The [[reactivemongo.api.MongoConnection]] that will be used to query this database. */
55   def connection: MongoConnection
56 
57   /** This database name. */
58   def name: String
59 
60   /** A failover strategy for sending requests. */
61   def failoverStrategy: FailoverStrategy
62 
63   /**
64    * $resolveDescription (alias for the `collection` method).
65    *
66    * @param name $nameParam
67    */
68   def apply[C <: Collection](name: String, failoverStrategy: FailoverStrategy = failoverStrategy)(implicit producer: CollectionProducer[C] = collections.bson.BSONCollectionProducer): C = collection(name, failoverStrategy)
69 
70   /**
71    * $resolveDescription.
72    *
73    * @param name $nameParam
74    */
75   def collection[C <: Collection](name: String, failoverStrategy: FailoverStrategy = failoverStrategy)(implicit producer: CollectionProducer[C] = collections.bson.BSONCollectionProducer): C = producer(this, name, failoverStrategy)
76 
77   @inline def defaultReadPreference: ReadPreference =
78     connection.options.readPreference
79 
80   /**
81    * Sends a command and get the future result of the command.
82    *
83    * @param command The command to send.
84    * @param readPreference The ReadPreference to use for this command (defaults to [[MongoConnectionOptions.readPreference]]).
85    *
86    * @return a future containing the result of the command.
87    */
88   @deprecated("Consider using reactivemongo.api.commands along with `DefaultDB.runCommand` methods", "0.11.0")
89   def command[T](command: CoreCommand[T], readPreference: ReadPreference = defaultReadPreference)(implicit ec: ExecutionContext): Future[T] =
90     Failover(
91       command.apply(name).maker(readPreference),
92       connection, failoverStrategy
93     ).future.mapEither(command.ResultMaker(_))
94 
95   /** Authenticates the connection on this database. */
96   def authenticate(user: String, password: String)(implicit timeout: FiniteDuration): Future[SuccessfulAuthentication] = connection.authenticate(name, user, password)
97 
98   /**
99    * Returns the database of the given name on the same MongoConnection.
100    * @see [[sibling1]]
101    */
102   def sibling(name: String, failoverStrategy: FailoverStrategy = failoverStrategy)(implicit ec: ExecutionContext): DefaultDB = connection(name, failoverStrategy)
103 
104   /** Returns the database of the given name on the same MongoConnection. */
105   def sibling1(name: String, failoverStrategy: FailoverStrategy = failoverStrategy)(implicit ec: ExecutionContext): Future[DefaultDB] = connection.database(name, failoverStrategy)
106 
107 }

134 /** A mixin that provides commands about this database itself. */
135 trait DBMetaCommands { self: DB =>
136   import reactivemongo.core.protocol.MongoWireVersion
137   import reactivemongo.api.commands.{
138     Command,
139     DropDatabase,
140     ListCollectionNames,
141     ServerStatus,
142     ServerStatusResult,
143     UserRole,
144     WriteConcern
145   }
146   import reactivemongo.api.commands.bson.{
147     CommonImplicits,
148     BSONDropDatabaseImplicits,
149     BSONServerStatusImplicits,
150     BSONCreateUserCommand
151   }
152   import reactivemongo.api.commands.bson.BSONListCollectionNamesImplicits._
153   import CommonImplicits._
154   import BSONDropDatabaseImplicits._
155   import BSONServerStatusImplicits._
156 
157   /** Drops this database. */
158   def drop()(implicit ec: ExecutionContext): Future[Unit] =
159     Command.run(BSONSerializationPack).unboxed(self, DropDatabase)
160 
161   /** Returns an index manager for this database. */
162   def indexesManager(implicit ec: ExecutionContext) = IndexesManager(self)
163 
164   private lazy val collectionNameReader = new BSONDocumentReader[String] {
165     val prefixLength = name.size + 1
166 
167     def read(bson: BSONDocument) = bson.get("name").collect {
168       case BSONString(value) => value.substring(prefixLength)
169     }.getOrElse(throw new Exception(
170       "name is expected on system.namespaces query"
171     ))
172 
173   }
174 
175   /** Returns the names of the collections in this database. */
176   def collectionNames(implicit ec: ExecutionContext): Future[List[String]] = {
177     val wireVer = connection.metadata.map(_.maxWireVersion)
178 
179     if (wireVer.exists(_ >= MongoWireVersion.V30)) {
180       Command.run(BSONSerializationPack)(self, ListCollectionNames).map(_.names)
181     } else collection("system.namespaces").as[BSONCollection]().
182       find(BSONDocument(
183         "name" -> BSONRegex("^[^\\$]+$", "") // strip off any indexes
184       )).cursor(defaultReadPreference)(
185         collectionNameReader, ec, CursorProducer.defaultCursorProducer
186       ).collect[List]()
187   }
188 
189   /**
190    * [[https://docs.mongodb.com/manual/reference/command/renameCollection/ Renames a collection]].
191    * Can only be executed if the this database reference is the `admin` one.
192    *
193    * @param db the name of the database where the collection exists with the `current` name
194    * @param current the current name of the collection, in the specified `db`
195    * @param to the new name of this collection (inside the same `db`)
196    * @param dropExisting If a collection of name `to` already exists, then drops that collection before renaming this one.
197    *
198    * @return a failure if the dropExisting option is false and the target collection already exists
199    */
200   def renameCollection[C <: Collection](db: String, from: String, to: String, dropExisting: Boolean = false, failoverStrategy: FailoverStrategy = failoverStrategy)(implicit ec: ExecutionContext, producer: CollectionProducer[C] = collections.bson.BSONCollectionProducer): Future[C] = {
201     import reactivemongo.api.commands.RenameCollection
202     import reactivemongo.api.commands.bson.BSONRenameCollectionImplicits._
203 
204     Command.run(BSONSerializationPack, failoverStrategy).unboxed(
205       self, RenameCollection(s"${db}.$from", s"${db}.$to", dropExisting)
206     ).map(_ => self.collection(to))
207   }
208 
209   /** Returns the server status. */
210   def serverStatus(implicit ec: ExecutionContext): Future[ServerStatusResult] =
211     Command.run(BSONSerializationPack)(self, ServerStatus)
212 
213   /**
214    * Create the specified user.
215    *
216    * @param name the name of the user to be created
217    * @param pwd the user password (not required if the database uses external credentials)
218    * @param roles the roles granted to the user, possibly an empty to create users without roles
219    * @param digestPassword when true, the mongod instance will create the hash of the user password (default: `true`)
220    * @param writeConcern the optional level of [[https://docs.mongodb.com/manual/reference/write-concern/ write concern]]
221    * @param customData the custom data to associate with the user account
222    *
223    * @see https://docs.mongodb.com/manual/reference/command/createUser/
224    */
225   def createUser(
226     name: String,
227     pwd: Option[String],
228     roles: List[UserRole],
229     digestPassword: Boolean = true,
230     writeConcern: WriteConcern = connection.options.writeConcern,
231     customData: Option[BSONDocument] = None
232   )(implicit ec: ExecutionContext): Future[Unit] = {
233     val command = BSONCreateUserCommand.CreateUser(
234       name, pwd, roles, digestPassword, Some(writeConcern), customData
235     )
236 
237     Command.run(BSONSerializationPack)(self, command).map(_ => {})
238   }
239 }
