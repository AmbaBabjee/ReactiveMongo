42 /** Type of Index */
43 sealed trait IndexType {
44   /** Value of the index (`{fieldName: value}`). */
45   def value: BSONValue
46   private[indexes] def valueStr: String
47 }
48 
49 object IndexType {
50   object Ascending extends IndexType {
51     def value = BSONInteger(1)
52     def valueStr = "1"
53   }
54 
55   object Descending extends IndexType {
56     def value = BSONInteger(-1)
57     def valueStr = "-1"
58   }
59 
60   object Geo2D extends IndexType {
61     def value = BSONString("2d")
62     def valueStr = "2d"
63   }
64 
65   object Geo2DSpherical extends IndexType {
66     def value = BSONString("2dsphere")
67     def valueStr = "2dsphere"
68   }
69 
70   object GeoHaystack extends IndexType {
71     def value = BSONString("geoHaystack")
72     def valueStr = "geoHaystack"
73   }
74 
75   object Hashed extends IndexType {
76     def value = BSONString("hashed")
77     def valueStr = "hashed"
78   }
79 
80   object Text extends IndexType {
81     def value = BSONString("text")
82     def valueStr = "text"
83   }
84 
85   def apply(value: BSONValue) = value match {
86     case BSONInteger(i) if i > 0             => Ascending
87     case BSONInteger(i) if i < 0             => Descending
88     case BSONDouble(i) if i > 0              => Ascending
89     case BSONDouble(i) if i < 0              => Descending
90     case BSONLong(i) if i > 0                => Ascending
91     case BSONLong(i) if i < 0                => Descending
92     case BSONString(s) if s == "2d"          => Geo2D
93     case BSONString(s) if s == "2dsphere"    => Geo2DSpherical
94     case BSONString(s) if s == "geoHaystack" => GeoHaystack
95     case BSONString(s) if s == "hashed"      => Hashed
96     case BSONString(s) if s == "text"        => Text
97     case _                                   => throw new IllegalArgumentException("unsupported index type")
98   }
99 }

101 /**
102  * A MongoDB index (excluding the namespace).
103  *
104  * Consider reading [[http://www.mongodb.org/display/DOCS/Indexes the documentation about indexes in MongoDB]].
105  *
106  * @param key The index key (it can be composed of multiple fields). This list should not be empty!
107  * @param name The name of this index. If you provide none, a name will be computed for you.
108  * @param unique Enforces uniqueness.
109  * @param background States if this index should be built in background. You should read [[http://www.mongodb.org/display/DOCS/Indexes#Indexes-background%3Atrue the documentation about background indexing]] before using it.
110  * @param dropDups States if duplicates should be discarded (if unique = true). Warning: you should read [[http://www.mongodb.org/display/DOCS/Indexes#Indexes-dropDups%3Atrue the documentation]].
111  * @param sparse States if the index to build should only consider the documents that have the indexed fields. See [[http://www.mongodb.org/display/DOCS/Indexes#Indexes-sparse%3Atrue the documentation]] on the consequences of such an index.
112  * @param version Indicates the [[http://www.mongodb.org/display/DOCS/Index+Versions version]] of the index (1 for >= 2.0, else 0). You should let MongoDB decide.
113  * @param options Optional parameters for this index (typically specific to an IndexType like Geo2D).
114  * @param partialFilter Optional [[https://docs.mongodb.com/manual/core/index-partial/#partial-index-with-unique-constraints partial filter]] (since MongoDB 3.2)
115  */
116 case class Index(
117     key: Seq[(String, IndexType)],
118     name: Option[String] = None,
119     unique: Boolean = false,
120     background: Boolean = false,
121     dropDups: Boolean = false, // Deprecated since 2.6, TODO: Remove
122     sparse: Boolean = false,
123     version: Option[Int] = None, // let MongoDB decide
124     // TODO: storageEngine (new for Mongo3)
125     partialFilter: Option[BSONDocument] = None,
126     options: BSONDocument = BSONDocument()
127 ) {
128 
129   /** The name of the index (a default one is computed if none). */
130   lazy val eventualName: String = name.getOrElse(key.foldLeft("") { (name, kv) =>
131     name + (if (name.length > 0) "_" else "") + kv._1 + "_" + kv._2.valueStr
132   })
133 }

135 /**
136  * A MongoDB namespaced index.
137  * A MongoDB index is composed with the namespace (the fully qualified collection name) and the other fields of [[reactivemongo.api.indexes.Index]].
138  *
139  * Consider reading [[http://www.mongodb.org/display/DOCS/Indexes the documentation about indexes in MongoDB]].
140  *
141  * @param namespace The fully qualified name of the indexed collection.
142  * @param index The other fields of the index.
143  */
144 case class NSIndex(namespace: String, index: Index) {
145   val (dbName: String, collectionName: String) = {
146     val spanned = namespace.span(_ != '.')
147     spanned._1 -> spanned._2.drop(1)
148   }
149 }

151 /** Indexes manager at database level. */
152 sealed trait IndexesManager {
153 
154   /** Gets a future list of all the index on this database. */
155   def list(): Future[List[NSIndex]]
156 
157   /**
158    * Creates the given index only if it does not exist on this database.
159    *
160    * Warning: given the options you choose, and the data to index, it can be a long and blocking operation on the database.
161    * You should really consider reading [[http://www.mongodb.org/display/DOCS/Indexes]] before doing this, especially in production.
162    *
163    * @param nsIndex The index to create.
164    *
165    * @return a future containing true if the index was created, false if it already exists.
166    */
167   def ensure(nsIndex: NSIndex): Future[Boolean]
168 
169   /**
170    * Creates the given index.
171    *
172    * Warning: given the options you choose, and the data to index, it can be a long and blocking operation on the database.
173    * You should really consider reading [[http://www.mongodb.org/display/DOCS/Indexes]] before doing this, especially in production.
174    *
175    * @param nsIndex The index to create.
176    */
177   def create(nsIndex: NSIndex): Future[WriteResult]
178 
179   /**
180    * Deletes the given index on the given collection.
181    *
182    * @return The deleted index number.
183    */
184   @deprecated("Use drop instead", "0.11.0")
185   def delete(nsIndex: NSIndex): Future[Int] =
186     delete(nsIndex.collectionName, nsIndex.index.eventualName)
187 
188   /**
189    * Deletes the given index on the given collection.
190    *
191    * @return The number of indexes that were dropped.
192    */
193   @deprecated("Use drop instead", "0.11.0")
194   def delete(collectionName: String, indexName: String): Future[Int] =
195     drop(collectionName, indexName)
196 
197   /**
198    * Drops the given index on the given collection.
199    *
200    * @return The number of indexes that were dropped.
201    */
202   def drop(nsIndex: NSIndex): Future[Int] =
203     drop(nsIndex.collectionName, nsIndex.index.eventualName)
204 
205   /**
206    * Drops the given index on the given collection.
207    *
208    * @return The number of indexes that were dropped.
209    */
210   def drop(collectionName: String, indexName: String): Future[Int]
211 
212   /**
213    * Drops all the indexes on the given collection.
214    */
215   def dropAll(collectionName: String): Future[Int]
216 
217   /** Gets a manager for the given collection. */
218   def onCollection(name: String): CollectionIndexesManager
219 }

221 /**
222  * A helper class to manage the indexes on a Mongo 2.x database.
223  *
224  * @param db The subject database.
225  */
226 final class LegacyIndexesManager(db: DB)(
227     implicit
228     context: ExecutionContext
229 ) extends IndexesManager {
230 
231   val collection = db("system.indexes")
232 
233   def list(): Future[List[NSIndex]] = collection.find(BSONDocument()).cursor(db.connection.options.readPreference)(IndexesManager.NSIndexReader, context, CursorProducer.defaultCursorProducer).collect[List]()
234 
235   def ensure(nsIndex: NSIndex): Future[Boolean] = {
236     val query = BSONDocument(
237       "ns" -> BSONString(nsIndex.namespace),
238       "name" -> BSONString(nsIndex.index.eventualName)
239     )
240 
241     collection.find(query).one.flatMap { idx =>
242       if (!idx.isDefined)
243         create(nsIndex).map(_ => true)
244       // there is a match, returning a future ok. TODO
245       else Future.successful(false)
246     }
247   }
248 
249   def create(nsIndex: NSIndex): Future[WriteResult] = {
250     implicit val writer = IndexesManager.NSIndexWriter
251     collection.insert(nsIndex)
252   }
253 
254   def drop(collectionName: String, indexName: String): Future[Int] = {
255     import reactivemongo.api.commands.bson.BSONDropIndexesImplicits._
256     db.collection(collectionName).runValueCommand(DropIndexes(indexName))
257   }
258 
259   def dropAll(collectionName: String): Future[Int] = drop(collectionName, "*")
260 
261   def onCollection(name: String): CollectionIndexesManager =
262     new LegacyCollectionIndexesManager(db.name, name, this)
263 }

265 /**
266  * A helper class to manage the indexes on a Mongo 3.x database.
267  *
268  * @param db The subject database.
269  */
270 final class DefaultIndexesManager(db: DB with DBMetaCommands)(
271     implicit
272     context: ExecutionContext
273 ) extends IndexesManager {
274 
275   private def listIndexes(collections: List[String], indexes: List[NSIndex]): Future[List[NSIndex]] = collections match {
276     case c :: cs => onCollection(c).list().flatMap(ix =>
277       listIndexes(cs, indexes ++ ix.map(NSIndex(s"${db.name}.$c", _))))
278     case _ => Future.successful(indexes)
279   }
280 
281   def list(): Future[List[NSIndex]] =
282     db.collectionNames.flatMap(listIndexes(_, Nil))
283 
284   def ensure(nsIndex: NSIndex): Future[Boolean] =
285     onCollection(nsIndex.collectionName).ensure(nsIndex.index)
286 
287   def create(nsIndex: NSIndex): Future[WriteResult] =
288     onCollection(nsIndex.collectionName).create(nsIndex.index)
289 
290   def drop(collectionName: String, indexName: String): Future[Int] =
291     onCollection(collectionName).drop(indexName)
292 
293   def dropAll(collectionName: String): Future[Int] =
294     onCollection(collectionName).dropAll()
295 
296   def onCollection(name: String): CollectionIndexesManager =
297     new DefaultCollectionIndexesManager(db, name)
298 }

395 private class DefaultCollectionIndexesManager(db: DB, collectionName: String)(
396     implicit
397     context: ExecutionContext
398 ) extends CollectionIndexesManager {
399 
400   import reactivemongo.api.commands.{
401     CreateIndexes,
402     Command,
403     ListIndexes
404   }
405   import reactivemongo.api.commands.bson.BSONListIndexesImplicits._
406   import reactivemongo.api.commands.bson.BSONCreateIndexesImplicits._
407 
408   private lazy val collection = db(collectionName)
409   private lazy val listCommand = ListIndexes(db.name)
410 
411   def list(): Future[List[Index]] =
412     Command.run(BSONSerializationPack)(collection, listCommand) recoverWith {
413       case err: WriteResult if err.code.exists(_ == 26 /* no database */ ) =>
414         Future.successful(List.empty[Index])
415 
416       case err => Future.failed(err)
417     }
418 
419   def ensure(index: Index): Future[Boolean] = for {
420     is <- list().map(_.dropWhile(_.key != index.key).headOption.isDefined)
421     cr <- if (!is) create(index).map(_ => true) else Future.successful(false)
422   } yield cr
423 
424   def create(index: Index): Future[WriteResult] =
425     Command.run(BSONSerializationPack)(
426       collection,
427       CreateIndexes(db.name, List(index))
428     )
429 
430   @deprecated("Use drop instead", "0.11.0")
431   def delete(index: Index) = drop(index.eventualName)
432 
433   @deprecated("Use drop instead", "0.11.0")
434   def delete(name: String) = drop(name)
435 
436   @deprecated("Use [[IndexesManager.drop]]", "0.11.0")
437   def drop(nsIndex: NSIndex): Future[Int] = {
438     import reactivemongo.api.commands.bson.BSONDropIndexesImplicits._
439     Command.run(BSONSerializationPack)(
440       db(nsIndex.collectionName), DropIndexes(nsIndex.index.eventualName)
441     ).
442       map(_.value)
443   }
444 
445   def drop(indexName: String): Future[Int] = {
446     import reactivemongo.api.commands.bson.BSONDropIndexesImplicits._
447     Command.run(BSONSerializationPack)(collection, DropIndexes(indexName)).
448       map(_.value)
449   }
450 
451   @inline def dropAll(): Future[Int] = drop("*")
452 }

454 /** Factory for indexes manager scoped with a specified collection. */
455 object CollectionIndexesManager {
456   /**
457    * Returns an indexes manager for specified collection.
458    *
459    * @param db the database
460    * @param collectionName the collection name
461    */
462   def apply(db: DB, collectionName: String)(implicit context: ExecutionContext): CollectionIndexesManager = {
463     val wireVer = db.connection.metadata.map(_.maxWireVersion)
464 
465     if (wireVer.exists(_ >= MongoWireVersion.V30)) {
466       new DefaultCollectionIndexesManager(db, collectionName)
467     } else new LegacyCollectionIndexesManager(db.name, collectionName,
468       new LegacyIndexesManager(db))
469   }
470 }

472 object IndexesManager {
473   import reactivemongo.util.option
474 
475   /**
476    * Returns an indexes manager for specified database.
477    *
478    * @param db the database
479    */
480   def apply(db: DB with DBMetaCommands)(implicit context: ExecutionContext): IndexesManager = {
481     val wireVer = db.connection.metadata.map(_.maxWireVersion)
482 
483     if (wireVer.exists(_ >= MongoWireVersion.V30)) new DefaultIndexesManager(db)
484     else new LegacyIndexesManager(db)
485   }
486 
487   protected def toBSONDocument(nsIndex: NSIndex) = {
488     BSONDocument(
489       "ns" -> BSONString(nsIndex.namespace),
490       "name" -> BSONString(nsIndex.index.eventualName),
491       "key" -> BSONDocument(
492         (for (kv <- nsIndex.index.key)
493         yield kv._1 -> kv._2.value).toStream
494       ),
495       "background" -> option(nsIndex.index.background, BSONBoolean(true)),
496       "dropDups" -> option(nsIndex.index.dropDups, BSONBoolean(true)),
497       "sparse" -> option(nsIndex.index.sparse, BSONBoolean(true)),
498       "unique" -> option(nsIndex.index.unique, BSONBoolean(true)),
499       "partialFilterExpression" -> nsIndex.index.partialFilter
500     ) ++ nsIndex.index.options
501   }
502 
503   implicit object NSIndexWriter extends BSONDocumentWriter[NSIndex] {
504     def write(nsIndex: NSIndex): BSONDocument = {
505       if (nsIndex.index.key.isEmpty)
506         throw new RuntimeException("the key should not be empty!")
507       toBSONDocument(nsIndex)
508     }
509   }
510 
511   implicit object IndexReader extends BSONDocumentReader[Index] {
512     def read(doc: BSONDocument): Index = doc.getAs[BSONDocument]("key").map(
513       _.elements.map { elem => elem._1 -> IndexType(elem._2) }.toList
514     ).
515       fold[Index](throw new Exception("the key must be defined")) { key =>
516         val options = doc.elements.filterNot { element =>
517           element._1 == "ns" || element._1 == "key" || element._1 == "name" ||
518             element._1 == "unique" || element._1 == "background" ||
519             element._1 == "dropDups" || element._1 == "sparse" ||
520             element._1 == "v" || element._1 == "partialFilterExpression"
521         }.toSeq
522 
523         (for {
524           name <- doc.getAsUnflattenedTry[String]("name")
525           unique <- doc.getAsUnflattenedTry[BSONBooleanLike]("unique").
526             map(_.fold(false)(_.toBoolean))
527 
528           background <- doc.getAsUnflattenedTry[BSONBooleanLike]("background").
529             map(_.fold(false)(_.toBoolean))
530 
531           dropDups <- doc.getAsUnflattenedTry[BSONBooleanLike]("dropDups").
532             map(_.fold(false)(_.toBoolean))
533 
534           sparse <- doc.getAsUnflattenedTry[BSONBooleanLike]("sparse").
535             map(_.fold(false)(_.toBoolean))
536 
537           version <- doc.getAsUnflattenedTry[BSONNumberLike]("v").
538             map(_.map(_.toInt))
539 
540           partialFilter <- doc.getAsUnflattenedTry[BSONDocument](
541             "partialFilterExpression"
542           )
543         } yield Index(key, name, unique, background, dropDups,
544           sparse, version, partialFilter, BSONDocument(options))).get
545       }
546   }
547 
548   implicit object NSIndexReader extends BSONDocumentReader[NSIndex] {
549     def read(doc: BSONDocument): NSIndex =
550       doc.getAs[BSONString]("ns").map(_.value).fold[NSIndex](
551         throw new Exception("the namespace ns must be defined")
552       )(
553           NSIndex(_, doc.as[Index])
554         )
555   }
