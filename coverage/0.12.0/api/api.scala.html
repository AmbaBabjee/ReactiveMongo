69 /**
70  * A helper that sends the given message to the given actor,
71  * following a failover strategy.
72  * This helper holds a future reference that is completed with a response,
73  * after 1 or more attempts (specified in the given strategy).
74  * If the all the tryouts configured by the given strategy were unsuccessful,
75  * the future reference is completed with a Throwable.
76  *
77  * Should not be used directly for most use cases.
78  *
79  * @tparam T Type of the message to send.
80  * @param message The message to send to the given actor. This message will be wrapped into an ExpectingResponse message by the `expectingResponseMaker` function.
81  * @param connection The reference to the MongoConnection the given message will be sent to.
82  * @param strategy The Failover strategy.
83  * @param expectingResponseMaker A function that takes a message of type `T` and wraps it into an ExpectingResponse message.
84  */
85 class Failover[T](message: T, connection: MongoConnection, strategy: FailoverStrategy)(expectingResponseMaker: T => ExpectingResponse)(implicit ec: ExecutionContext) {
86   import Failover2.logger
87   import reactivemongo.core.errors._
88   import reactivemongo.core.actors.Exceptions._
89 
90   private val promise = Promise[Response]()
91 
92   /**
93    * A future that is completed with a response,
94    * after 1 or more attempts (specified in the given strategy).
95    */
96   val future: Future[Response] = promise.future
97 
98   private def send(n: Int) {
99     val expectingResponse = expectingResponseMaker(message)
100     connection.mongosystem ! expectingResponse
101     expectingResponse.future.onComplete {
102       case Failure(e) if isRetryable(e) =>
103         if (n < strategy.retries) {
104           val `try` = n + 1
105           val delayFactor = strategy.delayFactor(`try`)
106           val delay = Duration.unapply(strategy.initialDelay * delayFactor).map(t => FiniteDuration(t._1, t._2)).getOrElse(strategy.initialDelay)
107 
108           logger.debug(s"Got an error, retrying... (try #${`try`} is scheduled in ${delay.toMillis} ms)", e)
109 
110           connection.actorSystem.scheduler.scheduleOnce(delay)(send(`try`))
111         } else {
112           // generally that means that the primary is not available or the nodeset is unreachable
113           logger.error("Got an error, no more attempts to do. Completing with a failure...", e)
114           promise.failure(e)
115         }
116 
117       case Failure(e) => {
118         logger.trace(
119           "Got an non retryable error, completing with a failure...", e
120         )
121         promise.failure(e)
122       }
123 
124       case Success(response) => {
125         logger.trace("Got a successful result, completing...")
126         promise.success(response)
127       }
128     }
129   }
130 
131   private def isRetryable(throwable: Throwable) = throwable match {
132     case e: ChannelNotFound             => e.retriable
133     case _: PrimaryUnavailableException => true
134     case _: NodeSetNotReachable         => true
135     case _: ConnectionException         => true
136     case _: ConnectionNotInitialized    => true
137     case e: DatabaseException =>
138       e.isNotAPrimaryError || e.isUnauthorized
139 
140     case _ => false
141   }
142 
143   send(0)
144 }

172       case err => Failure(err)

174       case Failure(e) if isRetryable(e) => {
175         if (n < strategy.retries) {

184         } else {

196       }

199         trace(s"[$lnm] Got a successful result, completing...")

201       }
202     }

326   /** Returns a future that will be successful when node set is available. */
327   private[api] def waitIsAvailable(failoverStrategy: FailoverStrategy)(implicit ec: ExecutionContext): Future[Unit] = {
328     logger.debug(s"[$lnm] Waiting is available...")
329 
330     @inline def nextTimeout(i: Int): FiniteDuration = {
331       val delayFactor: Double = failoverStrategy.delayFactor(i)
332 
333       Duration.unapply(failoverStrategy.initialDelay * delayFactor).
334         fold(failoverStrategy.initialDelay)(t => FiniteDuration(t._1, t._2))
335     }
336 
337     type Retry = (FiniteDuration, Throwable)
338 
339     @inline def finalErr(lastErr: Throwable): Throwable = {
340       val error = if (lastErr == null) {
341         new NodeSetNotReachable(supervisor, name, history())
342       } else lastErr
343 
344       error.setStackTrace(databaseSTE +: error.getStackTrace)
345 
346       error
347     }
348 
349     def wait(iteration: Int, attempt: Int, timeout: FiniteDuration, lastErr: Throwable = null): Future[Unit] = {
350       logger.trace(
351         s"[$lnm] Wait is available: $attempt @ ${System.currentTimeMillis}"
352       )
353 
354       if (attempt == 0) {
355         Future.failed(finalErr(lastErr))
356       } else {
357         @inline def res: Either[Retry, Unit] = try {
358           val before = System.currentTimeMillis
359           val unavail = Await.result(probe, timeout)
360           val duration = System.currentTimeMillis - before
361 
362           unavail match {
363             case Some(reason) => Left(FiniteDuration(
364               timeout.toMillis - duration, MILLISECONDS
365             ) -> reason)
366 
367             case _ => Right({})
368           }
369         } catch {
370           case e: Throwable => Left(timeout -> e)
371         }
372 
373         @inline def doRetry(delay: FiniteDuration, reason: Throwable): Future[Unit] = after(delay, actorSystem.scheduler) {
374           val nextIt = iteration + 1
375           wait(nextIt, attempt - 1, nextTimeout(nextIt), reason)
376         }
377 
378         res match {
379           case Left((delay, error)) => {
380             logger.trace(s"[$lnm] Got an error, retrying", error)
381             // TODO: Keep an explicit stacktrace accross the retries
382             // TODO: Transform error into a single StackTraceElement to add it
383 
384             doRetry(delay, error)
385           }
386 
387           case _ => Future.successful({})
388         }
389       }
390     }
391 
392     wait(0, 1 + failoverStrategy.retries, failoverStrategy.initialDelay).
393       flatMap { _ =>
394         metadata match {
395           case Some(ProtocolMetadata(
396             _, MongoWireVersion.V24AndBefore, _, _, _)) =>
397             Future.failed[Unit](ConnectionException(
398               s"unsupported MongoDB version < 2.6 ($lnm)"
399             ))
400 
401           case Some(_) => Future successful {}
402           case _ => Future.failed[Unit](ConnectionException(
403             s"protocol metadata not available ($lnm)"
404           ))
405         }
406       }
407   }

618   /**
619    * Parses a MongoURI.
620    *
621    * @param uri the connection URI (see [[http://docs.mongodb.org/manual/reference/connection-string/ the MongoDB URI documentation]] for more information)
622    */
623   def parseURI(uri: String): Try[ParsedURI] = {
624     val prefix = "mongodb://"
625 
626     Try {
627       val useful = uri.replace(prefix, "")
628       def opts = makeOptions(parseOptions(useful))
629 
630       if (opts._2.maxIdleTimeMS != 0 &&
631         opts._2.maxIdleTimeMS < opts._2.monitorRefreshMS) {
632 
633         throw new URIParsingException(s"Invalid URI options: maxIdleTimeMS(${opts._2.maxIdleTimeMS}) < monitorRefreshMS(${opts._2.monitorRefreshMS})")
634       }
635 
636       // ---
637 
638       if (useful.indexOf("@") == -1) {
639         val (db, hosts) = parseHostsAndDbName(useful)
640         val (unsupportedKeys, options) = opts
641 
642         ParsedURI(hosts, options, unsupportedKeys, db, None)
643       } else {
644         val WithAuth = """([^:]+):([^@]*)@(.+)""".r
645 
646         useful match {
647           case WithAuth(user, pass, hostsPortsAndDbName) => {
648             val (db, hosts) = parseHostsAndDbName(hostsPortsAndDbName)
649 
650             db.fold[ParsedURI](throw new URIParsingException(s"Could not parse URI '$uri': authentication information found but no database name in URI")) { database =>
651               val (unsupportedKeys, options) = opts
652 
653               ParsedURI(hosts, options, unsupportedKeys, Some(database), Some(Authenticate.apply(options.authSource.getOrElse(database), user, pass)))
654             }
655           }
656 
657           case _ => throw new URIParsingException(s"Could not parse URI '$uri'")
658         }
659       }
660     }
661   }

663   private def parseHosts(hosts: String) = hosts.split(",").toList.map { host =>
664     host.split(':').toList match {
665       case host :: port :: Nil => host -> {
666         try {
667           val p = port.toInt
668           if (p > 0 && p < 65536) p
669           else throw new URIParsingException(s"Could not parse hosts '$hosts' from URI: invalid port '$port'")
670         } catch {
671           case _: NumberFormatException => throw new URIParsingException(s"Could not parse hosts '$hosts' from URI: invalid port '$port'")
672           case NonFatal(e)              => throw e
673         }
674       }
675       case host :: Nil => host -> DefaultPort
676       case _           => throw new URIParsingException(s"Could not parse hosts from URI: invalid definition '$hosts'")
677     }
678   }
