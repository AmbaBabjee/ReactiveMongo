36 case class BSONQueryBuilder(
37   collection: Collection,
38   failover: FailoverStrategy,
39   queryOption: Option[BSONDocument] = None,
40   sortOption: Option[BSONDocument] = None,
41   projectionOption: Option[BSONDocument] = None,
42   hintOption: Option[BSONDocument] = None,
43   explainFlag: Boolean = false,
44   snapshotFlag: Boolean = false,
45   commentString: Option[String] = None,
46   options: QueryOpts = QueryOpts(),
47   maxTimeMsOption: Option[Long] = None
48 )
49     extends GenericQueryBuilder[BSONSerializationPack.type] {
50   import reactivemongo.util.option
51 
52   type Self = BSONQueryBuilder
53   val pack = BSONSerializationPack
54 
55   def copy(
56     queryOption: Option[BSONDocument] = queryOption,
57     sortOption: Option[BSONDocument] = sortOption,
58     projectionOption: Option[BSONDocument] = projectionOption,
59     hintOption: Option[BSONDocument] = hintOption,
60     explainFlag: Boolean = explainFlag,
61     snapshotFlag: Boolean = snapshotFlag,
62     commentString: Option[String] = commentString,
63     options: QueryOpts = options,
64     failover: FailoverStrategy = failover,
65     maxTimeMsOption: Option[Long] = maxTimeMsOption
66   ): BSONQueryBuilder =
67     BSONQueryBuilder(collection, failover, queryOption, sortOption, projectionOption, hintOption, explainFlag, snapshotFlag, commentString, options, maxTimeMsOption)
68 
69   def merge(readPreference: ReadPreference): BSONDocument = {
70     // Primary and SecondaryPreferred are encoded as the slaveOk flag;
71     // the others are encoded as $readPreference field.
72 
73     def pref = {
74       val mode = readPreference match {
75         case ReadPreference.Primary                    => "primary"
76         case ReadPreference.PrimaryPreferred(filter)   => "primaryPreferred"
77         case ReadPreference.Secondary(filter)          => "secondary"
78         case ReadPreference.SecondaryPreferred(filter) => "secondaryPreferred"
79         case ReadPreference.Nearest(filter)            => "nearest"
80       }
81       val base = Seq[BSONElement]("mode" -> BSONString(mode))
82 
83       BSONDocument(readPreference match {
84         case ReadPreference.Taggable(tagSet) =>
85           base :+ BSONElement("tags", BSONArray(
86             tagSet.map(tags => BSONDocument(tags.toList.map {
87               case (k, v) => k -> BSONString(v)
88             }))
89           ))
90 
91         case _ => base
92       })
93     }
94 
95     val optional = List(
96       queryOption.map { "$query" -> _ },
97       sortOption.map { "$orderby" -> _ },
98       hintOption.map { "$hint" -> _ },
99       maxTimeMsOption.map { "$maxTimeMS" -> BSONLong(_) },
100       commentString.map { "$comment" -> BSONString(_) },
101       option(explainFlag, "$explain" -> BSONBoolean(true)),
102       option(snapshotFlag, "$snapshot" -> BSONBoolean(true))
103     ).flatten
104 
105     BSONDocument(optional :+ ("$readPreference" -> pref))
106   }
107 }
