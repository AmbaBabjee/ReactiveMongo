266   def bulkInsert(documents: Stream[pack.Document], ordered: Boolean, writeConcern: WriteConcern = defaultWriteConcern, bulkSize: Int, bulkByteSize: Int)(implicit ec: ExecutionContext): Future[MultiBulkWriteResult] = watchFailure {
267     def createBulk[R, A <: BulkMaker[R, A]](docs: Stream[pack.Document], command: A with BulkMaker[R, A]): Future[List[R]] = {
268       val (tail, nc) = command.fill(docs)
269 
270       command.send().flatMap { wr =>
271         if (nc.isDefined) createBulk(tail, nc.get).map(wr2 => wr :: wr2)
272         else Future.successful(List(wr)) // done
273       }
274     }
275 
276     if (!documents.isEmpty) {
277       val havingMetadata = db.connection.metadata.
278         fold(Future.failed[ProtocolMetadata](MissingMetadata()))(Future.successful)
279 
280       havingMetadata.flatMap { metadata =>
281         if (metadata.maxWireVersion >= MongoWireVersion.V26) {
282           createBulk(documents, Mongo26WriteCommand.insert(ordered, writeConcern, metadata)).map { _.foldLeft(MultiBulkWriteResult())(_ merge _) }
283         } else {
284           // Mongo 2.4
285           Future.failed[MultiBulkWriteResult](new scala.RuntimeException(
286             s"unsupported MongoDB version: $metadata"
287           ))
288         }
289       }
290     } else Future.successful(MultiBulkWriteResult(
291       ok = true,
292       n = 0,
293       nModified = 0,
294       upserted = Seq.empty,
295       writeErrors = Seq.empty,
296       writeConcernError = None,
297       code = None,
298       errmsg = None,
299       totalN = 0
300     ))
301   }

303   /**
304    * Inserts a document into the collection and wait for the [[reactivemongo.api.commands.WriteResult]].
305    *
306    * Please read the documentation about [[reactivemongo.core.commands.GetLastError]] to know how to use it properly.
307    *
308    * @tparam T the type of the document to insert. An implicit `Writer[T]` typeclass for handling it has to be in the scope.
309    *
310    * @param document the document to insert.
311    * @param writeConcern the [[reactivemongo.core.commands.GetLastError]] command message to send in order to control how the document is inserted. Defaults to GetLastError().
312    *
313    * @return a future [[reactivemongo.api.commands.WriteResult]] that can be used to check whether the insertion was successful.
314    */
315   def insert[T](document: T, writeConcern: WriteConcern = defaultWriteConcern)(implicit writer: pack.Writer[T], ec: ExecutionContext): Future[WriteResult] =
316     db.connection.metadata match {
317       case Some(metadata) if metadata.maxWireVersion >= MongoWireVersion.V26 =>
318         Failover2(db.connection, failoverStrategy) { () =>
319           runCommand(BatchCommands.InsertCommand.Insert(
320             writeConcern = writeConcern
321           )(document)).flatMap { wr =>
322             val flattened = wr.flatten
323             if (!flattened.ok) {
324               // was ordered, with one doc => fail if has an error
325               Future.failed(WriteResult.lastError(flattened).
326                 getOrElse[Exception](GenericDriverException(
327                   s"fails to insert: $document"
328                 )))
329 
330             } else Future.successful(wr)
331           }
332         }.future
333 
334       case Some(metadata) => // Mongo < 2.6
335         Future.failed[WriteResult](new scala.RuntimeException(
336           s"unsupported MongoDB version: $metadata"
337         ))
338 
339       case _ =>
340         Future.failed(MissingMetadata())
341     }

355    * @return a future [[reactivemongo.api.commands.WriteResult]] that can be used to check whether the update was successful.

359       metadata.maxWireVersion >= MongoWireVersion.V26
360     ) => {
361       import BatchCommands.UpdateCommand.{ Update, UpdateElement }
362 
363       Failover2(db.connection, failoverStrategy) { () =>
364         runCommand(Update(writeConcern = writeConcern)(
365           UpdateElement(selector, update, upsert, multi)
366         )).flatMap { wr =>
367           val flattened = wr.flatten
368           if (!flattened.ok) {
369             // was ordered, with one doc => fail if has an error
370             Future.failed(WriteResult.lastError(flattened).
371               getOrElse[Exception](GenericDriverException(
372                 s"fails to update: $update"
373               )))
374 
375           } else Future.successful(wr)
376         }
377       }.future
378     }
379 
380     case Some(metadata) => // Mongo < 2.6
381       Future.failed[UpdateWriteResult](new scala.RuntimeException(
382         s"unsupported MongoDB version: $metadata"
383       ))
384 
385     case _ => Future.failed(MissingMetadata())
386   }
