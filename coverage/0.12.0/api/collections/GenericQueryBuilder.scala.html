16 package reactivemongo.api.collections
17 
18 import scala.concurrent.{ ExecutionContext, Future }
19 import reactivemongo.api._
20 import reactivemongo.core.protocol.{ Query, QueryFlags }
21 import reactivemongo.core.netty.{ BufferSequence, ChannelBufferWritableBuffer }
22 
23 /**
24  * A builder that helps to make a fine-tuned query to MongoDB.
25  *
26  * When the query is ready, you can call `cursor` to get a [[Cursor]], or `one` if you want to retrieve just one document.
27  *
28  * @define oneFunction Sends this query and gets a future `Option[T]` (alias for [[reactivemongo.api.Cursor.headOption]])
29  * @define readPrefParam The [[reactivemongo.api.ReadPreference]] for this query. If the `ReadPreference` implies that this query can be run on a secondary, the slaveOk flag will be set.
30  * @define readerParam the reader for the results type
31  * @define resultTParam the results type
32  * @define requireOneFunction Sends this query and gets a future `T` (alias for [[reactivemongo.api.Cursor.head]])
33  */
34 trait GenericQueryBuilder[P <: SerializationPack] {
35   val pack: P
36   type Self <: GenericQueryBuilder[pack.type]
37 
38   def queryOption: Option[pack.Document]
39   def sortOption: Option[pack.Document]
40   def projectionOption: Option[pack.Document]
41   def hintOption: Option[pack.Document]
42   def explainFlag: Boolean
43   def snapshotFlag: Boolean
44   def commentString: Option[String]
45   def options: QueryOpts
46   def failover: FailoverStrategy
47   def collection: Collection
48   def maxTimeMsOption: Option[Long]
49 
50   def merge(readPreference: ReadPreference): pack.Document
51 
52   def copy(
53     queryOption: Option[pack.Document] = queryOption,
54     sortOption: Option[pack.Document] = sortOption,
55     projectionOption: Option[pack.Document] = projectionOption,
56     hintOption: Option[pack.Document] = hintOption,
57     explainFlag: Boolean = explainFlag,
58     snapshotFlag: Boolean = snapshotFlag,
59     commentString: Option[String] = commentString,
60     options: QueryOpts = options,
61     failover: FailoverStrategy = failover,
62     maxTimeMsOption: Option[Long] = maxTimeMsOption
63   ): Self
64 
65   private def write(document: pack.Document, buffer: ChannelBufferWritableBuffer = ChannelBufferWritableBuffer()): ChannelBufferWritableBuffer = {
66     pack.writeToBuffer(buffer, document)
67     buffer
68   }
69 
70   /**
71    * Sends this query and gets a [[Cursor]] of instances of `T`.
72    */
73   @deprecated(
74     message = "Use `cursor()` or `cursor(readPreference)`",
75     since = "0.11.0"
76   )
77   def cursor[T](implicit reader: pack.Reader[T], ec: ExecutionContext, cp: CursorProducer[T]): cp.ProducedCursor = cursor(ReadPreference.primary)
78 
79   /**
80    * Makes a [[Cursor]] of this query, which can be enumerated.
81    *
82    * @param readPreference $readPrefParam
83    * @param reader $readerParam
84    *
85    * @tparam T $resultTParam
86    */
87   def cursor[T](readPreference: ReadPreference = ReadPreference.primary, isMongo26WriteOp: Boolean = false)(implicit reader: pack.Reader[T], ec: ExecutionContext, cp: CursorProducer[T]): cp.ProducedCursor = cp.produce(defaultCursor[T](readPreference, isMongo26WriteOp))
88 
89   private def defaultCursor[T](readPreference: ReadPreference, isMongo26WriteOp: Boolean = false)(implicit reader: pack.Reader[T], ec: ExecutionContext): Cursor[T] = {
90     val documents = BufferSequence {
91       val buffer = write(merge(readPreference), ChannelBufferWritableBuffer())
92       projectionOption.map { projection =>
93         write(projection, buffer)
94       }.getOrElse(buffer).buffer
95     }
96 
97     val flags = if (readPreference.slaveOk) options.flagsN | QueryFlags.SlaveOk else options.flagsN
98 
99     val op = Query(flags, collection.fullCollectionName, options.skipN, options.batchSizeN)
100 
101     DefaultCursor.query(pack, op, documents, readPreference, collection.db.connection, failover, isMongo26WriteOp)(reader)
102   }
103 
104   /**
105    * $oneFunction.
106    *
107    * @param reader $readerParam
108    *
109    * @tparam T $resultTParam
110    */
111   def one[T](implicit reader: pack.Reader[T], ec: ExecutionContext): Future[Option[T]] = one(ReadPreference.primary)
112 
113   /**
114    * $oneFunction.
115    *
116    * @param readPreference $readPrefParam
117    * @param reader $readerParam
118    *
119    * @tparam T $resultTParam
120    */
121   def one[T](readPreference: ReadPreference)(implicit reader: pack.Reader[T], ec: ExecutionContext): Future[Option[T]] = copy(options = options.batchSize(1)).defaultCursor(readPreference)(reader, ec).headOption
122 
123   /**
124    * $requireOneFunction.
125    *
126    * @param reader $readerParam
127    *
128    * @tparam T $resultTParam
129    */
130   def requireOne[T](implicit reader: pack.Reader[T], ec: ExecutionContext): Future[T] = requireOne(ReadPreference.primary)
131 
132   /**
133    * $requireOneFunction.
134    *
135    * @param readPreference $readPrefParam
136    * @param reader $readerParam
137    *
138    * @tparam T $resultTParam
139    */
140   def requireOne[T](readPreference: ReadPreference)(implicit reader: pack.Reader[T], ec: ExecutionContext): Future[T] = copy(options = options.batchSize(1)).defaultCursor(readPreference)(reader, ec).head
141 
142   /**
143    * Sets the selector document.
144    *
145    * @tparam Qry The type of the query. An implicit `Writer[Qry]` typeclass for handling it has to be in the scope.
146    */
147   def query[Qry](selector: Qry)(implicit writer: pack.Writer[Qry]): Self =
148     copy(queryOption = Some(pack.serialize(selector, writer)))
149 
150   /** Sets the query (the selector document). */
151   def query(selector: pack.Document): Self = copy(queryOption = Some(selector))
152 
153   /** Sets the sorting document. */
154   def sort(document: pack.Document): Self = copy(sortOption = Some(document))
155 
156   def options(options: QueryOpts): Self = copy(options = options)
157 
158   /**
159    * Sets the projection document (for [[http://docs.mongodb.org/manual/core/read-operations-introduction/ retrieving only a subset of fields]]).
160    *
161    * @tparam Pjn The type of the projection. An implicit `Writer[Pjn]` typeclass for handling it has to be in the scope.
162    */
163   def projection[Pjn](p: Pjn)(implicit writer: pack.Writer[Pjn]): Self =
164     copy(projectionOption = Some(pack.serialize(p, writer)))
165 
166   /**
167    * Sets the projection document (for [[http://docs.mongodb.org/manual/core/read-operations-introduction/ retrieving only a subset of fields]]).
168    */
169   def projection(p: pack.Document): Self = copy(projectionOption = Some(p))
170 
171   /** Sets the hint document (a document that declares the index MongoDB should use for this query). */
172   def hint(document: pack.Document): Self = copy(hintOption = Some(document))
173 
174   /** Sets the hint document (a document that declares the index MongoDB should use for this query). */
175   // TODO def hint(indexName: String): Self = copy(hintOption = Some(BSONDocument(indexName -> BSONInteger(1))))
176 
177   /** Toggles [[https://docs.mongodb.org/manual/reference/method/cursor.explain/#cursor.explain explain mode]]. */
178   def explain(flag: Boolean = true): Self = copy(explainFlag = flag)
179 
180   /** Toggles [[https://docs.mongodb.org/manual/faq/developers/#faq-developers-isolate-cursors snapshot mode]]. */
181   def snapshot(flag: Boolean = true): Self = copy(snapshotFlag = flag)
182 
183   /** Adds a comment to this query, that may appear in the MongoDB logs. */
184   def comment(message: String): Self = copy(commentString = Some(message))
185 
186   /** Adds maxTimeMs to query https://docs.mongodb.org/v3.0/reference/operator/meta/maxTimeMS/ */
187   def maxTimeMs(p: Long): Self = copy(maxTimeMsOption = Some(p))
188 
189 }
