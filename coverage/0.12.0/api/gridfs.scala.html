248   /** Concats two array - fast way */
249   private def concat[T](a1: Array[T], a2: Array[T])(implicit m: Manifest[T]): Array[T] = {
250     var i, j = 0
251     val result = new Array[T](a1.length + a2.length)
252     while (i < a1.length) {
253       result(i) = a1(i)
254       i = i + 1
255     }
256     while (j < a2.length) {
257       result(i + j) = a2(j)
258       j = j + 1
259     }
260     result
261   }

263   /**
264    * Gets an `Iteratee` that will consume data to put into a GridFS store.
265    * @param file Metadata of the file to store.
266    * @param chunkSize Size of the chunks. Defaults to 256kB.
267    *
268    * @tparam Id Type of the id of this file (generally `BSONObjectID` or `BSONValue`).
269    *
270    * @return An `Iteratee` that will consume data to put into a GridFS store.
271    */
272   @deprecated("Will be moved to the separate iteratee module", "0.12.0")
273   def iteratee[Id <: pack.Value](file: FileToSave[pack.type, Id], chunkSize: Int = 262144)(implicit readFileReader: pack.Reader[ReadFile[Id]], ctx: ExecutionContext, idProducer: IdProducer[Id], docWriter: BSONDocumentWriter[file.pack.Document]): Iteratee[Array[Byte], Future[ReadFile[Id]]] = {
274     implicit def ec = db.connection.actorSystem
275     import java.security.MessageDigest
276 
277     case class Chunk(
278         previous: Array[Byte] = new Array(0),
279         n: Int = 0,
280         md: MessageDigest = MessageDigest.getInstance("MD5"),
281         length: Int = 0
282     ) {
283 
284       def feed(chunk: Array[Byte]): Future[Chunk] = {
285         val wholeChunk = self.concat(previous, chunk)
286 
287         val normalizedChunkNumber = wholeChunk.length / chunkSize
288 
289         logger.debug(s"wholeChunk size is ${wholeChunk.length} => ${normalizedChunkNumber}")
290 
291         val zipped =
292           for (i <- 0 until normalizedChunkNumber)
293             yield Arrays.copyOfRange(
294             wholeChunk, i * chunkSize, (i + 1) * chunkSize
295           ) -> i
296 
297         val left = Arrays.copyOfRange(
298           wholeChunk, normalizedChunkNumber * chunkSize, wholeChunk.length
299         )
300 
301         Future.traverse(zipped) { ci =>
302           writeChunk(n + ci._2, ci._1)
303         }.map { _ =>
304           logger.debug("all futures for the last given chunk are redeemed.")
305           Chunk(
306             if (left.isEmpty) Array.empty else left,
307             n + normalizedChunkNumber,
308             md, //{ md.update(chunk) ; md },
309             length + chunk.length
310           )
311         }
312       }
313 
314       import reactivemongo.api.collections.bson.{
315         BSONCollection,
316         BSONCollectionProducer
317       }
318 
319       def finish(): Future[ReadFile[Id]] = {
320         logger.debug(s"writing last chunk (n=$n)!")
321 
322         val uploadDate = file.uploadDate.getOrElse(System.currentTimeMillis)
323 
324         writeChunk(n, previous).flatMap { f =>
325           val bson = BSONDocument(idProducer("_id" -> file.id)) ++ (
326             "filename" -> file.filename.map(BSONString(_)),
327             "chunkSize" -> BSONInteger(chunkSize),
328             "length" -> BSONLong(length),
329             "uploadDate" -> BSONDateTime(uploadDate),
330             "contentType" -> file.contentType.map(BSONString(_)),
331             "metadata" -> option(!pack.isEmpty(file.metadata), file.metadata)
332           )
333 
334           files.as[BSONCollection]().insert(bson).map { _ =>
335             val buf = ChannelBufferWritableBuffer()
336             BSONSerializationPack.writeToBuffer(buf, bson)
337             pack.readAndDeserialize(buf.toReadableBuffer, readFileReader)
338           }
339         }
340       }
341 
342       def writeChunk(n: Int, array: Array[Byte]) = {
343         logger.debug(s"writing chunk $n")
344 
345         val bson = BSONDocument(
346           "files_id" -> file.id,
347           "n" -> BSONInteger(n),
348           "data" -> BSONBinary(array, Subtype.GenericBinarySubtype)
349         )
350 
351         chunks.as[BSONCollection]().insert(bson)
352       }
353     }
354 
355     Iteratee.foldM(Chunk()) { (previous, chunk: Array[Byte]) =>
356       logger.debug(s"processing new enumerated chunk from n=${previous.n}...\n")
357       previous.feed(chunk)
358     }.map(_.finish)
359   }

361   /**
362    * Produces an enumerator of chunks of bytes from the `chunks` collection
363    * matching the given file metadata.
364    *
365    * @param file the file to be read
366    */
367   def enumerate[Id <: pack.Value](file: ReadFile[Id])(implicit ctx: ExecutionContext, idProducer: IdProducer[Id]): Enumerator[Array[Byte]] = {
368     import reactivemongo.api.collections.bson.{
369       BSONCollection,
370       BSONCollectionProducer
371     }
372 
373     def selector = BSONDocument(idProducer("files_id" -> file.id)) ++ (
374       "n" -> BSONDocument(
375         "$gte" -> 0,
376         "$lte" -> BSONLong(file.length / file.chunkSize + (
377           if (file.length % file.chunkSize > 0) 1 else 0
378         ))
379       )
380     )
381 
382     @inline def cursor = chunks.as[BSONCollection]().find(selector).
383       sort(BSONDocument("n" -> 1)).cursor[BSONDocument](defaultReadPreference)
384 
385     @inline def pushChunk(chan: Concurrent.Channel[Array[Byte]], doc: BSONDocument): Cursor.State[Unit] = doc.get("data") match {
386       case Some(BSONBinary(data, _)) => {
387         val array = new Array[Byte](data.readable)
388         data.slice(data.readable).readBytes(array)
389         Cursor.Cont(chan push array)
390       }
391 
392       case _ => {
393         val errmsg = "not a chunk! failed assertion: data field is missing: ${BSONDocument pretty doc}"
394 
395         logger.error(errmsg)
396         Cursor.Fail(ReactiveMongoException(errmsg))
397       }
398     }
399 
400     Concurrent.unicast[Array[Byte]] { chan =>
401       cursor.foldWhile({})(
402         (_, doc) => pushChunk(chan, doc),
403         Cursor.FailOnError()
404       ).onComplete { case _ => chan.eofAndEnd() }
405     }
406   }
