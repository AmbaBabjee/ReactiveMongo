29 /**
30  * A helper to make the query options.
31  * You may use the methods to set the fields of this class, or set them directly.
32  *
33  * @param skipN the number of documents to skip.
34  * @param batchSizeN the upper limit on the number of documents to retrieve per batch.
35  * @param flagsN the query flags
36  */
37 case class QueryOpts(
38     skipN: Int = 0,
39     batchSizeN: Int = 0,
40     flagsN: Int = 0
41 ) {
42 
43   /** Sets the number of documents to skip. */
44   def skip(n: Int) = copy(skipN = n)
45 
46   /** Sets an upper limit on the number of documents to retrieve per batch. Defaults to 0 (meaning no upper limit - MongoDB decides). */
47   def batchSize(n: Int) = copy(batchSizeN = n)
48 
49   /** Sets the query flags. */
50   def flags(n: Int) = copy(flagsN = n)
51 
52   /** Toggles TailableCursor: Makes the cursor not to close after all the data is consumed. */
53   def tailable = copy(flagsN = flagsN ^ QueryFlags.TailableCursor)
54 
55   /** Toggles SlaveOk: The query is might be run on a secondary. */
56   def slaveOk = copy(flagsN = flagsN ^ QueryFlags.SlaveOk)
57 
58   /** Toggles OplogReplay */
59   def oplogReplay = copy(flagsN = flagsN ^ QueryFlags.OplogReplay)
60 
61   /** Toggles NoCursorTimeout: The cursor will not expire automatically */
62   def noCursorTimeout = copy(flagsN = flagsN ^ QueryFlags.NoCursorTimeout)
63 
64   /**
65    * Toggles AwaitData: Block a little while waiting for more data instead of returning immediately if no data.
66    * Use along with TailableCursor.
67    */
68   def awaitData = copy(flagsN = flagsN ^ QueryFlags.AwaitData)
69 
70   /** Toggles Exhaust */
71   def exhaust = copy(flagsN = flagsN ^ QueryFlags.Exhaust)
72   /** Toggles Partial: The response can be partial - if a shard is down, no error will be thrown. */
73   def partial = copy(flagsN = flagsN ^ QueryFlags.Partial)
74 }
