270     private def parseUnionTypes: Option[List[c.Type]] = {
271       val unionOption = c.typeOf[Macros.Options.UnionType[_]]
272       val union = c.typeOf[Macros.Options.\/[_, _]]
273 
274       @annotation.tailrec
275       def parseUnionTree(trees: List[Type], found: List[Type]): List[Type] =
276         trees match {
277           case tree :: rem => if (tree <:< union) {
278             tree match {
279               case TypeRef(_, _, List(a, b)) =>
280                 parseUnionTree(a :: b :: rem, found)
281 
282               case _ => c.abort(c.enclosingPosition,
283                 "Union type parameters expected: $tree")
284             }
285           } else parseUnionTree(rem, tree :: found)
286 
287           case _ => found
288         }
289

291         case t @ TypeRef(_, _, lst) if t <:< unionOption =>
292           lst.headOption
293

295           types.filter(_ <:< unionOption).flatMap {
296             case TypeRef(_, _, args) => args
297           }.headOption
298 
299         case _ => None
300       }

302       tree.map { t => parseUnionTree(List(t), Nil) }

311         case Some(cls: ClassSymbol) if (
312           tpeSym != cls && cls.selfType.baseClasses.contains(tpeSym)
313         ) => {
314           val newSub: Set[Type] = if (!cls.isCaseClass) {

316             Set.empty
317           } else if (!cls.typeParams.isEmpty) {
318             c.warning(c.enclosingPosition, s"cannot handle class ${cls.fullName}: type parameter not supported")
319             Set.empty

323         }

326           o.companion == NoSymbol && // not a companion object
327             tpeSym != c && o.typeSignature.baseClasses.contains(tpeSym)
328         ) => {
329           val newSub: Set[Type] = if (!o.moduleClass.asClass.isCaseClass) {
330             c.warning(c.enclosingPosition, s"cannot handle object ${o.fullName}: no case accessor")
331             Set.empty
332           } else Set(o.typeSignature)
333 
334           allSubclasses(path.tail, subclasses ++ newSub)
335         }
336 
337         case Some(o: ModuleSymbol) if (
338           o.companion == NoSymbol // not a companion object
339         ) => allSubclasses(path.tail, subclasses)
340 
341         case Some(_) => allSubclasses(path.tail, subclasses)
342 
343         case _ => subclasses
344       }
345 
346       if (tpeSym.isSealed && tpeSym.isAbstract) {
347         Some(allSubclasses(tpeSym.owner.typeSignature.decls, Set.empty).toList)

415     /* Deep check for type compatibility */
416     @annotation.tailrec
417     private def conforms(types: Seq[(Type, Type)]): Boolean =
418       types.headOption match {
419         case Some((TypeRef(NoPrefix, a, _),
420           TypeRef(NoPrefix, b, _))) => { // for generic parameter
421           if (a.fullName != b.fullName) {
422             c.warning(c.enclosingPosition,
423               s"Type symbols are not compatible: $a != $b")
424 
425             false
426           }
427           else conforms(types.tail)
428         }
429 
430         case Some((a, b)) if (a.typeArgs.size != b.typeArgs.size) => {
431           c.warning(c.enclosingPosition,
432             s"Type parameters are not matching: $a != $b")
433 
434           false
435         }
436 
437         case Some((a, b)) if a.typeArgs.isEmpty =>
438           if (a =:= b) conforms(types.tail) else {
439             c.warning(c.enclosingPosition,
440               s"Types are not compatible: $a != $b")
441 
442             false
443           }
444 
445         case Some((a, b)) if (a.baseClasses != b.baseClasses) => {
446           c.warning(c.enclosingPosition,
447             s"Generic types are not compatible: $a != $b")
448 
449           false
450         }
451 
452         case Some((a, b)) =>
453           conforms((a.typeArgs, b.typeArgs).zipped ++: types.tail)
454 
455         case _ => true
456       }

458     private def matchingApplyUnapply(implicit tpe: Type): (MethodSymbol, MethodSymbol) = {
459       val applySymbol = applyMethod(tpe)
460       val unapply = unapplyMethod(tpe)
461       val alternatives = applySymbol.asTerm.alternatives.map(_.asMethod)
462       val u = unapplyReturnTypes(unapply)
463 
464       val applys = alternatives.filter { alt =>
465         alt.paramLists match {
466           case params :: ps if (ps.isEmpty || ps.headOption.flatMap(
467             _.headOption).exists(_.isImplicit)
468           ) => conforms((params.map(_.typeSignature), u).zipped.toSeq)
469 
470           case _ => {
471             c.warning(c.enclosingPosition, s"""Constructor with multiple parameter lists is not supported: ${tpe.typeSymbol.name}${alt.typeSignature}""")
472 
473             false
474           }
475         }
476       }
477 
478       val apply = applys.headOption.getOrElse(
479         c.abort(c.enclosingPosition, "No matching apply/unapply found"))

504     /* Refactor the input types, by replacing any type matching the `filter`,
505      * by the given `replacement`.
506      */
507     @annotation.tailrec
508     private def refactor(boundTypes: Map[String, Type])(in: List[Type], base: TypeSymbol, out: List[Type], tail: List[(List[Type], TypeSymbol, List[Type])], filter: Type => Boolean, replacement: Type, altered: Boolean): (Type, Boolean) = in match {
509       case tpe :: ts =>
510         boundTypes.getOrElse(tpe.typeSymbol.fullName, tpe) match {
511           case t if (filter(t)) =>
512             refactor(boundTypes)(ts, base, (replacement :: out), tail,
513               filter, replacement, true)
514 
515           case TypeRef(_, sym, as) if as.nonEmpty =>
516             refactor(boundTypes)(
517               as, sym.asType, List.empty, (ts, base, out) :: tail,
518               filter, replacement, altered)
519 
520           case t => refactor(boundTypes)(
521             ts, base, (t :: out), tail, filter, replacement, altered)
522         }
523 
524       case _ => {
525         val tpe = appliedType(base, out.reverse)
526 
527         tail match {
528           case (x, y, more) :: ts => refactor(boundTypes)(
529             x, y, (tpe :: more), ts, filter, replacement, altered)
530 
531           case _ => tpe -> altered
532         }
533       }
534     }

536     /**
537      * Replaces any reference to the type itself by the Placeholder type. 
538      * @return the normalized type + whether any self reference has been found
539      */
540     private def normalized(boundTypes: Map[String, Type])(tpe: Type): (Type, Boolean) = boundTypes.getOrElse(tpe.typeSymbol.fullName, tpe) match {
541         case t if (t =:= A) => PlaceholderType -> true
542 
543         case TypeRef(_, sym, args) if args.nonEmpty =>
544           refactor(boundTypes)(args, sym.asType, List.empty, List.empty,
545             _ =:= A, PlaceholderType, false)
546 
547         case t => t -> false
548       }

550     /* Restores reference to the type itself when Placeholder is found. */
551     private def denormalized(boundTypes: Map[String, Type])(ptype: Type): Type =
552       ptype match {
553         case PlaceholderType => A
554 
555         case TypeRef(_, sym, args) if args.nonEmpty =>
556           refactor(boundTypes)(args, sym.asType, List.empty, List.empty,
557             _ == PlaceholderType, A, false)._1
558 
559         case _ => ptype
560       }

562     private class ImplicitTransformer(
563       boundTypes: Map[String, Type],
564       forwardSuffix: String
565     ) extends Transformer {
566       private val denorm = denormalized(boundTypes) _
567       val forwardName = TermName(s"forward$forwardSuffix")
568 
569       override def transform(tree: Tree): Tree = tree match {
570         case tt: TypeTree =>
571           super.transform(TypeTree(denorm(tt.tpe)))
572 
573         case Select(Select(This(TypeName("Macros")), t), sym) if (
574           t.toString == "Placeholder" && sym.toString == "Handler"
575         ) => super.transform(q"$forwardName")
576 
577         case _ => super.transform(tree)
578       }
579     }

581     private def createImplicit(boundTypes: Map[String, Type])(tc: Type, ptype: Type, tx: Transformer): Implicit = {
582       val tpe = ptype match {
583         case TypeRef(_, _, targ :: _) => {
584           // Option[_] needs special treatment because we need to use XXXOpt
585           if (ptype.typeConstructor <:< typeOf[Option[_]].typeConstructor) targ
586           else ptype
587         }
588 
589         case SingleType(_, _) | TypeRef(_, _, _) => ptype
590       }
591 
592       val (ntpe, selfRef) = normalized(boundTypes)(tpe)
593       val ptpe = boundTypes.get(ntpe.typeSymbol.fullName).getOrElse(ntpe)
594 
595       // infers implicit
596       val neededImplicitType = appliedType(tc.typeConstructor, ptpe)
597       val neededImplicit = if (!selfRef) {
598         c.inferImplicitValue(neededImplicitType)
599       } else c.untypecheck(
600         // Reset the type attributes on the refactored tree for the implicit
601         tx.transform(c.inferImplicitValue(neededImplicitType))
602       )
603 
604       neededImplicit -> selfRef
605     }
