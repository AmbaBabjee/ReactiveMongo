23   /** Concats two array - fast way */
24   @deprecated(message = "Use array concat operation", since = "0.12.0")
25   def concat[T](a1: Array[T], a2: Array[T])(implicit m: Manifest[T]): Array[T] = {
26     var i, j = 0
27     val result = new Array[T](a1.length + a2.length)
28     while (i < a1.length) {
29       result(i) = a1(i)
30       i = i + 1
31     }
32     while (j < a2.length) {
33       result(i + j) = a2(j)
34       j = j + 1
35     }
36     result
37   }

72 case class EitherMappableFuture[A](future: Future[A]) {
73   def mapEither[E <: Throwable, B](f: A => Either[E, B])(implicit ec: ExecutionContext) = {
74     future.flatMap(
75       f(_) match {
76         case Left(e)  => Future.failed(e)
77         case Right(b) => Future.successful(b)
78       }
79     )
80   }
